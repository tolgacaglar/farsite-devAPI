166 {
167 strcat(Message, " dry mass density out of range (kg/m3)");

174 }

177 strcat(Message, " heat capacity out of range (J/kg/K");

200

223 // strcat(Message, " duff dry weight loading out of range (kg/m2)");
224 else if (dfm<dfm1 || dfm>dfm2)
225 strcat(Message, " duff moisture out of range (fraction)");
226
227 if (strlen(Message) > 0)
228 return false;
229
230 AllocBurnStruct(); // allocate for ntimes of output data
231
232 return true;
233}

402 long nruns = 0;

425

483 nruns = 0;

487 }
488 }
489 else
490 return false;
491 }
492 while (Repeat == true);
493
494 return true;

516 tis += dt;
517 };
518 return false;
519 }
520 SetBurnStruct(tis, now);
521 if (now > ntimes)
522 return false;
523

569//c cond = fuel ovendry thermal conductivity , W / m K
570//c chtd = fuel ovendry specific heat capacity , J / kg K
571//c fmof = fuel moisture content , fraction dry weight

576
577double BurnUp::ff(double x, double tpfi, double tpig)
578{
579 const double a03 = -1.3371565;
580 const double a13 = 0.4653628;
581 const double a23 = -0.1282064;
582 double b03;
583
584 b03 = a03 * (tpfi - tpig) / (tpfi - tamb);
585
586 return b03 + x * (a13 + x * (a23 + x));
587}

590double BurnUp::TIgnite(double tpdr, double tpig, double tpfi, double cond,
591 double chtd, double fmof, double dend, double hbar)
592{
593 const double pinv = 2.125534;
594 const double hvap = 2.177e+06;
595 const double cpm = 4186.0;
596 const double conc = 4.27e-04;
597 double xlo, xhi, xav, fav, beta, conw, dtb, dti, ratio, rhoc, tmig;
598
599
600 // MOVED TO function FF(...)
601 //------------------------------------------------------------------------------
602 //c approximate function of beta to be solved is ff( x ) where
603 //c x = 1 / ( 1 + p * beta ) { Hastings, Approximations for
604 //c digital computers } and we employ pinv = 1 / p
605 // ff( x ) = b03 + x * ( a13 + x * ( a23 + x ) )
606
607 //c radiant heating equivalent form gives end condition fixes beta
608 // b03 = a03 * ( tpfi - tpig ) / ( tpfi - tpam )
609 //------------------------------------------------------------------------------
610
611
612 //c find x that solves ff( x ) = 0 ; method is binary search
613
614 xlo = 0.0;
615 xhi = 1.0;
616
617 do
618 {
619 xav = 0.5 * (xlo + xhi);
620 fav = ff(xav, tpfi, tpig);
621 if (fabs(fav) > small)
622 {
623 if (fav < 0.0)
624 xlo = xav;
625 if (fav > 0.0)
626 xhi = xav;
627 }
628 }
629 while (fabs(fav) > small);
630
631 beta = pinv * (1.0 - xav) / xav;
632 conw = cond + conc * dend * fmof;
633 dtb = tpdr - tamb;
634 dti = tpig - tamb;
635 ratio = (hvap + cpm * dtb) / (chtd * dti);
636 rhoc = dend * chtd * (1.0 + fmof * ratio);
637 tmig = pow2(beta / hbar) * conw * rhoc;
638
639 return tmig;
640}

650void BurnUp::DuffBurn(double wdf, double dfm, double* dfi, double* tdf)
651{

653
654 double ff;
655
656 *dfi = 0.0;
657 *tdf = 0.0;

659 return;
660 *dfi = 11.25 - 4.05 * dfm;

662 *tdf = 1.e+04 * ff * wdf / (7.5 - 2.7 * dfm);
663
664 Smoldering[MAXNO] += ((ff * wdf) / (*tdf));

787 xmat[kj] = elam[k - 1][j - 1];
788 wo[kj] = wtk * xmat[kj];
789 }
790 }

800double BurnUp::TempF(double q, double r)
801{
802 // function TEMPF( q , r , tamb )
803
804 //c Returns a fire environment temperature , TEMPF , given the fire intensity
805 //c q in kW / square meter , the ambient temperature tamb in Kelvins, and the
806 //c dimensionless mixing parameter r.
807
808 const double err = 1.0e-04;
809 const double aa = 20.0;
810
811 double term, rlast, den, rnext, test, tempf;
812
813 term = r / (aa * q);
814 rlast = r;
815 do
816 {
817 den = 1.0 + term * (rlast + 1.0) * (rlast * rlast + 1.0);
818 rnext = 0.5 * (rlast + 1.0 + r / den);
819 test = fabs(rnext - rlast);
820 if (test < err)
821 {
822 tempf = rnext * tamb;
823
824 break;
825 }
826 rlast = rnext;
827 }
828 while (test >= err);
829
830 return tempf;
831}

933 double aint, tav1, tav2, tav3, tavg;

937 double dteff, heff, tfe, dtlite, qd, delt, factor;
938 double conwet, dtcum, he, dtef, thd, biot, cpwet, tbar, fac;
939 bool flag;
940
941 *ncalls += 1;
942 now = *ncalls;
943 tnow = tin;
944 tnext = tnow + dt;
945 //c tifi = time when fire ignition phase ended ( at now = 1 )
946 tifi = tnow - ((double) (now - 1)) * dt;
947 next = now + 1;
948
949 for (k = 1; k <= number; k++) //do k = 1 , number
950 {
951 c = condry[k - 1];
952 for (l = 0; l <= k; l++) //do l = 0 , k
953 {
954 kl = loc(k, l);
955 tdun = tout[kl];
956
957 //c See if k of ( k , l ) pair burned out
958
959 if (tnow >= tdun)
960 {
961 ddot[kl] = 0.0;
962 wodot[kl] = 0.0;
963
964 continue; // goto 10
965 }
966 if (tnext >= tdun)
967 {
968 tgo = tdun - tnow;
969 ddot[kl] = diam[kl] / tgo;
970 wodot[kl] = wo[kl] / tgo;
971 wo[kl] = 0.0;
972 diam[kl] = 0.0;
973
974 continue;
975 }
976
977 //c k has not yet burned out ... see if k of ( k , l ) pair is ignited
978
979 tlit = tign[kl];
980 if (tnow >= tlit)
981 {
982 ts = tchar[k - 1];
983 if (l == 0)
984 {
985 r = r0 + 0.5 * dr;
986 gi = fi + fid;
987 }
988 else if (l == k)
989 {
990 r = r0 + 0.5 * (1.0 + flit[k - 1]) * dr;
991 gi = fi + flit[k - 1] * fint[k - 1];
992 }
993 else//(l!=0 && l!=k)
994 {
995 r = r0 + 0.5 * (1.0 + flit[l - 1]) * dr;
996 gi = fi + fint[k - 1] + flit[l - 1] * fint[l - 1];
997 }
998 tf = TempF(gi, r);
999 dia = diam[kl];
1000 HeatExchange(dia, tf, ts, &hf, &hb, c, &e);
1001 qqq = hb * max(tf - ts, 0.0);
1002 tst = max(tlit, tifi);
1003 nspan = max((long) 1, Nint((tnext - tst) / dt)); //nint((tnext-tst)/dt));
1004 if (nspan <= MXSTEP)
1005 qdot[kl][nspan - 1] = qqq;
1006 else if (nspan > MXSTEP)
1007 {
1008 for (mu = 2; mu <= MXSTEP; mu++) //do mu = 2 , MXSTEP
1009 qdot[kl][mu - 2] = qdot[kl][mu - 1];
1010 qdot[kl][MXSTEP - 1] = qqq;
1011 }
1012 aint = pow2(c / hb);
1013 acum[kl] += (aint * dt);
1014 tav1 = tnext - tlit;
1015 tav2 = acum[kl] / alfa[k - 1];
1016 tav3 = pow2(dia / 4.0) / alfa[k - 1];
1017 tavg = tav1;
1018 if (tav2 < tavg)
1019 tavg = tav2;
1020 if (tav3 < tavg)
1021 tavg = tav3;
1022 index = min(nspan, MXSTEP); //index = 1+min(nspan, MXSTEP);
1023 qdsum = 0.0;
1024 tspan = 0.0;
1025 deltim = dt;
1026
1027 do
1028 {
1029 index -= 1;
1030 if (index == 0) //==1
1031 deltim = tnext - tspan - tlit;
1032 if ((tspan + deltim) >= tavg)
1033 deltim = tavg - tspan;

1035 tspan += deltim;
1036 if (tspan >= tavg)
1037 break;
1038 }
1039 while (index > 0);
1040
1041 qdavg = max(qdsum / tspan, 0.0);

1044 wnext = wo[kl] * pow2(dnext / dia);
1045 if ((dnext == 0.0) && (ddot[kl] > 0.0))
1046 tout[kl] = tnow + dia / ddot[kl];
1047 else if ((dnext > 0.0) && (dnext < dia))
1048 {
1049 rate = dia / (dia - dnext);
1050 tout[kl] = tnow + rate * dt;
1051 }
1052 if (qdavg <= (double) MXSTEP) // <=20.0 in Albini's code
1053 tout[kl] = 0.5 * (tnow + tnext);
1054 ddt = min(dt, (tout[kl] - tnow));
1055 wodot[kl] = (wo[kl] - wnext) / ddt;
1056 diam[kl] = dnext;
1057 wo[kl] = wnext;
1058
1059 continue; //goto 10
1060 }
1061
1062 //c See if k of ( k , l ) has reached outer surface drying stage yet
1063
1064 dryt = tdry[kl];
1065 if (tnow >= dryt && tnow < tlit)
1066 {
1067 if (l == 0)
1068 {
1069 r = r0;
1070 gi = fi + fid;
1071 }
1072 else if (l == k)
1073 {
1074 r = r0;
1075 gi = fi;
1076 }
1077 else// if(l!=0 && l!=k)
1078 {
1079 r = r0 + 0.5 * flit[l - 1] * dr;

1081 }

1086 //call heatx( u , d , dia , tf , ts , hf , hb , c , e )
1087 dtemp = max(0.0, tf - ts);
1088 dqdt = hb * dtemp;
1089 qcum[kl] += (dqdt * dt);
1090 tcum[kl] += (dtemp * dt);
1091 dteff = tcum[kl] / (tnext - dryt);
1092 heff = qcum[kl] / tcum[kl];
1093 tfe = ts + dteff;
1094 dtlite = rindef;
1095 if (tfe > (tpig[k - 1] + 10.0))
1096 dtlite = TIgnite(tpdry, tpig[k - 1], tfe, condry[k - 1],
1097 cheat[k - 1], fmois[k - 1], dendry[k - 1],
1098 heff);
1099 tign[kl] = 0.5 * (dryt + dtlite);
1100
1101 //c If k will ignite before time step over , must interpolate
1102
1103 if (tnext > tign[kl])
1104 {
1105 ts = tchar[k - 1];
1106 HeatExchange(dia, tf, ts, &hf, &hb, c, &e);
1107 qdot[kl][0] = hb * max(tf - ts, 0.0);
1108 qd = qdot[kl][0];
1109 ddot[kl] = qd * work[k - 1];
1110 delt = tnext - tign[kl];
1111 dnext = max(0.0, dia - delt * ddot[kl]);
1112 wnext = wo[kl] * pow2(dnext / dia);
1113 if (dnext == 0.0)
1114 tout[kl] = tnow + dia / ddot[kl];
1115 else if ((dnext > 0.0) && dnext < dia)
1116 {
1117 rate = dia / (dia - dnext);
1118 tout[kl] = tnow + rate * dt;
1119 }
1120 if (tout[kl] > tnow)
1121 {
1122 ddt = min(dt, (tout[kl] - tnow));
1123 wodot[kl] = (wo[kl] - wnext) / ddt;
1124 }
1125 else
1126 wodot[kl] = 0.0;
1127 diam[kl] = dnext;
1128 wo[kl] = wnext;
1129 }
1130
1131 continue; // goto 10
1132 }
1133
1134 //c If k of ( k , l ) still coming up to drying temperature , accumulate
1135 //c heat input and driving temperature difference , predict drying start
1136
1137 if (tnow < dryt)
1138 {
1139 factor = fmois[k - 1] * dendry[k - 1];
1140 conwet = condry[k - 1] + 4.27e-04 * factor;
1141 if (l == 0)
1142 {
1143 r = r0;
1144 gi = fi + fid;
1145 }
1146 else if (l == k)
1147 {
1148 r = r0;
1149 gi = fi;
1150 }
1151 else if ((l != 0) && (l != k))
1152 {
1153 r = r0 + 0.5 * flit[l - 1] * dr;
1154 gi = fi + flit[l - 1] * fint[l - 1];
1155 }
1156 tf = TempF(gi, r);
1157 if (tf <= (tpdry + 10.0))
1158 continue; // goto 10
1159 dia = diam[kl];
1160 ts = 0.5 * (tamb + tpdry);
1161 HeatExchange(dia, tf, ts, &hf, &hb, c, &e);
1162 dtcum = max((tf - ts) * dt, 0.0);
1163 tcum[kl] += dtcum;
1164 qcum[kl] += (hb * dtcum);
1165 he = qcum[kl] / tcum[kl];
1166 dtef = tcum[kl] / tnext;
1167 thd = (tpdry - tamb) / dtef;
1168 if (thd > 0.9)
1169 continue;
1170 biot = he * dia / conwet;
1171
1172 dryt = DryTime(biot, thd);
1173 cpwet = cheat[k - 1] + ch2o * fmois[k - 1];
1174 fac = pow2(0.5 * dia) / conwet;
1175 fac = fac * cpwet * dendry[k - 1];
1176 tdry[kl] = fac * dryt;
1177 if (tdry[kl] < tnext)
1178 {
1179 ts = tpdry;
1180 HeatExchange(dia, tf, ts, &hf, &hb, c, &e);
1181 dqdt = hb * (tf - ts);
1182 delt = tnext - tdry[kl];
1183 qcum[kl] = dqdt * delt;
1184 tcum[kl] = (tf - ts) * delt;
1185 tbar = 0.5 * (tpdry + tpig[k - 1]);
1186
1187 //c See if ignition to occur before time step complete
1188
1189 if (tf <= (tpig[k - 1] + 10.0))
1190 continue;
1191 dtlite = TIgnite(tpdry, tpig[k - 1], tf, condry[k - 1],
1192 cheat[k - 1], fmois[k - 1], dendry[k - 1], hb);
1193 tign[kl] = 0.5 * (tdry[kl] + dtlite);
1194 if (tnext > tign[kl])
1195 {
1196 ts = tchar[k - 1];
1197 qdot[kl][0] = hb * max(tf - ts, 0.0);
1198 }
1199 }
1200 }
1201 }
1202 }
1203
1204 //c Update fractions ignited and burned out , to apply at next step start
1205
1206 for (k = 1; k <= number; k++) // do k = 1 , number
1207 {
1208 flit[k - 1] = 0.0;
1209 fout[k - 1] = 0.0;
1210 for (l = 0; l <= k; l++) //do l = 0 , k
1211 {
1212 kl = loc(k, l);
1213 if (tnext >= tign[kl])
1214 flag = true;
1215 else
1216 flag = false;
1217 if (flag && tnext <= tout[kl])
1218 flit[k - 1] += xmat[kl];
1219 if (tnext > tout[kl])
1220 fout[k - 1] += xmat[kl];
1221 }
1222 }
1223}

1311bool BurnUp::Start(double dt, long now, long* ncalls)
1312{
1313 long k, l, kl, nlit;
1314 double delm, heatk, r, tf, ts, thd, tx, factor;
1315 double conwet, dia, hb, en, cpwet, fac;
1316 double hf, dryt, tsd, c, tigk, e, dtign, trt;
1317 double aint, ddt, dnext, wnext, df;
1318 const double rindef = 1.e+30;
1319
1320 //c Initialize time varying quantities and set up work( k )
1321 //c The diameter reduction rate of fuel component k is given
1322 //c by the product of the rate of heat transfer to it, per
1323 //c unit surface area, and the quantity work( k )
1324
1325 for (k = 1; k <= number; k++) //do k = 1 , number
1326 {
1327 fout[k - 1] = 0.0;
1328 flit[k - 1] = 0.0;
1329 alfa[k - 1] = condry[k - 1] / (dendry[k - 1] * cheat[k - 1]);
1330 //c effect of moisture content on burning rate (scale factor)
1331 delm = 1.67 * fmois[k - 1];
1332 //c effect of component mass density (empirical)
1333 heatk = dendry[k - 1] / 446.0;
1334 //c empirical burn rate factor, J / cu m - K
1335 heatk = heatk * 2.01e+06 * (1.0 + delm);
1336 //c normalize out driving temperature difference (Tfire - Tchar)
1337 //c to average value of lab experiments used to find above constants
1338 work[k - 1] = 1.0 / (255.0 * heatk);
1339 for (l = 0; l <= k; l++) //do l = 0 , k
1340 {
1341 kl = loc(k, l);
1342 tout[kl] = rindef;
1343 tign[kl] = rindef;
1344 tdry[kl] = rindef;
1345 tcum[kl] = 0.0;
1346 qcum[kl] = 0.0;
1347 }
1348 }
1349
1350 //c Make first estimate of drying start times for all components
1351 //c These times are usually brief and make little or no difference
1352
1353 r = r0 + 0.25 * dr;
1354 tf = TempF(fi, r);
1355 ts = tamb;
1356 if (tf <= (tpdry + 10.0))
1357 {
1358 strcat(Message, " STOP: Igniting fire cannot dry fuel");
1359
1360 return false;
1361 }
1362 thd = (tpdry - ts) / (tf - ts);
1363 tx = 0.5 * (ts + tpdry);
1364 //tpamb=tamb;
1365
1366 for (k = 1; k <= number; k++) //do k = 1 , number
1367 {
1368 factor = dendry[k - 1] * fmois[k - 1];
1369 conwet = condry[k - 1] + 4.27e-04 * factor;
1370 for (l = 0; l <= k; l++) //do l = 0 , k
1371 {
1372 kl = loc(k, l);
1373 dia = diam[kl];
1374 HeatExchange(dia, tf, tx, &hf, &hb, conwet, &en);
1375 dryt = DryTime(en, thd);
1376 cpwet = cheat[k - 1] + fmois[k - 1] * ch2o;
1377 fac = pow2(0.5 * dia) / conwet;
1378 fac = fac * dendry[k - 1] * cpwet;
1379 dryt = fac * dryt;
1380 tdry[kl] = dryt;
1381 }
1382 }
1383
1384 //c Next , determine which components are alight in spreading fire
1385
1386 tsd = tpdry;
1387
1388 for (k = 1; k <= number; k++) //do k = 1 , number
1389 {
1390 c = condry[k - 1];
1391 tigk = tpig[k - 1];
1392 for (l = 0; l <= k; l++) //do l = 0 , k
1393 {
1394 kl = loc(k, l);
1395 dryt = tdry[kl];
1396 if (dryt >= dt)
1397 continue;
1398 dia = diam[kl];
1399 ts = 0.5 * (tsd + tigk);
1400 HeatExchange(dia, tf, ts, &hf, &hb, c, &e);
1401 tcum[kl] = max((tf - ts) * (dt - dryt), 0.0);
1402 qcum[kl] = hb * tcum[kl];
1403 if (tf <= (tigk + 10.0))
1404 continue;
1405 dtign = TIgnite(tpdry, tpig[k - 1], tf, condry[k - 1],
1406 cheat[k - 1], fmois[k - 1], dendry[k - 1], hb);
1407 trt = dryt + dtign;
1408 tign[kl] = 0.5 * trt;
1409 if (dt > trt)
1410 flit[k - 1] += xmat[kl];
1411 }
1412 }
1413 nlit = 0;
1414 trt = rindef;
1415
1416 //c Determine minimum ignition time and verify ignition exists
1417
1418 for (k = 1; k <= number; k++) //do k = 1 , number
1419 {
1420 if (flit[k - 1] > 0.0)
1421 nlit += 1;
1422 for (l = 0; l <= k; l++) //do l = 0 , k
1423 {
1424 kl = loc(k, l);
1425 trt = min(trt, tign[kl]);
1426 }
1427 }
1428 if (nlit == 0)
1429 {
1430 strcat(Message, " STOP: No Fuel Ignited");
1431
1432 return false;
1433 }
1434 //c Deduct trt from all time estimates , resetting time origin
1435
1436 for (k = 1; k <= number; k++) //do k = 1 , number
1437 {
1438 for (l = 0; l <= k; l++)// do l = 0 , k
1439 {
1440 kl = loc(k, l);
1441 if (tdry[kl] < rindef)
1442 tdry[kl] -= trt;
1443 if (tign[kl] < rindef)
1444 tign[kl] -= trt;
1445 }
1446 }
1447
1448 //c Now go through all component pairs and establish burning rates
1449 //c for all the components that are ignited; extrapolate to end time dt
1450
1451 for (k = 1; k <= number; k++) //do k = 1 , number
1452 {
1453 if (flit[k - 1] == 0.0)
1454 {
1455 for (l = 0; l <= k; l++)// do l = 0 , k
1456 {
1457 kl = loc(k, l);
1458 ddot[kl] = 0.0;
1459 tout[kl] = rindef;
1460 wodot[kl] = 0.0;
1461 }
1462 }
1463 else
1464 {
1465 ts = tchar[k - 1];
1466 c = condry[k - 1];
1467 for (l = 0; l <= k; l++) //do l = 0 , k
1468 {
1469 kl = loc(k, l);
1470 dia = diam[kl];
1471 HeatExchange(dia, tf, ts, &hf, &hb, c, &e);
1472
1473 qdot[kl][now - 1] = hb * max((tf - ts), 0.0);
1474 aint = pow2(c / hb);
1475 ddt = dt - tign[kl];
1476 acum[kl] = aint * ddt;
1477 ddot[kl] = qdot[kl][now - 1] * work[k - 1];
1478 tout[kl] = dia / ddot[kl];
1479 dnext = max(0.0, (dia - ddt * ddot[kl]));
1480 wnext = wo[kl] * pow2(dnext / dia);
1481 wodot[kl] = (wo[kl] - wnext) / ddt;
1482 diam[kl] = dnext;
1483 wo[kl] = wnext;
1484 df = 0.0;
1485 if (dnext <= 0.0)
1486 {
1487 df = xmat[kl];
1488 wodot[kl] = 0.0;
1489 ddot[kl] = 0.0;
1490 }
1491 flit[k - 1] -= df;
1492 fout[k - 1] += df;
1493 }
1494 }
1495 }
1496 *ncalls = 0;
1497
1498 return true;
1499}

1513//c For elements with same size, order determined on increasing moisture
1514//c content (fmois). If items have same size and moisture content, order

1530
1531 //c Replacement sort: order on increasing size , moisture , density
1532
1533 for (j = 2; j <= number; j++) //do j = 2 , number
1534 {

1536 fm = fmois[j - 1];
1537 de = dendry[j - 1];

1544 diam = true;
1545 else
1546 diam = false;

1548 break;// goto 10
1549 if (usi == s)
1550 tied = true;

1552 tied = false;

1599void BurnUp::OverLaps()
1600{
1601 // wdry , sigma , dendry , ak , number , maxno , maxkl , beta , elam , alone , area )
1602
1603 // Computes the interaction matrix elam( j , k ) which apportions the
1604 // influence of smaller and equal size pieces on each size class for the
1605 // purpose of establishing the rates at which the elements burn out.
1606 // Input quantities are: wdry , the ovendry mass per unit area of each
1607 // element available for burning after the passage of the igniting surface
1608 // fire; sigma , the particle's surface / volume ratio , and dendry , the
1609 // ovendry mass density of the particle; ak a dimensionless parameter that
1610 // scales the planform area of a particle to its area of influence. There
1611 // are "number" separate particle classes, of a maximum number = maxno.
1612 // It is assumed that the lists are ordered on size class (nonincreasing
1613 // surface / volume ratio). List "alone" gives the fraction of each loading
1614 // that is not influenced by any other category.
1615
1616 //double wdry[MAXNO] , sigma[MAXNO] , dendry[MAXNO];
1617 //double beta[MAXKL] , elam[MAXNO][MAXNO] , alone[MAXNO];
1618 //double area[MAXNO];
1619
1620 long j, k, l, kj, kl;
1621 double a, bb, pi, siga, frac;
1622
1623
1624 pi = fabs(acos(-1.0));
1625 for (j = 1; j <= number; j++) //do j = 1 , number
1626 {
1627 alone[j - 1] = 0.0;
1628 for (k = 1; k <= j; k++)// do k = 1 , j
1629 {
1630 kj = loc(j, k);
1631 xmat[kj] = 0.0;
1632 }
1633 for (k = 1; k <= number; k++) //do k = 1 , number
1634 elam[j - 1][k - 1] = 0.0;
1635 }
1636 for (k = 1; k <= number; k++) // do k = 1 , number
1637 {
1638 for (l = 1; l <= k; l++) //do l = 1 , k
1639 {
1640 ak = 3.25 * exp(-20.0 * pow2(fmois[l - 1]));
1641 siga = ak * sigma[k - 1] / pi;
1642 kl = loc(k, l);
1643 a = siga * wdry[l - 1] / dendry[l - 1];
1644 if (k == l)
1645 {
1646 bb = 1.0 - exp(-a);
1647 if (bb < 1e-30)
1648 bb = 1e-30;
1649 area[k - 1] = bb;
1650 }
1651 else //if(k!=1)
1652 bb = min(1.0, a);
1653 xmat[kl] = bb;
1654 }
1655 }
1656 if (number == 1)
1657 {
1658 elam[0][0] = xmat[1];
1659 alone[0] = 1.0 - elam[0][0];
1660
1661 return;
1662 }
1663 for (k = 1; k <= number; k++) // do k = 1 , number
1664 {
1665 frac = 0.0;
1666 for (l = 1; l <= k; l++) // do l = 1 , k
1667 {
1668 kl = loc(k, l);
1669 frac += xmat[kl];
1670 }
1671 if (frac > 1.0)
1672 {
1673 for (l = 1; l <= k; l++) //do l = 1 , k
1674 {
1675 kl = loc(k, l);
1676 elam[k - 1][l - 1] = xmat[kl] / frac;
1677 }
1678 alone[k - 1] = 0.0;
1679 }
1680 else
1681 {
1682 for (l = 1; l <= k; l++) // do l = 1 , k
1683 {
1684 kl = loc(k, l);
1685 elam[k - 1][l - 1] = xmat[kl];
1686 }
1687 alone[k - 1] = 1.0 - frac;
1688 }
1689 }
1690}

1700
1701void BurnUp::HeatExchange(double dia, double tf, double ts, double* hfm,
1702 double* hbar, double cond, double* en)// returns en
1703{
1704 // Given horizontal windspeed u at height d [top of fuelbed], cylindrical
1705 // fuel particle diameter dia, fire environment temperature tf, and mean
1706 // surface temperature, ts, subroutine returns film heat transfer coefficient
1707 // hfm and an "effective" film heat transfer coefficient including radiation
1708 // heat transfer, hbar. Using the wood's thermal conductivity, cond, the
1709 // modified Nusselt number [ en ] used to estimate onset of surface drying
1710 // is returned as well.
1711
1712 double v, re, enuair, conair, fac, hfmin;
1713
1714 const double g = 9.8;
1715 const double vis = 7.5e-05;
1716 const double a = 8.75e-03;
1717 const double b = 5.75e-05;
1718 const double rad = 5.67e-08;
1719 const double fmfac = 0.382;
1720 const double hradf = 0.5;
1721 double hrad;
1722
1723 *hfm = 0.0;
1724 if (dia > b)
1725 {
1726 v = sqrt(u * u + 0.53 * g * d);
1727 re = v * dia / vis;
1728 enuair = 0.344 * pow(re, 0.56);
1729 conair = a + b * tf;
1730 fac = sqrt(fabs(tf - ts) / dia);
1731 hfmin = fmfac * sqrt(fac);
1732 *hfm = max((enuair * conair / dia), hfmin);
1733 }
1734 hrad = hradf * rad * (tf + ts) * (tf * tf + ts * ts);
1735 *hbar = *hfm + hrad;
1736 *en = *hbar * dia / cond;
1737}

1747double BurnUp::FireIntensity()
1748{
1749 //wodot , ash , htval , maxno , number , maxkl , area , fint , fi )
1750
1751 // Computes fi = site avg fire intensity given the burning rates of all
1752 // interacting pairs of fuel components [ wodot ] , the mineral ash content
1753 // of each component [ ash ] , the heat of combustion value [ htval ] for
1754 // each , and the number of fuel components [ number ] , where max = maxno.
1755 // fi is in kW / sq m , while htval is in J / kg.
1756
1757 // fint( k ) is the correction to fi to adjust
1758 // the intensity level to be the local value where size k is burning.
1759
1760 // real*4 ash( maxno ) , htval( maxno )
1761 // real*4 wodot( maxkl ) , area( maxno ) , fint( maxno )
1762 // data small / 1.e-06 /
1763
1764 long k, l, kl, k0;
1765 double sum, wdotk, ark, term;
1766 double wnoduff, wnoduffsum, noduffsum, test, noduffterm;//, fintnoduff[MAXNO];
1767 double fracf;
1768
1769 sum = noduffsum = wnoduffsum = 0.0;
1770 for (k = 1; k <= number; k++)
1771 {
1772 wdotk = wnoduff = 0.0;
1773 for (l = 0; l <= k; l++)
1774 {
1775 kl = loc(k, l);
1776 wdotk += wodot[kl];
1777 //if(l>0)
1778 // wnoduff+=wodot[kl];
1779 //else
1780 // Smoldering[k-1]+=wodot[kl];
1781 }
1782 term = (1.0 - ash[k - 1]) * htval[k - 1] * wdotk * 1.e-03;
1783 ark = area[k - 1];
1784 if (ark > small)
1785 fint[k - 1] = term / ark - term;
1786 else
1787 fint[k - 1] = 0.0;
1788 k0 = loc(k, 0);
1789 Smoldering[k - 1] = wodot[k0];
1790 wnoduff = wdotk - Smoldering[k - 1];
1791 noduffterm = (1.0 - ash[k - 1]) * htval[k - 1] * wnoduff * 1.e-03;
1792 if (wnoduff > 0.0)
1793 {
1794 fracf = wnoduff / wdotk;
1795 test = fracf * fint[k - 1];
1796 }
1797 else
1798 test = 0.0;
1799
1800 //--------------------------------------
1801 // flaming and smoldering decision here
1802 //--------------------------------------
1803 if (test > FintSwitch / ark - FintSwitch)
1804 Flaming[k - 1] += wnoduff; //wdotk;
1805 else
1806 Smoldering[k - 1] += wnoduff; //wdotk;
1807 //--------------------------------------
1808 //--------------------------------------
1809 sum += term;
1810 noduffsum += noduffterm;
1811 wnoduffsum += wnoduff;
1812 }
1813
1814 return sum;
1815}

1837double BurnUp::func(double h, double theta)
1838{

1840 const double b = 0.4653628;
1841 const double c = 0.1282064;
1842

1847double BurnUp::DryTime(double enu, double theta)
1848{
1849 // Given a Nusselt number ( enu , actually Biot number = h D / k )
1850 // and the dimensionless temperature rise required for the start
1851 // of surface drying ( theta ), returns the dimensionless time ( tau )
1852 // needed to achieve it. The time is given multiplied by thermal
1853 // diffusivity and divided by radius squared. Solution by binary search.
1854
1855 long n;
1856 double tau;
1857 double x, xl, xh, xm;
1858 const double p = 0.47047;
1859
1860 xl = 0.0;
1861 xh = 1.0;
1862 for (n = 0; n < 15; n++)
1863 {
1864 xm = 0.5 * (xl + xh);
1865 if (func(xm, theta) < 0.0)
1866 xl = xm;
1867 else
1868 xh = xm;
1869 }
1870 x = (1.0 / xm - 1.0) / p;
1871 tau = pow2(0.5 * x / enu);
1872
1873 return tau;
1874}
