583static int Fms_CreateParameters(Fms* fms)
584{
585 int i;
586 double a2, rcav, ro, ri, vwt;
587
588 /* Create new nodal parameters. */
589 // if ( ( fms->x = (double *) malloc( fms->n * sizeof(double)) ) == NULL
590 // || ( fms->d = (double *) malloc( fms->n * sizeof(double)) ) == NULL
591 // || ( fms->t = (double *) malloc( fms->n * sizeof(double)) ) == NULL
592 // || ( fms->s = (double *) malloc( fms->n * sizeof(double)) ) == NULL
593 // || ( fms->w = (double *) malloc( fms->n * sizeof(double)) ) == NULL
594 // || ( fms->v = (double *) malloc( fms->n * sizeof(double)) ) == NULL )
595 fms->x = new double[fms->n];
596 fms->d = new double[fms->n];
597 fms->t = new double[fms->n];
598 fms->s = new double[fms->n];
599 fms->w = new double[fms->n];
600 fms->v = new double[fms->n];
601
602 if (fms->x == NULL ||
603 fms->d == NULL ||
604 fms->t == NULL ||
605 fms->s == NULL ||
606 fms->w == NULL ||
607 fms->v == NULL)
608 {
609 Fms_Destroy(fms);
610 return(-1);
611 }
612
613 /* dx = internodal distance (cm). */
614 fms->dx = fms->a / (fms->n - 1);
615
616 /* wmax = maximum possible stick moisture content (g/g). */
617 fms->wmax = (1. / fms->dp) - (1. / 1.53);
618
619 /* Stick nodal values. */
620 for (i = 0; i < fms->n; i++)
621 {
622 /* x = radial distance from center of stick (cm). */
623 fms->x[i] = fms->a - (fms->dx * i);
624 /* d = bound water diffusivity (cm2/h). */
625 fms->d[i] = 0.;
626 /* t = temperature (oC). */
627 fms->t[i] = 20.;
628 /* s = saturation (g/g). */
629 fms->s[i] = 0.5 * fms->wmx;
630 /* w = water content (g/g). */
631 fms->w[i] = 0.5 * fms->wmx;
632 /* v = fraction of stick volume at this node (g3/g3) */
633 fms->v[i] = 0.0;
634 }
635
636 /* Volume weighting factors. */
637 ro = fms->a;
638 ri = ro - 0.5 * fms->dx;
639 a2 = fms->a * fms->a;
640 fms->v[0] = (ro * ro - ri * ri) / a2;
641 vwt = fms->v[0];
642 for (i = 1; i < fms->n - 1; i++)
643 {
644 ro = ri;
645 ri = ro - fms->dx;
646 fms->v[i] = (ro * ro - ri * ri) / a2;
647 vwt += fms->v[i];
648 }
649 fms->v[fms->n - 1] = ri * ri / a2;
650 vwt += fms->v[fms->n - 1];
651
652 /* Mark as uninitialized (in terms of weather). */
653 fms->init = 0;
654 fms->updates = 0;
655 fms->state = FMS_STATE_None;
656
657 /*------------------------------------------------------------------------*/
658 /* Stick property-dependent optimization factors. */
659 /*------------------------------------------------------------------------*/
660
661 /* hwf = hw and aml computation factor used in Fms_Update(). */
662 fms->hwf = 0.622 * fms->hc * pow((Pr / Sc), 0.667);
663
664 /* amlf = aml optimization factor */
665 fms->amlf = fms->hwf / (0.24 * fms->dp * fms->a);
666
667 /* capf = cap optimization factor. */
668 rcav = 0.5 * Aw * Wl;
669 fms->capf = 3600. * Pi * St * rcav * rcav /
670 (16. * fms->a * fms->a * fms->al * fms->dp);
671
672 /* sf = s factor used in Fms_Update() loop. */
673 fms->sf = 3600. * fms->mdt / (2. * fms->dx * fms->dp);
674
675 /* Vf = optimization factor used in Fms_Update(). */
676 /* WAS: fms->Vf = St / (Aw * Wl * Scr); */
677 /* WAS: fms->Vf = St / (Wl * Scr); */
678 fms->vf = St / (fms->dp * Wl * Scr);
679
680 /* Some silly little time shavers. */
681 fms->dx_2 = fms->dx * 2.;
682 fms->mdt_2 = fms->mdt * 2.;
683
684 return(0);
685}

731static void Fms_Diffusivity(Fms* fms)
732{
733 int i;
734 double c1; /* Emc sorption isotherm parameter (g/g) */
735 double c2; /* Emc sorption isotherm parameter (g/g) */
736 double con; /* Correction for tortuous paths in cell wall */
737 double cpv; /* Specific heat of water vapor (cal/(mol*K)) */
738 double daw; /* Density of adsorbed water (g/cm3) */
739 double dhdm; /* Reciprocal slope of the sorption isotherm */
740 double dv;
741 double dvpr;
742 double e; /* Activation energy for bound water diffusion (cal/mol) */
743 double fac; /* Converts D from wood substance to whole wood basis */
744 double ps1; /* Water satur vapor press at surface temp (cal/cm3) */
745 double qv; /* Latent heat of vaporization of water (cal/mol) */
746 double qw; /* Differential heat of sorption ofwater (cal/mol) */
747 double rfcw; /* Converts D from wood substance to whole wood basis */
748 double svaw; /* Specific volume of adsorbed water (cm3/g) */
749 double tk; /* Stick temperature (oK) */
750 double vfaw; /* Volume fraction of adborbed water (dl) */
751 double vfcw; /* Volume fraction of moist cell wall (dl) */
752 double wc; /* Lesser of nodal or fiber saturation moisture (g/g) */
753
754 for (i = 0; i < fms->n; i++)
755 {
756 tk = fms->t[i] + 273.2;
757 qv = 13550. - 10.22 * tk;
758 cpv = 7.22 + .002374 * tk + 2.67e-07 * tk * tk;
759 /* Sea level atmospheric pressure = 0.0242 cal/cm3 */
760 dv = 0.22 * 3600. * (0.0242 / fms->bp) * pow((tk / 273.2), 1.75);
761 ps1 = 0.0000239 * exp(20.58 - (5205. / tk));
762 c1 = 0.1617 - 0.001419 * fms->t[i];
763 c2 = 0.4657 + 0.003578 * fms->t[i];
764 if (fms->w[i] < fms->wsa)
765 {
766 wc = fms->w[i];
767 dhdm = (1.0 - fms->hf) * pow(-log(1.0 - fms->hf), (1.0 - c2)) /
768 (c1 * c2);
769 }
770 else
771 {
772 wc = fms->wsa;
773 dhdm = (1.0 - Hfs) * pow(Wsf, (1.0 - c2)) / (c1 * c2);
774 }
775 daw = 1.3 - 0.64 * wc;
776 svaw = 1. / daw;
777 vfaw = svaw * wc / (0.685 + svaw * wc);
778 vfcw = (0.685 + svaw * wc) / ((1.0 / fms->dp) + svaw * wc);
779 rfcw = 1.0 - sqrt(1.0 - vfcw);
780 fac = 1.0 / (rfcw * vfcw);
781 con = 1.0 / (2.0 - vfaw);
782 qw = 5040. * exp(-14.0 * wc);
783 e = (qv + qw - cpv * tk) / 1.2;
784
785 /*------------------------------------------------------------------------*/
786 /* The factor 0.016 below is a correction for hindered water vapor */
787 /* diffusion (this is 62.5 times smaller than the bulk vapor diffusion); */
788 /* 0.0242 cal/cm3 = sea level atm pressure */
789 /* -- Ralph Nelson */
790 /*------------------------------------------------------------------------*/
791
792 dvpr = 18.0 * 0.016 * (1.0 - vfcw) * dv * ps1 * dhdm /
793 (fms->dp * 1.987 * tk);
794 fms->d[i] = dvpr +
795 3600. * 0.0985 * con * fac * exp(-e / (1.987 * tk));
796 }
797 return;
798}

1103 double hf_log; /* -log(1. - fms->hf) (g/g) */
1104 double hr; /* longwave radiative surface heat transfer (cal/cm2-h-C) */
1105 double hw; /* stick heat transfer coeff for vapor diffusion above FSP*/
1106 double rai; /* rainfall runoff factor ((g/g)/h) */
1107 double p1; /* water vapor press at the stick surface (cal/cm3) */
1108 double pa; /* water vapor pressure in air (cal/cm3) */
1109 double par; /* dummy variable for s (dl) */
1110 double ps1; /* water satur vapor press at surface temp (cal/cm3) */
1111 double psa; /* water satur vapor press in ambient air (cal/cm3) */
1112 double psd; /* water satur vapor press at dewpoint (cal/cm3) */
1113 double qv; /* latent heat of vaporization of water (cal/mol) */
1114 double qw; /* differential heat of sorption of water (cal/mol) */
1115 double sr; /* solar rad received by half the stick (cal/(cm2*h) */
1116 double sv; /* current solar radiation (millivolts) */
1117 double tfd; /* intermediate stick surface temperature (oC) */
1118 double tfract; /* fraction of time elapsed between updates (dl) */
1119 double ta; /* air temperature interpolated between obs (oC) */
1120 double tdp; /* dewpoint temperature (oC) */
1121 double tdw; /* dewpoint temperature (oK) */
1122 double tka; /* air temperature (oK) */
1123 double tkf; /* stick temperature (oK) */
1124 double tsk; /* sky temperature (oK) */
1125 double tt; /* elapsed moisture computation time (h) */
1126 double oldw; /* previous value of w[0] (g/g) */
1127 double wdiff; /* maximum minus current fiber saturation (g/g) */
1128 double wold[FMS_MAX_NODES]; /* moisture content at previous time step */
1129 double sold[FMS_MAX_NODES]; /* saturation at previous time step */
1130 double told[FMS_MAX_NODES]; /* temperature at previous time step (oC) */
1131 double g[FMS_MAX_NODES]; /* free water transport coefficient (cm2/h) */
1132 double o[FMS_MAX_NODES]; /* used to redistribute moisture content */
1133 double v[FMS_MAX_NODES]; /* used to redistribute fuel temperature */
1134 double ae, ap, ar, aw; /* interpolation parameters */
1135 int i, less, do_gnu;
1136
1137 /* Increment update counter. */
1138 fms->updates++;
1139
1140 /* Catch bad data here. */
1141 if (rcumCm < fms->ra1)
1142 {
1143 fprintf(stderr, "****** Update %ld has bad rcumCom = %f\n",
1144 fms->updates, rcumCm);
1145 /* Assume a RAWS station reset. */
1146 fms->rc1 = rcumCm;
1147 fms->ra0 = 0.;
1148 return;
1149 }
1150 if (hFtn <0.001 || hFtn> 1.0)
1151 {
1152 fprintf(stderr, "****** Update %ld has bad hFtn = %f\n", fms->updates,
1153 hFtn);
1154 return;
1155 }
1156 if (aC <-40. || aC> 50.)
1157 {
1158 fprintf(stderr, "****** Update %ld has bad aC = %f\n", fms->updates,
1159 aC);
1160 return;
1161 }
1162 if (sW <0.0 || sW> 2000.)
1163 {
1164 fprintf(stderr, "****** Update %ld has bad sW = %f\n", fms->updates,
1165 sW);
1166 return;
1167 }
1168
1169 /* First save the previous weather observation values. */
1170 fms->ta0 = fms->ta1; /* Previous air temperature (oC). */
1171 fms->ha0 = fms->ha1; /* Previous air relative humidity (g/g). */
1172 fms->sv0 = fms->sv1; /* Previous pyranometer voltage (millivolts). */
1173 fms->rc0 = fms->rc1; /* Previous cumulative rainfall (cm). */
1174 fms->ra0 = fms->ra1; /* Previous period's rainfall amount (cm). */
1175
1176 /* Then save the current weather observation values. */
1177 fms->ta1 = aC; /* Current air temperature (oC). */
1178 fms->ha1 = hFtn; /* Current air relative humidity (g/g). */
1179 fms->sv1 = sW / Smv; /* Current pyranometer voltage (millivolts). */
1180 fms->rc1 = rcumCm; /* Current cumulative rainfall (cm). */
1181
1182 /* Precipitation amount since last observation adjusted by Pi (cm). */
1183 fms->ra1 = (fms->rc1 - fms->rc0) / Pi;
1184
1185 /* If the elapsed time < 0.00027 (<0.01 sec), then treat this as */
1186 /* a duplicate or corrected observation and return. */
1187 if (eHr < 0.0000027)
1188 {
1189 fprintf(stderr, "****** Update %ld has bad eHr = %f\n", fms->updates,
1190 eHr);
1191 return;
1192 }
1193
1194 /* rai = factor related to runoff during rain (g/(g-h)). */
1195 /* Its value depends on whether this is the first rainfall observation, */
1196 /* but does not vary within an observation, so its calculated here. */
1197 if (fms->ra1 > 0.00001 && fms->ra0 < 0.00001)
1198 {
1199 rai = fms->rai0 * (1.0 - exp(-100. * fms->ra1));
1200 if (fms->ha1 < fms->ha0)
1201 {
1202 rai = 0.15 * rai;
1203 }
1204 }
1205 else
1206 {
1207 rai = fms->rai1 * fms->ra1 / eHr;
1208 }
1209 rai *= fms->mdt;
1210
1211 /* ddtNext = next time (tt) to run diffusivity computations. */
1212 ddtNext = fms->ddt;
1213
1214 /* Loop for each moisture time step between environmental inputs. */
1215 for (tt = fms->mdt; tt <= eHr + fms->mdt; tt += fms->mdt)
1216 {
1217 /* Interpolate environmental values between old and current obs. */
1218 tfract = tt / eHr;
1219 ta = fms->ta0 + (fms->ta1 - fms->ta0) * tfract;
1220 ha = fms->ha0 + (fms->ha1 - fms->ha0) * tfract;
1221 sv = fms->sv0 + (fms->sv1 - fms->sv0) * tfract;
1222
1223 /* fsc = fraction of the solar constant. */
1224 fsc = 0.07 * sv;
1225
1226 /* tka = ambient air temperature (oK). */
1227 tka = ta + Kelvin;
1228
1229 /* tdw = dew point temperature (oK). */
1230 tdw = 5205. / ((5205. / tka) - log(ha));
1231
1232 /* tdp = dew point temperature (oC). */
1233 tdp = tdw - Kelvin;
1234
1235 /* Day- and night-time parameters. */
1236 /* tsk = sky temperature (oK). */
1237 /* hr = longwave radiative surface heat transfer coeff (cal/cm2-h-C) */
1238 /* sr = solar radiation received by half the stick (cal/cm2-h). */
1239 if (fsc < 0.000001)
1240 {
1241 tsk = Tcn + Kelvin;
1242 hr = Hrn;
1243 sr = 0;
1244 }
1245 else
1246 {
1247 tsk = Tcd + Kelvin;
1248 hr = Hrd;
1249 sr = Srf * fsc;
1250 }
1251
1252 /* psa = water saturation vapor pressure in ambient air (cal/cm3). */
1253 psa = 0.0000239 * exp(20.58 - (5205. / tka));
1254
1255 /* pa = water vapor pressure in air (cal/cm3). */
1256 pa = ha * psa;
1257
1258 /*--------------------------------------------------------------------*/
1259 /* Compute stick temperature and surface humidity. */
1260 /*--------------------------------------------------------------------*/
1261
1262 /* tfd = intermediate stick surface temperature (oC). */
1263 tfd = ta + (sr - hr * (ta - tsk + Kelvin)) / (hr + fms->hc);
1264
1265 /* qv = latent heat of vaporization of water (cal/mole). */
1266 qv = 13550. - 10.22 * (tfd + Kelvin);
1267
1268 /* qw = differential heat of sorption of water (cal/mole). */
1269 qw = 5040. * exp(-14. * fms->w[0]);
1270
1271 /* hw = stick heat transfer coeff for vapor diffusion above FSP. */
1272 hw = (fms->hwf * Ap / 0.24) * qv / 18.;
1273
1274 /* fms->t[0] = stick surface temperature (oC). */
1275 fms->t[0] = tfd - (hw * (tfd - ta) / (hr + fms->hc + hw));
1276 tkf = fms->t[0] + Kelvin;
1277
1278 /* c1 = EMC sorption isotherm parameter (g/g). */
1279 c1 = 0.1617 - 0.001419 * fms->t[0];
1280
1281 /* c2 = EMC sorption isotherm parameter (g/g). */
1282 c2 = 0.4657 + 0.003578 * fms->t[0];
1283
1284 /* wsa = stick fiber saturation point (g/g). */
1285 fms->wsa = c1 * pow(Wsf, c2);
1286 wdiff = fms->wmax - fms->wsa;
1287
1288 /* ps1 = water saturation vapor pressure at surface temp (cal/cm3). */
1289 ps1 = 0.0000239 * exp(20.58 - (5205. / tkf));
1290
1291 /* p1 = water vapor pressure at stick surface (cal/cm3). */
1292 p1 = pa + Ap * fms->bp * (qv / (qv + qw)) * (tka - tkf);
1293 if (p1 < 0.000001)
1294 {
1295 p1 = 0.000001;
1296 }
1297
1298 /* fms->hf = stick surface humidity (g/g). */
1299 if ((fms->hf = p1 / ps1) > Hfs)
1300 {
1301 fms->hf = Hfs;
1302 }
1303
1304 /* fms->sem = stick equilibrium moisture content (g/g). */
1305 hf_log = -log(1. - fms->hf);
1306 fms->sem = c1 * pow(hf_log, c2);
1307
1308 /*--------------------------------------------------------------------*/
1309 /* Compute surface moisture content. */
1310 /*--------------------------------------------------------------------*/
1311
1312 fms->state = FMS_STATE_None;
1313
1314 /*......1: if it is raining: */
1315 if (fms->ra1 > 0.0)
1316 {
1317 fms->t[0] = tfd;
1318 fms->hf = Hfs;
1319
1320 /*..........1a: If this is a rainstorm: */
1321 if (fms->ra1 >= fms->stv)
1322 {
1323 fms->state = FMS_STATE_Rainstorm;
1324 fms->wfilm = fms->wfilmk;
1325 fms->w[0] = fms->wmx;
1326 }
1327
1328 /*..........1b: Else this is NOT a rainstorm: */
1329 else
1330 {
1331 fms->state = FMS_STATE_Rainfall;
1332 fms->wfilm = 0.0;
1333 fms->w[0] += rai;
1334 if (fms->w[0] > fms->wmx)
1335 {
1336 fms->w[0] = fms->wmx;
1337 }
1338 fms->s[0] = (fms->w[0] - fms->wsa) / wdiff;
1339 if (fms->s[0] < 0.0)
1340 {
1341 fms->s[0] = 0.0;
1342 }
1343 }
1344 }
1345
1346 /*......2: else it is not raining; */
1347 else
1348 {
1349 fms->wfilm = 0.;
1350
1351 /*.........2a: If moisture content exceeds the fiber saturation point: */
1352 if (fms->w[0] > fms->wsa)
1353 {
1354 p1 = ps1;
1355 fms->hf = Hfs;
1356
1357 /* psd = water saturation vapor pressure at dewpt (cal/cm3). */
1358 psd = 0.0000239 * exp(20.58 - (5205. / tdw));
1359
1360 /* aml = factor related to rate of evap or conden ((g/g)/h). */
1361 if (fms->t[0] <= tdp && p1 > psd)
1362 {
1363 aml = 0.;
1364 }
1365 else
1366 {
1367 aml = fms->amlf * (ps1 - psd) / fms->bp;
1368 }
1369
1370 oldw = fms->w[0];
1371 fms->w[0] = fms->w[0] - aml * fms->mdt_2;
1372 if (aml > 0.)
1373 {
1374 gnu = 0.00439 + 0.00000177 * pow((338.76 - tkf), 2.1237);
1375 fms->w[0] = fms->w[0] - fms->mdt * fms->capf / gnu;
1376 }
1377
1378 if (fms->w[0] > fms->wmx)
1379 {
1380 fms->w[0] = fms->wmx;
1381 }
1382
1383 /*..............2a1: if moisture content is rising: condensation */
1384 if (fms->w[0] > oldw)
1385 {
1386 fms->state = FMS_STATE_Condensation;
1387 fms->s[0] = (fms->w[0] - fms->wsa) / wdiff;
1388 }
1389
1390 /*..............2a2: else if moisture content is steady: stagnation */
1391 else if (fms->w[0] == oldw)
1392 {
1393 fms->state = FMS_STATE_Stagnation;
1394 }
1395
1396 /*..............2a3: else if moisture content is falling: evaporation */
1397 else if (fms->w[0] < oldw)
1398 {
1399 fms->state = FMS_STATE_Evaporation;
1400 if ((fms->s[0] = (fms->w[0] - fms->wsa) / wdiff) < 0.)
1401 {
1402 fms->s[0] = 0.;
1403 }
1404 }
1405 }
1406 /*..........2b: else if fuel temperature is less than dewpoint: condensation */
1407 else if (fms->t[0] <= tdp)
1408 {
1409 fms->state = FMS_STATE_Condensation;
1410
1411 /* psd = water saturation vapor pressure at dewpt (cal/cm3). */
1412 psd = 0.0000239 * exp(20.58 - (5205. / tdw));
1413 if (p1 > psd)
1414 {
1415 aml = 0.0;
1416 }
1417 else
1418 {
1419 aml = fms->amlf * (p1 - psd) / fms->bp;
1420 }
1421 fms->w[0] = fms->w[0] - aml * fms->mdt_2;
1422 if ((fms->s[0] = (fms->w[0] - fms->wsa) / wdiff) < 0.)
1423 {
1424 fms->s[0] = 0.;
1425 }
1426 }
1427 /*..........2c: else surface moisture content less than fiber saturation pt */
1428 /* and stick temperature greater than dewpoint ... */
1429 else
1430 {
1431 /* fms->sem computations were moved out of here back to the */
1432 /* top of the function so its available regardless of state. */
1433
1434 if (fms->w[0] != fms->sem)
1435 {
1436 /*..............2c1: if surface moisture greater than equilibrium: desorption */
1437 if (fms->w[0] > fms->sem)
1438 {
1439 fms->state = FMS_STATE_Desorption;
1440 bi = fms->stcd * fms->dx / fms->d[0];
1441 }
1442 /*..............2c2: else surface moisture less than equilibrium: adsorption */
1443 else
1444 {
1445 fms->state = FMS_STATE_Adsorption;
1446 bi = fms->stca * fms->dx / fms->d[0];
1447 }
1448 fms->w[0] = (fms->w[1] + bi * fms->sem) / (1. + bi);
1449 fms->s[0] = 0.;
1450 }
1451 else
1452 {
1453 }
1454 }
1455 } /* end of not raining */
1456
1457 /*--------------------------------------------------------------------*/
1458 /* Compute interior nodal moisture content values. */
1459 /*--------------------------------------------------------------------*/
1460
1461 for (i = 0; i < fms->n; i++)
1462 {
1463 wold[i] = fms->w[i];
1464 sold[i] = fms->s[i];
1465 told[i] = fms->t[i];
1466 v[i] = Thdiff * fms->x[i];
1467 o[i] = 0.0;
1468 g[i] = 0.0;
1469 }
1470
1471 /* Skip the rest if in stagnation state. */
1472 if (fms->state != FMS_STATE_Stagnation)
1473 {
1474 do_gnu = 1; /* do gnu computation once per mdt iteration */
1475
1476 for (i = 0; i < fms->n; i++)
1477 {
1478 // par = dummy parameter for fms->s[i]
1479 par = (fms->w[i] - fms->wsa) / wdiff;
1480 fms->s[i] = 0.0;
1481 if (par > 0.)
1482 {
1483 fms->s[i] = par;
1484 if (par > Sir && par < Scr)
1485 {
1486 /* ak = permeability of stick under nonsaturated */
1487 /* conditions (cm2). */
1488 ak = Aks * (2. * sqrt(par / Scr) - 1.);
1489
1490 /* gnu = kinematic viscosity of liquid water (cm2/s). */
1491 if (do_gnu == 1) /* do this just one time */
1492 {
1493 gnu = 0.00439 + 0.00000177 * pow((338.76 - tkf),
1494 2.1237);
1495 do_gnu = 0; /* don't do this again */
1496 }
1497
1498 /* g = free water transport coefficient (cm2/h) */
1499 /* for calculating new s[]. */
1500 g[i] = (ak / (gnu * wdiff)) * fms->x[i] * fms->vf * pow((Scr /
1501 fms->s[i]),
1502 1.5) ;
1503 }
1504 } /* end if par > 0. */
1505 o[i] = fms->x[i] * fms->d[i];
1506 }
1507
1508 // Propagate saturation changes
1509 for (i = 1; i < fms->n - 1; i++)
1510 {
1511 ae = g[i + 1] / fms->dx;
1512 aw = g[i - 1] / fms->dx;
1513 ar = fms->x[i] * fms->dx / fms->mdt;
1514 ap = ae + aw + ar;
1515 fms->s[i] = (ae * sold[i + 1] +
1516 aw * sold[i - 1] +
1517 ar * sold[i]) /
1518 ap;
1519 if (fms->s[i] > 1.)
1520 {
1521 fms->s[i] = 1.;
1522 }
1523 }
1524 fms->s[fms->n - 1] = fms->s[fms->n - 2];
1525
1526 /* Check if fms->s[] is less than Sir (limit of */
1527 /* continuous liquid columns) at ANY stick node. */
1528 for (less = 0, i = 1; i < fms->n - 1; i++)
1529 {
1530 if (fms->s[i] < Sir)
1531 {
1532 less = 1;
1533 break;
1534 }
1535 }
1536
1537 /* If all nodes have continuous liquid columns (s >= Sir): */
1538 if (less == 0)
1539 {
1540 for (i = 1; i < fms->n - 1; i++)
1541 {
1542 fms->w[i] = fms->wsa + fms->s[i] * wdiff;
1543 if (fms->w[i] > fms->wmx)
1544 {
1545 fms->w[i] = fms->wmx;
1546 }
1547 }
1548 }
1549 /* Else at least one node has s < Sir: */
1550 else
1551 {
1552 // Propagate moisture changes
1553 for (i = 1; i < fms->n - 1; i++)
1554 {
1555 ae = o[i + 1] / fms->dx;
1556 aw = o[i - 1] / fms->dx;
1557 ar = fms->x[i] * fms->dx / fms->mdt;
1558 ap = ae + aw + ar;
1559 fms->w[i] = (ae * wold[i + 1] +
1560 aw * wold[i - 1] +
1561 ar * wold[i]) /
1562 ap;
1563 if (fms->w[i] > fms->wmx)
1564 {
1565 fms->w[i] = fms->wmx;
1566 }
1567 }
1568 }
1569 fms->w[fms->n - 1] = fms->w[fms->n - 2];
1570 }
1571
1572 // Propagate temperature changes
1573 for (i = 1; i < fms->n - 1; i++)
1574 {
1575 ae = v[i + 1] / fms->dx;
1576 aw = v[i - 1] / fms->dx;
1577 ar = fms->x[i] * fms->dx / fms->mdt;
1578 ap = ae + aw + ar;
1579 fms->t[i] = (ae * told[i + 1] + aw * told[i - 1] + ar * told[i]) /
1580 ap;
1581 if (fms->t[i] > 71.)
1582 {
1583 fms->t[i] = 71.;
1584 }
1585 }
1586 fms->t[fms->n - 1] = fms->t[fms->n - 2];
1587
1588 /* Update moisture diffusivity if within less than half a time step. */
1589 if ((ddtNext - tt) < (0.5 * fms->mdt))
1590 {
1591 Fms_Diffusivity(fms);
1592 ddtNext += fms->ddt;
1593 }
1594 } /* Next observation */
1595
1596 return;
1597}
