50void Vectorize::ArcFileFormat()
51{
52 otpfile = fopen(GetVectorFileName(), "a");
53 if (otpfile == NULL)
54 return;
55
56 if (CurFire > -1)
57 {
58 double firstx, firsty;
59 double days = 0, hours = 0, mins;
60
61 I = GetNumPoints(CurFire);
62 if (t >= 1440)
63 {
64 days = t / 1440.0;
65 t -= (((long) days) * 1440);
66 }
67 if (t >= 60)
68 {
69 hours = t / 60.0;
70 t -= (((long) hours) * 60);
71 }
72 mins = (long) t;
73 fprintf(otpfile, "%02ld%02ld%02ld%05ld\n", (long) days, (long) hours,
74 (long) mins, CurFire + 1); // time as polygon label
75 // fprintf(otpfile, "%06ld%05ld\n", ((long) t), CurFire+1); // time as polygon label
76 i = 0;
77 GetXY();
78 fprintf(otpfile, "%13.6lf %13.6lf\n", x, y); // XY coordinates
79 firstx = x; firsty = y;
80 for (i = 1; i < I; i++)
81 {
82 GetXY();
83 fprintf(otpfile, "%13.6lf %13.6lf\n", x, y); // XY coordinates
84 }
85 fprintf(otpfile, "%13.6lf %13.6lf\n", firstx, firsty); // XY coordinates
86 fprintf(otpfile, "%s\n", "END"); // end of polygon block
87 }
88 else
89 fprintf(otpfile, "%s\n", "END"); // end of file
90 fclose(otpfile);
91}

109void Vectorize::GetXY()
110{
111 x = ConvertEastingOffsetToUtm(GetPerimeter1Value(CurFire, i, XCOORD));
112 y = ConvertNorthingOffsetToUtm(GetPerimeter1Value(CurFire, i, YCOORD));
113 if (GetFileOutputOptions(RAST_FIREDIRECTION))
114 {
115 double xl, yl, xn, yn;
116 long j, k;
117 j = i - 1;
118 k = i + 1;
119 if (j < 0)
120 j = GetNumPoints(CurFire) - 1;
121 if (k > GetNumPoints(CurFire))
122 k = 0;
123 xl = ConvertEastingOffsetToUtm(GetPerimeter1Value(CurFire, j, XCOORD));
124 yl = ConvertNorthingOffsetToUtm(GetPerimeter1Value(CurFire, j, YCOORD));
125 xn = ConvertEastingOffsetToUtm(GetPerimeter1Value(CurFire, k, XCOORD));
126 yn = ConvertNorthingOffsetToUtm(GetPerimeter1Value(CurFire, k, YCOORD));
127 if (pow2(xn - xl) + pow2(yn - yl) < 1.0e-6)
128 d = 0;
129 else
130 d = (long) (360 * atan2(-(xn - xl), (yn - yl)) / (2.0 * PI));
131 if (d < 0)
132 d += 360;
133 d -= 90;
134 if (d < 0)
135 d += 360;
136 }
137}
