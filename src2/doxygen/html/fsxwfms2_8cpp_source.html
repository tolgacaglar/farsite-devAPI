70long Chrono(double SIMTIME, long* hour, double* hours, bool RelCondit)
71 // Determines actual date and time from SIMTIME variable
72{
73 long date, month, day, min;
74 double mins, days;
75
76 *hours = SIMTIME / 60.0;
77 *hour = (long) * hours; // truncates time to nearest hour
78 min = (long) ((*hours - *hour) * 60);
79 mins = (*hours - *hour) * 60.0;
80 if (RelCondit && UseConditioningPeriod(GETVAL))
81 *hours = ((double) * hour) * 100.0; // elapsed time in hours
82 else
83 *hours = ((double) * hour) * 100.0 +
84 (double) GetStartHour() +
85 GetStartMin(); // elapsed time in hours
86 days = *hours / 2400.0;
87 day = (long) days;
88 if (RelCondit && UseConditioningPeriod(GETVAL))
89 {
90 date = day + GetConditDay();
91 month = GetJulianDays(GetConditMonth());
92 }
93 else
94 {
95 date = day + GetStartDay();
96 month = GetJulianDays(GetStartMonth());
97 }
98 date = date + month;
99 if (date > 365)
100 date -= 365;
101 days = day;
102 *hour = (long) (*hours - days * 2400) + min; // integer hour
103 *hours = (*hours - days * 2400.0) + mins; // double precision hours
104
105 return date;
106}

308void FELocalSite::windadj(long date, double hours, long* cloud)
309{
310 // FINDS MOST CURRENT WINDSPD AND CALCULATES MIDFLAME WINDSPDS FROM 20-FT WS (MPH)
311
312 int count, month, xmonth, xday, day, hhour = 0, xdate;
313 double MinTimeStep;
314
315 //count = WindLoc - 1;
316 count=WindLoc[StationNumber]-1;
317 do
318 {
319 count++;
320 month = GetWindMonth(StationNumber, count);
321 day = GetWindDay(StationNumber, count);
322 if (month < 13)
323 xmonth = GetJulianDays(month);
324 else
325 {
326 day = 0;
327 xmonth = date;
328 }
329 xdate = xmonth + day;
330 }
331 while (xdate < date);
332
333 if (month != 13) // if hit end of windspeed data
334 {
335 xday = day;
336 hhour = GetWindHour(StationNumber, count);
337 while (hours >= (double) hhour)
338 {
339 count++;
340 day = GetWindDay(StationNumber, count);
341 xmonth = GetWindMonth(StationNumber, count);
342 hhour = GetWindHour(StationNumber, count);
343 if (day > xday || xmonth > month)
344 break;
345 }
346 }
347 if (hours <= hhour)
348 {
349 MinTimeStep = (double) hhour - hours;
350 EventMinimumTimeStep(MinTimeStep); // load into global variable
351 }
352 count--;
353 twindspd = GetWindSpeed(StationNumber, count);
354 wwinddir = GetWindDir(StationNumber, count);
355 *cloud = GetWindCloud(StationNumber, count);
356 WindLoc[StationNumber] = count;
357 //twindspd=(double) wss;
358 windreduct(); // GET MIDFLAME WINDSPEED
359}

365 short twspd, wndir, cloud_s;

376 //printf("date:%d-%d\n",ddate,curdate);

379 if (month != -1) // if hit end of ATMDATA data
380 {
381 hhour = GetAtmosphereGrid()->GetAtmHour(count);
382 xday = day;
383 while (hours >= hhour)

387 xday = GetAtmosphereGrid()->GetAtmDay(count);

404 }

406 {
407 MinTimeStep = (double) hhour - hours;
408 EventMinimumTimeStep(MinTimeStep); // load into global variable

413void FELocalSite::windreduct()
414{
415 // FUNCTION TO REDUCE WINDSPEED (MI/HR) TO MIDFLAME OR VEGETATION HEIGHT
416
417 double ffactor, htfuel, htflame, m1, m2;
418 double canopyht = ld.height*3.2808;//GetDefaultCrownHeight()*3.28; // convert to feet
419
420 if (ld.cover <= 5 || canopyht == 0) // ld.cover==0
421 {
422 if (ld.fuel > 0)
423 {
424 switch (ld.fuel)
425 {
426 case 1:
427 htfuel = 1.0; break;
428 case 2:
429 htfuel = 1.0; break;
430 case 3:
431 htfuel = 2.5; break;
432 case 4:
433 htfuel = 6.0; break;
434 case 5:
435 htfuel = 2.0; break;
436 case 6:
437 htfuel = 2.5; break;
438 case 7:
439 htfuel = 2.5; break;
440 case 8:
441 htfuel = 0.2; break;
442 case 9:
443 htfuel = 0.2; break;
444 case 10:
445 htfuel = 1.0; break;
446 case 11:
447 htfuel = 1.0; break;
448 case 12:
449 htfuel = 2.3; break;
450 case 13:
451 htfuel = 3.0; break;
452 default:
453 htfuel = GetFuelDepth(ld.fuel); // retrieve from cust models
454 if (htfuel == 0.0)
455 htfuel = 0.01;
456 break;
457 }
458 }
459 else
460 htfuel = 0.01; // no fuel so height is essentially zero
461 htflame = htfuel; // from Baughman and Albini 6th conf. FFM 1980
462 m1 = (1.0 + 0.36 * (htfuel / htflame)) /
463 (log((20.0 + 0.36 * htfuel) / (0.13 * htfuel)));
464 m2 = log(((htflame / htfuel + 0.36) / .13)) - 1.0;
465 mwindspd = m1 * m2 * twindspd;
466 }
467 else
468 {
469 ffactor = ((double) ld.cover / 100.0) * 0.33333; // volume ratio of cone to cylinder
470 ffactor *= PI / 4.0; // area ratio of circle to square
471
472 m1 = 0.555 /
473 (sqrt(canopyht * ffactor) * log((20.0 + 0.36 * canopyht) /
474 (0.13 * canopyht)));
475 if (m1 > 1.0)
476 m1 = 1.0;
477 mwindspd = m1 * twindspd;
478 }
479}
