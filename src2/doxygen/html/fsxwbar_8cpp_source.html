99bool VectorBarrier::ReDiscretizeBarrier()
100{
101 long i, VertCount, NumNewVertices;
102 double xdiff, ydiff, testdist;
103 double xpt1, xpt2, ypt1, ypt2, newx, newy;
104 double MaxDist, MaxDistSquared;
105
106 MaxDist = GetDistRes() * MetricResolutionConvert();
107 MaxDistSquared = MaxDist * MaxDist;
108
109 do
110 {
111 NumNewVertices = 0;
112 VertCount = 0;
113
114 if ((BufBarrier = (double *) malloc( NumVertices * 4 * sizeof(double))) ==
115 NULL)
116 return false;
117 memset(BufBarrier,0x0,NumVertices*4*sizeof(double));
118 xpt1 = Barrier[0];
119 ypt1 = Barrier[1];
120 for (i = 1; i < NumVertices; i++)
121 {
122 xpt2 = Barrier[i * 2];
123 ypt2 = Barrier[i * 2 + 1];
124 xdiff = xpt1 - xpt2;
125 ydiff = ypt1 - ypt2;
126 testdist = xdiff * xdiff + ydiff * ydiff;
127 if (testdist > MaxDistSquared)
128 {
129 BufBarrier[VertCount * 2] = xpt1;
130 BufBarrier[VertCount * 2 + 1] = ypt1;
131 VertCount++;
132 newx = xpt1 - xdiff / 2.0;
133 newy = ypt1 - ydiff / 2.0;
134 BufBarrier[VertCount * 2] = newx;
135 BufBarrier[VertCount * 2 + 1] = newy;
136 VertCount++;
137 NumNewVertices++;
138 }
139 else
140 {
141 BufBarrier[VertCount * 2] = xpt1;
142 BufBarrier[VertCount * 2 + 1] = ypt1;
143 VertCount++;
144 }
145 xpt1 = xpt2;
146 ypt1 = ypt2;
147 }
148 BufBarrier[VertCount * 2] = xpt1;
149 BufBarrier[VertCount * 2 + 1] = ypt1;
150 VertCount++;
151 free(Barrier);
152 Barrier = 0;
153 AllocBarrier(VertCount);
154 for (i = 0; i < VertCount; i++)
155 SetBarrierVertex(i, BufBarrier[i * 2], BufBarrier[i * 2 + 1]);
156 free(BufBarrier);
157 }
158 while (NumNewVertices > 0);
159
160 BufBarrier = (double *) malloc( NumVertices * 4 * sizeof(double));
161 memset(BufBarrier,0x0,NumVertices * 4 * sizeof(double));
162
163
164 return true;
165}

168void VectorBarrier::BufferBarrier(double DistResMult)
169{
170 long i, j;
171 double xpt, ypt, xptl, yptl, xptn, yptn;
172 double xdiffl, xdiffn, ydiffl, ydiffn, tempx, tempy;
173 double xdiff, ydiff, dist, distl, distn;
174 double xbuf1, ybuf1;
175 double xbuf2, ybuf2;
176 double A1, A2, A3, DR;
177
178 ReDiscretizeBarrier();
179
180 DistanceResolution = GetDistRes() * DistResMult /
181 2.0 * MetricResolutionConvert();
182 xpt = xptl = Barrier[0];
183 ypt = yptl = Barrier[1];
184 xptn = Barrier[2];
185 yptn = Barrier[3];
186
187 for (i = 1; i <= NumVertices; i++)
188 {
189 xdiffl = xpt - xptl;
190 ydiffl = ypt - yptl;
191 xdiffn = xpt - xptn;
192 ydiffn = ypt - yptn;
193 distl = sqrt(xdiffl * xdiffl + ydiffl * ydiffl);
194 distn = sqrt(xdiffn * xdiffn + ydiffn * ydiffn);
195
196 if (distl > 0.0 && distn > 0.0)
197 {
198 if (distl < distn)
199 {
200 tempx = xpt - xdiffn * distl / distn;
201 tempy = ypt - ydiffn * distl / distn;
202 xdiff = xptl - tempx;
203 ydiff = yptl - tempy;
204 }
205 else
206 {
207 if (distn < distl)
208 {
209 tempx = xpt - xdiffl * distn / distl;
210 tempy = ypt - ydiffl * distn / distl;
211 xdiff = tempx - xptn;
212 ydiff = tempy - yptn;
213 }
214 else
215 {
216 xdiff = xptl - xptn;
217 ydiff = yptl - yptn;
218 }
219 }
220 }
221 else
222 {
223 xdiff = xptl - xptn;
224 ydiff = yptl - yptn;
225 }
226
227 dist = sqrt(xdiff * xdiff + ydiff * ydiff);
228 if (fabs(xdiffl) < 1e-9 && fabs(ydiffl) < 1e-9)
229 A1 = 0.0;
230 else
231 A1 = atan2((xdiffl), (ydiffl));
232 if (fabs(xdiff) < 1e-9 && fabs(ydiff) < 1e-9)
233 A2 = 0.0;
234 else
235 A2 = atan2((xdiff), (ydiff));
236 A3 = cos(A1) * cos(A2) + sin(A1) * sin(A2);
237 if (fabs(A3) > 1e-2 && distl > 1e-9)
238 DR = DistanceResolution / fabs(A3);
239 else
240 DR = DistanceResolution;
241 if (dist == 0.0)
242 dist = 1.0;
243 xbuf1 = xpt - DR / dist * ydiff; // perpendicular to xpt,ypt
244 ybuf1 = ypt + DR / dist * xdiff;
245 xbuf2 = xpt + DR / dist * ydiff; // perpendicular to xpt,ypt
246 ybuf2 = ypt - DR / dist * xdiff;
247
248 //else
249 //{ xbuf1=xbuf2=xpt;
250 // ybuf1=ypt+DistanceResolution;
251 // ybuf2=ypt-DistanceResolution;
252 //}
253 BufBarrier[(i - 1) * 2] = xbuf2;
254 BufBarrier[(i - 1) * 2 + 1] = ybuf2;
255 BufBarrier[(NumVertices * 2 - i) * 2] = xbuf1;
256 BufBarrier[(NumVertices * 2 - i) * 2 + 1] = ybuf1;
257
258 if (i < NumVertices - 1)
259 j = i + 1;
260 else
261 j = NumVertices - 1;
262 xptl = xpt;
263 yptl = ypt;
264 xpt = xptn;
265 ypt = yptn;
266 xptn = Barrier[j * 2];
267 yptn = Barrier[j * 2 + 1];
268 }
269}

316 k = 0;

333 ydiff = yptl - tempy;

358 if (fabs(xdiff) > 0.0)
