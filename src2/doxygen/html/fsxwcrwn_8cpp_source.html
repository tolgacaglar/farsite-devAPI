87void Crown::CrownBurn(double avgros, double fli, double AccelConstant)
88{
89 // uses original crown fire spread calculations from Finney 1998
90 double a;
91
92 SFC=fli/(300.0*avgros); // calculate surface fuel consumption from Van Wagner 1990, kg/m2;
93 CritRos=3.0/CrownBulkDensity;
94
95 Ro=Io/(300.0*SFC);
96 if(CritRos>Ro)
97 { a=-log(0.1)/(0.9*(CritRos-Ro));
98 CrownFractionBurned=1.0-exp(-a*(fli-Io)*(avgros/fli));
99 if(CrownFractionBurned<=0.0) // debugging
100 CrownFractionBurned=0.01;
101 else if(CrownFractionBurned>1.0)
102 CrownFractionBurned=1.0;
103 }
104 else
105 CrownFractionBurned=1.0;
106
107 //CrownFractionBurned=1.0-exp(-.238*((fli-Io)/(300.0*SFC))); // Van Wagner 1990,
108 CrownLoadingBurned=CrownFractionBurned*CrownBulkDensity*fabs(TreeHeight-HLCB); // calc loading (kg/m2)
109 A=AccelConstant-18.8*pow2(CrownFractionBurned)*sqrt(CrownFractionBurned)*exp(-8.0*CrownFractionBurned); // CFFBPS 1992, equation 72, page 41, subst .3 for .115
110 //A=AccelConstant-18.8*pow(CrownFractionBurned,2.5)*exp(-8.0*CrownFractionBurned); // CFFBPS 1992, equation 72, page 41, subst .3 for .115
111}

114double Crown::CrownBurn2(double avgros, double fli, double AccelConstant, void *vt)
115{
116 // uses crown fire calculations from Scott and Reinhardt 2001
117 double R10, Rratio, Eratio, Ractual, Rsa;
118 double cw1, cw2, cs, cpw1, cpw2, X, inc, CritWind;
119 double sw1, sw2, ss, spw1, spw2;
120 BurnThread *bt;
121
122 bt=(BurnThread *) vt;
123
124
125 SFC=fli/(300.0*avgros); // calculate surface fuel consumption from Van Wagner 1990, kg/m2;
126 CritRos=3.0/CrownBulkDensity;
127
128 // get surface fire behavior parameters
129 bt->GetEquationTerms(&sw1, &ss, &spw1, &spw2);
130 Eratio=avgros/(bt->fros*(1.0+sw1+ss)); // ratio of actual to heading spread rate of surface fire
131
132 // now calculate crown fire behavior
133 Ro=Io/(300.0*SFC); // class variable
134 R10=bt->spreadrate(bt->ld.slope, bt->m_twindspd*0.4, 10);
135 if(Ro>CritRos)
136 CrownFractionBurned=1.0;
137 else
138 { bt->GetEquationTerms(&cw1, &cs, &cpw1, &cpw2);
139 Ractual=3.34*R10*(1.0+cs+cw1); // with wind and slope
140 Rratio=(Ractual*Eratio)/CritRos;
141 if(Rratio<1.0)
142 { cw2=cw1;
143 inc=1.0;
144 do
145 { X=Rratio*(1.0+cs+cw2)-(1.0+cs); // iterate to solve for phiW
146 if(X>cw1)
147 { cw2-=inc;
148 inc/=2.0;
149 }
150 cw2+=inc;
151 }while(fabs(cw1-X)>1e-6);
152
153 CritWind=pow(cw2/cpw2, 1.0/cpw1);
154 CritWind/=0.4;
155 if(bt->m_twindspd<1e-6)
156 sw2=0.0;
157 else
158 sw2=pow(CritWind*bt->m_windspd/bt->m_twindspd, spw1)*spw2;
159 Rsa=bt->fros*(1.0+ss+sw2);
160 if(Rsa>CritRos)
161 Rsa=CritRos;
162 CrownFractionBurned=(avgros-Ro)/(Rsa-Ro); // maxcrownfraction burned in heading fire
163 if(CrownFractionBurned>1.0)
164 CrownFractionBurned=1.0;
165 }
166 else
167 CrownFractionBurned=1.0;
168 }
169
170 CrownLoadingBurned=CrownFractionBurned*CrownBulkDensity*fabs(TreeHeight-HLCB); // calc loading (kg/m2)
171 A=AccelConstant-18.8*pow2(CrownFractionBurned)*sqrt(CrownFractionBurned)*exp(-8.0*CrownFractionBurned); // CFFBPS 1992, equation 72, page 41, subst .3 for .115
172
173 return R10;
174}
