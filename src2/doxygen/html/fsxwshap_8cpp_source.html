118long ShapeFileUtils::ImportShapeData(const char* FileName, long PolygonType)
119{
120 SHPHandle hSHP;
121 SHPObject* pSHP;
122 int nShapeType, nEntities;
123 long i, j, k, start, end;
124 double xpt, ypt, fli;
125 double maxx, maxy, minx, miny;
126 double MinBound[4], MaxBound[4];
127
128 hSHP = SHPOpen(FileName, "rb");
129 if (hSHP == NULL)
130 return false;
131
132 SHPGetInfo(hSHP, &nEntities, &nShapeType, MinBound, MaxBound);
133 if (nShapeType == SHPT_POINT ||
134 nShapeType == SHPT_POINTZ ||
135 nShapeType == SHPT_POINTM ||
136 nShapeType == SHPT_MULTIPOINT ||
137 nShapeType == SHPT_MULTIPOINTZ ||
138 nShapeType == SHPT_MULTIPOINTM)
139 {
140 SHPClose(hSHP);
141
142 return -1;
143 }
144 if (PolygonType < 3)
145 fli = 0.0;
146 else
147 fli = -1.0;
148 for (i = 0; i < nEntities; i++)
149 {
150 pSHP = SHPReadObject(hSHP, i);
151 if (pSHP->nVertices > 0)
152 {
153 for (k = 0; k < pSHP->nParts; k++)
154 {
155 start = pSHP->panPartStart[k];
156 if (k < pSHP->nParts - 1)
157 end = pSHP->panPartStart[k + 1];
158 else
159 end = pSHP->nVertices;
160 if (end - start <= 0)
161 continue;
162 AllocPerimeter1(GetNewFires(), (end - start) + 1);
163 for (j = start; j < end; j++)
164 {
165 xpt = ConvertUtmToEastingOffset(pSHP->padfX[j]);
166 ypt = ConvertUtmToNorthingOffset(pSHP->padfY[j]);
167 SetPerimeter1(GetNewFires(), j - start, xpt, ypt);
168 SetFireChx(GetNewFires(), j - start, -1.0, fli);
169 if (j == start)
170 {
171 minx = maxx = xpt;
172 miny = maxy = ypt;
173 }
174 else
175 {
176 if (xpt < minx)
177 minx = xpt;
178 else if (xpt > maxx)
179 maxx = xpt;
180 if (ypt < miny)
181 miny = ypt;
182 else if (ypt > maxy)
183 maxy = ypt;
184 }
185 }
186 SetPerimeter1(GetNewFires(), j - start, minx, maxx);
187 SetFireChx(GetNewFires(), j - start, miny, maxy);
188 SetInout(GetNewFires(), PolygonType);
189 SetNumPoints(GetNewFires(), end - start);//pSHP->nVertices);
190 IncNewFires(1);
191 }
192 }
193 SHPDestroyObject(pSHP);
194 }
195 SHPClose(hSHP);
196
197 if (nShapeType == SHPT_POLYGON ||
198 nShapeType == SHPT_POLYGONZ ||
199 nShapeType == SHPT_POLYGONM ||
200 nShapeType == SHPT_MULTIPATCH)
201 i = 0;
202 else
203 i = 1;
204
205 return i;
206}

262bool ShapeFileUtils::ShapeData(long CurrentFire, double SimTime)
263{
264 int NumVertices, NumParts;
265 //int *PartsArray;
266 long InOut, count, NumRecord;
267 long NumVertex;
268 char DataBaseID[64] = "";
269 char Description[256];
270 double xpt, ypt;//, ros;
271 double* VertexX = 0, * VertexY = 0, * VertexZ = 0;
272 SHPHandle hSHP;
273 SHPObject* pSHP;
274 DBFHandle hDBF;
275
276 InOut = GetInout(CurrentFire);
277 if (InOut == 0)
278 return true;
279
280 if (NewFile)
281 {
282 // create new shapefile
283 if (BarExport)
284 {
285 if (Polygons == 1)
286 BarExport = false;
287 }
288 else
289 RetrieveExportFileInfo();
290 if (Polygons)
291 nShapeType = SHPT_POLYGON;
292 else
293 nShapeType = SHPT_ARC;
294 hSHP = SHPCreate(ShapeFileName, nShapeType);
295 if (hSHP == NULL)
296 return false;
297 SHPClose(hSHP);
298 // Create the database.
299 hDBF = DBFCreate(DataBaseName);
300 if (hDBF == NULL)
301 return false;
302 sprintf(DataBaseID, "%s", "Fire_Type");
303 DBFAddField(hDBF, DataBaseID, FTString, 64, 0);
304 sprintf(DataBaseID, "%s", "Month");
305 DBFAddField(hDBF, DataBaseID, FTInteger, 32, 0);
306 sprintf(DataBaseID, "%s", "Day");
307 DBFAddField(hDBF, DataBaseID, FTInteger, 32, 0);
308 sprintf(DataBaseID, "%s", "Hour");
309 DBFAddField(hDBF, DataBaseID, FTInteger, 32, 0);
310 sprintf(DataBaseID, "%s", "Elapsed_Minutes");
311 DBFAddField(hDBF, DataBaseID, FTDouble, 32, 4);
312 DBFClose(hDBF);
313 NewFile = false;
314 }
315 hSHP = SHPOpen(ShapeFileName, "rb+");
316 hDBF = DBFOpen(DataBaseName, "rb+");
317 NumVertex = GetNumPoints(CurrentFire);
318 if (BarExport)
319 {
320 NumVertex /= 2;
321 NumVertices = NumVertex;
322 }
323 else
324 NumVertices = NumVertex + 1;
325 VertexX = new double[NumVertex + 1];
326 VertexY = new double[NumVertex + 1];
327 //VertexZ=new double[NumVertex+1];
328 NumParts = 0;
329 //PartsArray=new int[2];
330 //PartsArray[0]=0;
331 for (count = 0; count < NumVertex; count++)
332 {
333 if (!BarExport)
334 {
335 xpt = GetPerimeter1Value(CurrentFire, count, XCOORD);
336 ypt = GetPerimeter1Value(CurrentFire, count, YCOORD);
337 //ros=GetPerimeter1Value(CurrentFire, count, ROSVAL);
338 }
339 else
340 {
341 xpt = GetPerimeter2Value(count, XCOORD);
342 ypt = GetPerimeter2Value(count, YCOORD);
343 //ros=0.0;
344 }
345 VertexX[count] = ConvertEastingOffsetToUtm(xpt);
346 VertexY[count] = ConvertNorthingOffsetToUtm(ypt);
347 //VertexZ[count]=ros;
348 }
349 if (!BarExport) // don't repeat first point on barrier line
350 {
351 xpt = GetPerimeter1Value(CurrentFire, 0, XCOORD);
352 ypt = GetPerimeter1Value(CurrentFire, 0, YCOORD);
353 //ros=GetPerimeter1Value(CurrentFire, 0, ROSVAL);
354 VertexX[count] = ConvertEastingOffsetToUtm(xpt);
355 VertexY[count] = ConvertNorthingOffsetToUtm(ypt);
356 //VertexZ[count]=ros;
357 }
358 pSHP = SHPCreateObject(nShapeType, -1, NumParts, NULL, NULL, NumVertices,
359 VertexX, VertexY, VertexZ, NULL);
360 SHPWriteObject(hSHP, -1, pSHP);
361 SHPDestroyObject(pSHP);
362
363 NumRecord = DBFGetRecordCount(hDBF);
364
365 if (VertexX)
366 delete[] VertexX;
367 if (VertexY)
368 delete[] VertexY;
369 if (VertexZ)
370 delete[] VertexZ;
371 //if(PartsArray)
372 // delete[] PartsArray;
373 switch (InOut)
374 {
375 case 1:
376 sprintf(Description, "%s", "Expanding Fire");
377 break;
378 case 2:
379 sprintf(Description, "%s", "Enclave Fire");
380 break;
381 case 3:
382 sprintf(Description, "%s", "Barrier");
383 break;
384 default:
385 memset(Description, 0x0, sizeof(Description));
386 break;
387 }
388 ConvertSimTime(SimTime);
389 DBFWriteStringAttribute(hDBF, NumRecord, 0, Description);
390 DBFWriteIntegerAttribute(hDBF, NumRecord, 1, Month);
391 DBFWriteIntegerAttribute(hDBF, NumRecord, 2, Day);
392 DBFWriteIntegerAttribute(hDBF, NumRecord, 3, Hour);
393 DBFWriteDoubleAttribute(hDBF, NumRecord, 4, SimTime);
394 SHPClose(hSHP);
395 DBFClose(hDBF);
396
397 return true;
398}

401void ShapeFileUtils::ConvertSimTime(double SimTime)
402{
403 double Hr, Dy;
404 long min;
405
406 Month = GetStartMonth();
407 Day = 0;
408 Hour = 0;
409
410 if (SimTime >= 1440.0)
411 {
412 Dy = SimTime / 1440.0;
413 Day = (long) Dy;
414 SimTime -= (1440 * Day); // truncate to nearest day
415 }
416 if (SimTime >= 60.0)
417 {
418 Hr = SimTime / 60.0;
419 Hour = (long) Hr;
420 SimTime -= (60 * Hour); // truncate to nearesat hour
421 }
422 min = (long) SimTime;//*60.0); // minutes left over
423
424 Hour += (GetStartHour() / 100);
425 if (Hour >= 24)
426 {
427 Hour -= 24;
428 Day++;
429 }
430 Day += GetStartDay();
431 long days, oldMonth;
432 do
433 {
434 oldMonth = Month;
435 switch (Month)
436 {
437 case 1:
438 days = 31; break; // days in each Monthnth, ignoring leap year
439 case 2:
440 days = 28; break;
441 case 3:
442 days = 31; break;
443 case 4:
444 days = 30; break;
445 case 5:
446 days = 31; break;
447 case 6:
448 days = 30; break;
449 case 7:
450 days = 31; break;
451 case 8:
452 days = 31; break;
453 case 9:
454 days = 30; break;
455 case 10:
456 days = 31; break;
457 case 11:
458 days = 30; break;
459 case 12:
460 days = 31; break;
461 }
462 if (Day > days)
463 {
464 Day -= days;
465 Month++;
466 if (Month > 12)
467 Month -= 12;
468 }
469 }
470 while (Month != oldMonth); // allows startup of current clock at any time, will find cur month
471
472 Hour *= 100;
473 Hour += min;
474}
