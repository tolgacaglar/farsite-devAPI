61long FECalls::GetLandscapeData(double xpt, double ypt, LandscapeStruct& ls)
62{
63 if (xpt<GetLoEast() ||
64 xpt>GetHiEast() ||
65 ypt<GetLoNorth() ||
66 ypt>GetHiNorth())
67 return 0;
68 XLocation = xpt; // copy coordinates to FireEnvironment::Variables for use with gridded data
69 YLocation = ypt;
70
71 celldata cell;
72 crowndata crown;
73 grounddata ground;
74
75 CellData(xpt, ypt, cell, crown, ground, &posit);
76 ElevConvert(cell.e); // need to convert units and codes
77 SlopeConvert(cell.s);
78 AspectConvert(cell.a);
79 FuelConvert(cell.f);
80 CoverConvert(cell.c);
81 HeightConvert(crown.h);
82 BaseConvert(crown.b);
83 DensityConvert(crown.p);
84 DuffConvert(ground.d);
85 WoodyConvert(ground.w);
86 StationNumber = GetStationNumber(xpt, ypt) - 1;
87 ls = ld;
88
89 return posit;
90}

117void LandscapeData::SlopeConvert(short sl)
118{
119 ld.slope = sl;
120 if (ld.slope == -9999)
121 {
122 ld.slope =(short) 0.0;
123 return;
124 }
125
126 switch (GetTheme_Units(S_DATA))
127 {
128 case 0:
129 // degrees default
130 break;
131 case 1:
132 // percent
133 double fraction, ipart;
134 double slopef;
135
136 slopef = atan((double) ld.slope / 100.0) / PI * 180.0;
137 ld.slope =(short) slopef;
138 fraction = modf(slopef, &ipart);
139 if (fraction >= 0.5)
140 ld.slope++;
141 break;
142 }
143}

146void LandscapeData::AspectConvert(short as)
147{
148 if (as == -9999)
149 {
150 ld.aspect = 0;
151 ld.aspectf = 0.0;
152 return;
153 }
154 ld.aspect = as;
155 ld.aspectf = (double) as;
156 switch (GetTheme_Units(A_DATA))
157 {
158 case 0:
159 // grass 1-25 counterclockwise from east
160 if (ld.aspect != 25)
161 ld.aspectf = (2.0 * PI - (ld.aspectf / 12.0 * PI)) +
162 (7.0 * PI / 12.0); /* aspect from GRASS, east=1, north=7, west=13, south=19 */
163 else
164 {
165 ld.aspectf = 25.0;
166 ld.slope = 0;
167 }
168 break;
169 case 1:
170 // degrees 0 to 360 counterclockwise from east
171 ld.aspectf = (2.0 - ld.aspectf / 180.0) * PI + PI / 2.0;
172 break;
173 case 2:
174 ld.aspectf = ld.aspectf / 180.0 * PI; // arcinfo, degrees AZIMUTH
175 break;
176 }
177 if (ld.aspectf > (2.0 * PI))
178 ld.aspectf -= (2.0 * PI);
179}

409double FECalls::PIgnite(void)
410{
411 double mf = ohmc / 100, Qig, ProbIgnite,
412 FT = (5.0 / 9.0) * (FuelTemp - 32.0); // convert to deg C
413
414 Qig = 144.51 -
415 .266 * FT -
416 .00058 * pow2(FT) +
417 (mf / 100) * (100 - FT) +
418 18.54 * (1.0 - exp(-15.1 * mf)) +
419 640 * mf;
420 ProbIgnite = .000932 * pow((344 - Qig) / 10, 3.66) / 50.0;
421
422 return ProbIgnite;
423}

437void FireEnvironment::envt(double SIMTIME)
438{
439 int thisdate, curdate, curhour, date, hour, nearesthour, lastdate;
440 double xemc, D, Pd = 0, elevation;
441 double humid, temp, I, shade = 100, rain = 0;
442 double sunrise, sunset, tempref = 60, humref = 50;
443 double emcmax, emcmin, humidmx, humidmn, tempmx, tempmn, dayhours,
444 nitehours;
445 long elevref, cloud = 0;
446 long AtmGridExists = AtmosphereGridExists();
447 double LagRate;
448
449 FILE* radout;
450 radout = fopen("Radout.txt", "a");
451
452 elevation = (double) ld.elev * 3.28; // m to ft. conversion
453 date = chrono(SIMTIME, &hour); // determines date and hour of current iteration
454 if (hour >= 1400)
455 lastdate = date;
456 else
457 lastdate = date - 1;
458 curhour = 1400;
459 thmc = GetInitialFuelMoisture(ld.fuel, 1);//FuelMoisture.TL10[fuel-1];
460 hhmc = GetInitialFuelMoisture(ld.fuel, 2);//FuelMoisture.TL100[fuel-1];
461 for (curdate = GetStartDate(); curdate <= lastdate; curdate++)
462 {
463 thisdate = curdate;
464 if (thisdate > 365)
465 thisdate -= 365;
466 daylength(thisdate, &sunrise, &sunset);
467 if (AtmGridExists == 2)
468 {
469 AtmHumTemp(thisdate, curhour, &temp, &humid, &tempmn, &humidmn,
470 &rain, &cloud);
471 humidmx = humid;
472 tempmx = temp;
473 }
474 else // interpolate humidity and temperature from hi/lo
475 {
476 humtemp(thisdate, curhour, &tempref, &humref, &elevref, &rain,
477 &humidmx, &humidmn, &tempmx, &tempmn);
478 // rain/=100.0;
479 sitespec(elevref,(long) elevation, tempref, &temp, humref, &humid);
480 // adjusts to site specific temp and relhum at a given pixel
481 }
482 AirTemp = temp; AirHumid = humid;
483 if (AtmGridExists == 0)
484 windadj(thisdate, curhour, &cloud); // retrieves wind information and adjusts to fuel-level conditions
485 else
486 AtmWindAdjustments(thisdate, curhour, &cloud);
487 I = irradiance(cloud, thisdate, curhour, &shade);
488 if (shade < 70) // adjust humidity and temperature at fuel level only for shade <70%
489 adjust(&humid, &temp, I);
490 sfmc(rain, humid, temp);
491 if (AtmGridExists < 2)
492 {
493 sitespec(elevref, (long)elevation, tempmn, &tempmn, humidmx, &humidmx);
494 sitespec(elevref, (long)elevation, tempmx, &tempmx, humidmn, &humidmn);
495 }
496 if (shade < 70)
497 adjust(&humidmn, &tempmx, I);
498 emcmax = emc(humidmx, tempmx);
499 emcmin = emc(humidmn, tempmn);
500 dayhours = sunset - sunrise; /* using NFDRS 1978 equations */
501 nitehours = 24 - dayhours;
502 xemc = (nitehours * emcmax + dayhours * emcmin) / 24;
503 if (rain > 0)
504 Pd = (rain + .02) / .05;
505 D = ((24.0 - Pd) * xemc + (.5 * Pd + 41.0) * Pd) / 24.0;
506 hhmc = hhmc + (D - hhmc) * (1.0 - .87 * exp(-24.0 / 100.0)); // could put thousand hour moisture calculation in here
507 }
508 curdate--;
509 if (curdate < 1)
510 curdate = 365;
511 thisdate = curdate;
512 if (thisdate > 365)
513 thisdate -= 365;
514 nearesthour = (hour / 100) * 100;
515 if (nearesthour != 1400)
516 {
517 while (curhour != nearesthour) // calcs thmc for all hours since last day
518 {
519 curhour += 100;
520 if (curhour >= 2400)
521 {
522 curhour -= 2400;
523 curdate = date;
524 thisdate = curdate;
525 if (thisdate > 365)
526 thisdate -= 365;
527 }
528 daylength(thisdate, &sunrise, &sunset); // interpolates humidity and temperature from hi/lo
529 if (AtmGridExists == 2)
530 AtmHumTemp(thisdate, curhour, &temp, &humid, &tempmn,
531 &humidmn, &rain, &cloud);
532 else
533 {
534 humtemp(thisdate, curhour, &tempref, &humref, &elevref, &rain,
535 &humidmx, &humidmn, &tempmx, &tempmn);
536 sitespec(elevref, (long)elevation, tempref, &temp, humref, &humid); // adjusts to site specific temp and relhum at a given pixel
537 }
538 AirTemp = temp; AirHumid = humid;
539 if (AtmGridExists == 0)
540 windadj(thisdate, curhour, &cloud); // retrieves wind information and adjusts to fuel-level conditions
541 else
542 AtmWindAdjustments(thisdate, curhour, &cloud);
543 if (curhour / 100 < sunset && curhour / 100 > sunrise)
544 {
545 I = irradiance(cloud, thisdate, curhour, &shade);
546 fprintf(radout, "%d %lf\n", curhour, I);
547 if (shade < 70) // adjust humidity and temperature at fuel level only for shade <70%
548 adjust(&humid, &temp, I);
549 }
550 hffmc(humid, temp); // tenhour fuel moisture content
551 }
552 }
553 if (nearesthour != hour)
554 {
555 if (AtmGridExists == 0)
556 windadj(thisdate, hour, &cloud);
557 else
558 AtmWindAdjustments(thisdate, hour, &cloud);
559 }
560 xemc = emc(humid, temp);
561 ohmc = (4.0 * xemc + thmc) / 5.0; /* one hour fuel mc */
562 if (SIMTIME < 241.0)
563 {
564 LagRate = 1.0 - 1.0 / exp(SIMTIME / 60.0);
565 //ohmc=FuelMoisture.TL1[fuel-1]-LagRate*(FuelMoisture.TL1[fuel-1]-ohmc);
566 ohmc = GetInitialFuelMoisture(ld.fuel, 1) -
567 LagRate * (GetInitialFuelMoisture(ld.fuel, 1) - ohmc);
568 }
569 FuelTemp = temp;
570 // PtTemperature=temp;
571 // PtHumidity=humid;
572 PtTemperature = AirTemp;
573 PtHumidity = AirHumid;
574
575 fclose(radout);
576}

579void FireEnvironment::daylength(int date, double* sunrise, double* sunset)
580/* COMPUTES DAY LENGTH (WITHOUT LEAP YEARS */
581{
582 double delta, flatitude, hour, temp;
583
584 flatitude = (double (GetLatitude())) / 180.0 * PI;
585 delta = 23.45 * sin((0.9863 * (284.0 + date)) / 180.0 * PI);
586 delta = delta / 180.0 * PI;
587 temp = (-sin(flatitude) * sin(delta)) / (cos(flatitude) * cos(delta));
588 if (fabs(temp) > 1.0)
589 temp = 1.0;
590 hour = asin(temp);
591
592 *sunrise = (hour / PI * 180) / 15 + 6;
593 *sunset = ((1 - hour / PI) * 180) / 15 + 6;
594}

597void FireEnvironment::AtmHumTemp(int date, int hour, double* temp,
598 double* humid, double* tempmin, double* humidmin, double* rain,
599 long* cloud)
600{
601 long count = -1;
602 long ddate, month, day, hhour, xmonth, xday;
603 short humidx, tempx, ppt, twspd, wndir, cloud_s;
604 short humidmin1, humidmin2, tempmin1, tempmin2;
605 double MinTimeStep;
606
607 do
608 {
609 count++;
610 month = GetAtmosphereGrid()->GetAtmMonth(count);
611 if (month == -1) // hit end of ATMDATA
612 break;
613 day = GetAtmosphereGrid()->GetAtmDay(count);
614 ddate = day + GetJulianDays(month);
615 }
616 while (ddate != date);
617 if (month != -1) // if hit end of ATMDATA data
618 {
619 tempmin1 = -1;
620 humidmin1 = -1;
621 do
622 {
623 xmonth = GetAtmosphereGrid()->GetAtmMonth(count);
624 xday = GetAtmosphereGrid()->GetAtmDay(count);
625 hhour = GetAtmosphereGrid()->GetAtmHour(count);
626 if (xday > day || xmonth > month)
627 break;
628 if (hour == 1400) // use 1400 obs as max
629 {
630 GetAtmosphereGrid()->GetAtmosphereValue(ATMTEMP, XLocation,
631 YLocation, count, &tempmin2);
632 GetAtmosphereGrid()->GetAtmosphereValue(ATMHUMID, XLocation,
633 YLocation, count, &humidmin2);
634 if (tempmin1 == -1)
635 {
636 tempmin1 = tempmin2;
637 humidmin1 = humidmin2;
638 }
639 else
640 {
641 if (tempmin2 < tempmin1)
642 tempmin1 = tempmin2;
643 if (humidmin2 < humidmin1)
644 humidmin1 = humidmin2;
645 }
646 }
647 count++;
648 }
649 while (hour >= hhour);
650 --count;
651 GetAtmosphereGrid()->GetAtmosphereValue(ATMTEMP, XLocation, YLocation,
652 count, &tempx);
653 GetAtmosphereGrid()->GetAtmosphereValue(ATMHUMID, XLocation,
654 YLocation, count, &humidx);
655 GetAtmosphereGrid()->GetAtmosphereValue(ATMRAIN, XLocation, YLocation,
656 count, &ppt);
657 *temp = (double) tempx;
658 *humid = (double) humidx;
659 *rain = ((double) ppt) / 100.0;
660 *tempmin = (double) tempmin1;
661 *humidmin = (double) humidmin1;
662 GetAtmosphereGrid()->GetAtmosphereValue(ATMWSPD, XLocation, YLocation,
663 count, &twspd);
664 GetAtmosphereGrid()->GetAtmosphereValue(ATMWDIR, XLocation, YLocation,
665 count, &wndir);
666 GetAtmosphereGrid()->GetAtmosphereValue(ATMCLOUD, XLocation,
667 YLocation, count, &cloud_s);
668 *cloud = cloud_s;
669 twindspd = (double) twspd;
670 wwinddir = (double) wndir;
671 windreduct();
672 }
673 if (hour != 1400 && hours <= hhour)
674 {
675 MinTimeStep = (double) hhour - hours;
676 EventMinimumTimeStep(MinTimeStep); // load into global variable
677 }
678}

730void FireEnvironment::humtemp(int date, int hour, double* tempref,
731 double* humref, long* elevref, double* rain, double* humidmx,
732 double* humidmn, double* tempmx, double* tempmn)
733/* FINDS AND INTERPOLATES HUMIDTY AND TEMPERATURE FOR CURRENT HOUR */
734{
735 long count = -1, ddate = -1, month, day, hmorn, haft, Tmin, Tmax, Hmax,
736 Hmin;
737 long elref, hx, garbage, ppt;
738 double h1, h2, dtprime, dtmxmn, humid, temp, tempf, humf, sign;
739 // double dewptref, dewpt;
740
741 while (ddate != date)
742 {

870void FireEnvironment::sfmc(double rain, double humid, double temp)
871/* CALCULATES STANDARD DAILY FINE FUEL MOISTURE CONTENT */
872{
873 double fr, F, m, mr, Ed, Ew, X, f, delta, fo = thmc;
874
875 fo = 101 - fo;
876
877 if (rain > 1.5)
878 rain = 1.5;
879 if (rain > .02)
880 {
881 if (rain < .055)
882 F = -56 - 55.6 * log(rain + .04);
883 else
884 {
885 if (rain < .225)
886 F = -1 - 18.2 * log(rain - .04);
887 else
888 {
889 if (rain > .225)
890 F = 14 - 8.25 * log(rain - .075);
891 }
892 }
893
894 fr = (F * fo / 100) + 1 - 8.73 * exp(-.1117 * fo);
895 if (fr < 0)
896 fr = 0;
897 }
898 else
899 fr = fo;
900
901 mr = 101.0 - fr;
902 Ed = (.942 * pow(fabs(humid), 0.679)) + 11 * exp(humid / 10.0 - 10);
903 Ew = (.597 * pow(fabs(humid), 0.768)) + 14 * exp(humid / 8.0 - 12.5);
904 if (mr < Ed)
905 m = Ew + (mr - Ew) / 1.9953;
906 else
907 {
908 if (mr > Ed)
909 {
910 X = .424 * (1 -
911 pow((humid / 100.0), 1.7) +
912 .088 * sqrt(fabs(twindspd)) * (1.0 - pow(humid / 100.0, 8)));
913 m = Ed + (mr - Ed) / (pow(10.0, X));
914 }
915 else
916 m = mr;
917 }
918
919 if (fo < 99)
920 {
921 delta = (temp - 70.0) * (.63 - .0065 * fr);
922 if (delta < -16)
923 delta = -16;
924 }
925 else
926 delta = 0;
927 f = (101 - m + delta);
928 if (f > 99)
929 f = 99;
930 if (f < 0)
931 f = 0;
932 thmc = 101 - f;
933}

937void FireEnvironment::hffmc(double humid, double temp)
938/* CALCULATES HOURLY FINE FUEL MOISTURE INDEX FROM CANADIAN FBPS, ROTHERMEL ET AL. 1986 */
939{
940 double Ed, Ew, m, mo, ka, kd, kb, kw, Tc = (5.0 / 9.0) * (temp - 32.0); // Tc=temperature deg C
941 double twspd = twindspd*1.6096; /* convert mph to kph */
942
943 mo = m = thmc;
944 Ed = .942 * pow(fabs(humid), 0.679) +
945 11 * exp((humid - 100.0) / 10.0) +
946 .18 * (21.1 - Tc) * (1.0 - exp(-0.15 * humid));
947 if (mo > Ed)
948 {
949 ka = .424 * (1 - pow(fabs(humid / 100), 0.17)) +
950 .0694 * sqrt(fabs(twspd)) * (1 - pow((humid / 100), 8));
951 kd = .0579 * ka * exp(.0365 * Tc);
952 m = Ed + (mo - Ed) * exp(-2.303 * kd);
953 }
954 else
955 {
956 if (mo < Ed)
957 {
958 Ew = .618 * pow(fabs(humid), 0.753) +
959 10 * exp((humid - 100.0) /
960 10.0) +
961 .18 * (21.1 - Tc) * (1 - exp(-0.15 * humid));
962 if (mo < Ew)
963 {
964 kb = .424 * (pow(1.0 - ((100.0 - humid) / 100.0), 1.7)) +
965 .0694 * sqrt(fabs(twspd)) * (1 -
966 pow(((100.0 - humid) / 100.0),
967 8));
968 kw = .0579 * kb * exp(.0365 * temp);
969 m = Ew - (Ew - mo) * exp(-2.303 * kw);
970 }
971 }
972 else
973 m = mo;
974 }
975 thmc = m;
976}
