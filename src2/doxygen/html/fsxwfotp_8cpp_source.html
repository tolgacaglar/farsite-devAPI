100bool OutputFile::RastMemFile(long Type)
101{
102 double OutDoubleData;
103 long OutLongData;
104
105 // loop over all rows and all columns
106 for (long j=0; j < numrows; j++) {
107 for (long i=0; i < numcols; i++) {
108
109 // check to see if this cell has raster data written to it.
110 coordinate testxy = std::make_pair(i,j) ;
111 RasterMap::iterator data = rd.find(testxy) ;
112
113 // if this cell has no data yet, just print a -1
114 if (data == rd.end()) {
115 // output -1 as either a float or a long.
116 if (Type == RAST_FIREINTENSITY ||
117 Type == RAST_HEATPERAREA ||
118 Type == RAST_REACTIONINTENSITY ||
119 Type == RAST_ARRIVALTIME ||
120 Type == RAST_FLAMELENGTH ||
121 Type == RAST_SPREADRATE) {
122 fprintf(otpfile, "%09.3lf\n", -1.0);
123 } else {
124 fprintf(otpfile, "%03ld\n", -1l) ;
125 }
126 } else {
127 if (Type != RAST_ARRIVALTIME)
128 {
129 if ((data->second).Write == false)
130 continue;
131 }
132 x = (data->second).x;
133 y = (data->second).y;
134 t = (data->second).Time;
135 f = (data->second).Fli;
136 r = (data->second).Ros;
137 rx = (data->second).Rcx;
138 d = (data->second).Dir;
139 switch (Type)
140 {
141 case RAST_ARRIVALTIME:
142 OutDoubleData = t;
143 break;
144 case RAST_FIREINTENSITY:
145 OutDoubleData = f / convf1;
146 break;
147 case RAST_FLAMELENGTH:
148 Calcs(FLAME_LENGTH); // calculate flame length
149 OutDoubleData = l;
150 break;
151 case RAST_SPREADRATE:
152 OutDoubleData = r * convf2;
153 break;
154 case RAST_HEATPERAREA:
155 Calcs(HEAT_PER_AREA); // calculate heat/unit area
156 OutDoubleData = h;
157 break;
158 case RAST_REACTIONINTENSITY:
159 Calcs(HEAT_PER_AREA); // calculate heat/unit area
160 OutDoubleData = rx;
161 break;
162 case RAST_CROWNFIRE:
163 Calcs(CROWNFIRE);
164 OutLongData = c;
165 break;
166 case RAST_FIREDIRECTION:
167 OutLongData = d;
168 break;
169 }
170 if (Type == RAST_ARRIVALTIME)
171 {
172 fprintf(otpfile, "%09.3lf\n", OutDoubleData);
173 (data->second).Write = true;
174 } else if (Type == RAST_FIREINTENSITY ||
175 Type == RAST_HEATPERAREA ||
176 Type == RAST_REACTIONINTENSITY)
177 fprintf(otpfile, "%09.3lf\n", OutDoubleData);
178 else if (Type == RAST_FLAMELENGTH || Type == RAST_SPREADRATE)
179 fprintf(otpfile, "%05.2lf\n", OutDoubleData);
180 else
181 fprintf(otpfile, "%03ld\n", OutLongData);
182 }
183 }
184 }
185
186 return false;
187}

279{
280 long garbage;
281 double Io, Ro;
282 celldata cd;
283 crowndata rd;
284 grounddata gd;
285
286 CellData(x, y, cd, rd, gd, &garbage);
287 if (cd.c > 0)
288 {
289 ld.BaseConvert(rd.b);
290 ld.DensityConvert(rd.p);
291 c = 1;
292 Io = pow(0.010 * ld.ld.base * (460.0 + 25.9 * GetFoliarMC()), 1.5);
293 if (f >= Io)
294 {
295 c = 2;
296 if(ld.ld.density>0.0)
297 { if(GetCrownFireCalculation()==0)
298 Ro=2.8/ld.ld.density; // close to the ro+0.9(rac-ro)
299 else
300 Ro=3.0/ld.ld.density;
301 if(r>=Ro)
302 c=3;
303 }
304 }
305 }
306 else if (r > 1e-6)
307 c = 1;
308
309 switch (TYPE)
310 {
311 case FLAME_LENGTH:
312 if (c > 1)
313 l = ((0.2 * pow(f / 3.4613, 2.0 / 3.0)) / 3.2808) * convf2; // fl for crown fire from Thomas in Rothermel 1991 converted to m

315 l = 0.0775 * pow(f, 0.46) * convf2; // fl from Byram for surface fires
316 break;
317 case HEAT_PER_AREA:

327 l = 0.0775 * pow(f, 0.46) * convf2; // fl from Byram for surface fires

398void OutputFile::WriteFile(long Type)
399{
400 double xres, yres;
401 char RasterCopy[256] = "";
402 long numcells;
403 double MetersToKm = 1.0;
404
405 GetRastRes(&xres, &yres);
406 numcols = (long) ((East - West) / xres);//((GetEastUtm()-GetWestUtm())/xres);
407 numrows = (long) ((North - South) / yres);//((GetNorthUtm()-GetSouthUtm())/yres);
408 numcells = numrows * numcols;
409 memset(RasterCopy, 0x0, sizeof RasterCopy);
410 strcpy(RasterCopy, GetRasterFileName(Type));
411
412 if (CheckCellResUnits() == 2)
413 MetersToKm = 0.001;
414
415 otpfile = fopen(RasterCopy, "w");
416 if (otpfile == NULL)
417 {
418 //SetFileAttributes(RasterCopy, FILE_ATTRIBUTE_NORMAL);
419 chmod(RasterCopy,S_IRWXO);
420 otpfile = fopen(RasterCopy, "w");
421 }
422 switch (HeaderType)
423 {
424 case 1:
425 if (GetNorthUtm() == 0.0 || GetEastUtm() == 0.0)
426 {
427 fprintf(otpfile, "%s %lf\n", "north:", North);//*MetersToKm);
428 fprintf(otpfile, "%s %lf\n", "south:", South);//*MetersToKm);
429 fprintf(otpfile, "%s %lf\n", "east:", East);//*MetersToKm);
430 fprintf(otpfile, "%s %lf\n", "west:", West);//*MetersToKm);
431 }
432 else
433 {
434 fprintf(otpfile, "%s %lf\n", "north:",
435 ConvertNorthingOffsetToUtm(North));//*MetersToKm);
436 fprintf(otpfile, "%s %lf\n", "south:",
437 ConvertNorthingOffsetToUtm(South));//*MetersToKm);
438 fprintf(otpfile, "%s %lf\n", "east:",
439 ConvertEastingOffsetToUtm(East));//*MetersToKm);
440 fprintf(otpfile, "%s %lf\n", "west:",
441 ConvertEastingOffsetToUtm(West));//*MetersToKm);
442 }
443 fprintf(otpfile, "%s %ld\n", "rows:", numrows);
444 fprintf(otpfile, "%s %ld\n", "cols:", numcols);
445 break;
446 case 5:
447 fprintf(otpfile, "%s %ld\n", "NCOLS", numcols);
448 fprintf(otpfile, "%s %ld\n", "NROWS", numrows);
449 if (GetNorthUtm() == 0.0 || GetEastUtm() == 0.0)
450 {
451 fprintf(otpfile, "%s %lf\n", "XLLCORNER", West);//*MetersToKm);
452 fprintf(otpfile, "%s %lf\n", "YLLCORNER", South);//*MetersToKm);
453 }
454 else
455 {
456 fprintf(otpfile, "%s %lf\n", "XLLCORNER",
457 ConvertEastingOffsetToUtm(West));//*MetersToKm);
458 fprintf(otpfile, "%s %lf\n", "YLLCORNER",
459 ConvertNorthingOffsetToUtm(South));//*MetersToKm);
460 }
461 fprintf(otpfile, "%s %lf\n", "CELLSIZE", xres * MetersToKm); //GetRastRes()
462 fprintf(otpfile, "%s %s\n", "NODATA_VALUE", "-1");
463 break;
464 }
465
466 // write the actual data to the file.
467 RastMemFile(Type) ;
468 fclose(otpfile);
469}

523bool OutputFile::SetRastData(double xpt, double ypt, double time, double fli,
524 double ros, double rcx, long dir)
525{
526
527 double xres, yres ;
528 coordinate tempxy ;
529 RastData temp ;
530
531 GetRastRes(&xres, &yres);
532
533 // compute the grid cell value
534 tempxy.first = (long) ((xpt - West) / xres) ;
535 tempxy.second = (long) ((North - ypt) / yres) ;
536
537 // check to see if this cell already has an entry
538 RasterMap::iterator oldVal = rd.find(tempxy) ;
539
540
541 if (oldVal != rd.end()) {
542 // handle special cases where cell already has value
543 temp.Time = std::min((oldVal->second).Time, time) ;
544 } else {
545 // handle special cases where cell doesn't have value yet
546 temp.Time = time ;
547 }
548
549 // handle "standard" case where we blithely replace the value
550 temp.x = xpt ;
551 temp.y = ypt ;
552 temp.Fli = fli;
553 temp.Ros = ros;
554 temp.Rcx = rcx;
555 temp.Dir = dir;
556
557 // add this to the map (if it was already in the map, tell where)
558 if (oldVal != rd.end()) {
559 rd.insert(oldVal, std::make_pair(tempxy,temp)) ;
560 } else {
561 rd.insert(std::make_pair(tempxy,temp)) ;
562 }
563
564 NumRastData = rd.size() ;
565
566 return true;
567}
