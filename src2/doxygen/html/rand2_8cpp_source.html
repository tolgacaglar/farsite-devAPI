67double rand2(long* idum)
68{
69 long j, k;
70 static long idum2 = 123456789;
71 static long iy = 0;
72 static long iv[NTAB];
73 double temp;
74
75 if (*idum <= 0)
76 {
77 if (-(*idum) < 1)
78 *idum = 1;
79 else
80 *idum = -(*idum);
81 idum2 = (*idum);
82
83 for (j = NTAB + 7; j >= 0; j--)
84 {
85 k = (*idum) / IQ1;
86 *idum = IA1 * (*idum - k * IQ1) - k * IR1;
87 if (*idum < 0)
88 *idum += IM1;
89 if (j < NTAB)
90 iv[j] = *idum;
91 }
92 iy = iv[0];
93 }
94 k = (*idum) / IQ1;
95 *idum = IA1 * (*idum - k * IQ1) - k * IR1;
96 if (*idum < 0)
97 *idum += IM1;
98 k = idum2 / IQ2;
99 idum2 = IA2 * (idum2 - k * IQ2) - k * IR2;
100 if (idum2 < 0)
101 idum2 += IM2;
102 j = iy / NDIV;
103 if (j < 0)
104 j = labs(j);
105 iy = iv[j] - idum2;
106 iv[j] = *idum;
107 if (iy < 1)
108 iy += IMM1;
109 if ((temp = AM * iy) > RNMX)
110 return RNMX;
111
112 if (temp < 0.0)
113 temp *= -1.0;
114
115 return temp;
116}
