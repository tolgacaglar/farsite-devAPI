66FireRing* AllocFireRing(long NumPoints, double start, double end)
67{
68 long curplace, i, ThisRing;
69 double Ring;
70
71 modf(((double) NumRings / (double) RINGS_PER_STRUCT), &Ring);
72 ThisRing = (long) Ring;
73 curplace = NumRings - ThisRing * RINGS_PER_STRUCT;
74
75 if (FirstRing == 0)
76 {
77 if ((FirstRing = new RingStruct) == NULL)
78 return 0;
79 CurRing = FirstRing;
80 CurRing->NumFireRings = 0;
81 CurRing->StructNum = 0;
82 NumRingStructs++;
83 for (i = 0; i < RINGS_PER_STRUCT; i++)
84 {
85 CurRing->firering[i].perimpoints = 0;
86 CurRing->firering[i].mergepoints = 0;
87 CurRing->firering[i].NumPoints = 0;
88 }
89 }
90 else if (ThisRing >= NumRingStructs)
91 {
92 if (CurRing->StructNum != NumRingStructs - 1)
93 GetLastRingStruct();
94 //NextRing = (RingStruct *) CurRing->next = new RingStruct;
95 RingStruct* t_next;
96 t_next = new RingStruct;
97 CurRing->next=t_next;
98 //NextRing=CurRing->next;
99 NextRing = t_next;
100 CurRing = NextRing;
101 CurRing->NumFireRings = 0;
102 CurRing->StructNum = NumRingStructs;
103 NumRingStructs++;
104 for (i = 0; i < RINGS_PER_STRUCT; i++)
105 {
106 CurRing->firering[i].perimpoints = 0;
107 CurRing->firering[i].mergepoints = 0;
108 CurRing->firering[i].NumPoints = 0;
109 }
110 }
111 else if (ThisRing != CurRing->StructNum)
112 GetRing(NumRings);
113 CurRing->firering[curplace].perimpoints = new PerimPoints[NumPoints];
114 for (i = 0; i < NumPoints; i++)
115// ZeroMemory(&CurRing->firering[curplace].perimpoints[i], sizeof(PerimPoints));
116 memset(&CurRing->firering[curplace].perimpoints[i],0x0, sizeof(PerimPoints));
117 CurRing->firering[curplace].NumPoints = new long[2];
118 CurRing->firering[curplace].NumPoints[0] = NumPoints;
119 CurRing->firering[curplace].NumFires = 1;
120 CurRing->firering[curplace].StartTime = start;
121 CurRing->firering[curplace].ElapsedTime = end - start;
122 CurRing->firering[curplace].mergepoints = 0;
123 CurRing->firering[curplace].NumMergePoints[0] = 0;
124 CurRing->firering[curplace].NumMergePoints[1] = 0;
125
126 CurRing->NumFireRings++;
127 NumRings++;
128
129 return &CurRing->firering[curplace];
130}

280void CondenseRings(long RingNum)
281{
282 // searches starting with RingNum for FireRings with no points. Then it shifts
283 // the pointers down to the earliest vacant slot. This function is called after
284 // all mergers are completed.
285
286 if (RingNum < 0)
287 return;
288 else if (RingNum == GetNumRings() - 1)
289 return;
290
291 FireRing* ring1, * ring2;
292 long i, j, TotalPts;
293 long NewRingNum;//=GetNumRings();
294 long NewStructNum;
295
296 for (i = RingNum; i < GetNumRings(); i++)
297 {
298 ring1 = GetRing(i);
299 if (!ring1)
300 continue;
301 if (ring1->mergepoints)
302 {
303 delete[] ring1->mergepoints;
304 ring1->mergepoints = 0;
305 ring1->NumMergePoints[0] = 0;
306 }
307 if (ring1->perimpoints == NULL)
308 {
309 FreeFireRing(i);
310 TotalPts = 0;
311 for (j = i + 1; j < GetNumRings(); j++)
312 {
313 ring2 = GetRing(j);
314 if (!ring2)
315 continue;
316 if (ring2->perimpoints)
317 {
318 TotalPts = ring2->NumPoints[ring2->NumFires - 1];
319 ring1->perimpoints = ring2->perimpoints;
320 ring1->NumPoints = ring2->NumPoints;
321 ring2->perimpoints = 0;
322 ring2->NumPoints = 0;
323 ring1->NumFires = ring2->NumFires;
324 ring1->OriginalFireNumber = ring2->OriginalFireNumber;
325 ring1->StartTime = ring2->StartTime;
326 ring2->OriginalFireNumber = (long)0;
327 ring2->StartTime = 0.0;
328 ring1->ElapsedTime = ring2->ElapsedTime;
329 ring2->NumFires = 0;
330 ring2->OriginalFireNumber = -1;
331
332 break;
333 }
334 }
335 }
336 }
337
338 // free up CurRing if no points left!!!!!
339 RingStruct* LastRing;
340 CurRing = LastRing = FirstRing;
341 NewStructNum = 0;
342 for (i = 0; i < NumRingStructs; i++)
343 {
344 if (CurRing != NULL)
345 {
346 NewRingNum = 0;
347 for (j = 0; j < RINGS_PER_STRUCT; j++)
348 {
349 if (CurRing->firering[j].perimpoints != NULL)
350 NewRingNum++;
351 }
352 CurRing->NumFireRings = NewRingNum;
353 if (NewRingNum > 0)
354 NewStructNum++;
355 else
356 {
357 NextRing = (RingStruct *) CurRing->next;
358 LastRing->next = NextRing;
359 if (CurRing == LastRing) // will only occur if there are no rings now
360 {
361 delete FirstRing;
362 FirstRing = LastRing = 0;
363 }
364 else
365 delete CurRing;
366 CurRing = LastRing;
367 }
368 }
369 if (CurRing == NULL)
370 break;
371 LastRing = CurRing;
372 NextRing = (RingStruct *) CurRing->next;
373 CurRing = NextRing;
374 }
375 CurRing = FirstRing;
376 NewRingNum = 0;
377 for (i = 0; i < NewStructNum; i++)
378 {
379 NewRingNum += CurRing->NumFireRings;
380 NextRing = (RingStruct *) CurRing->next;
381 CurRing = NextRing;
382 }
383 SetNumRings(NewRingNum);//(NewStructNum-1)*RINGS_PER_STRUCT+NewRingNum);
384 NumRingStructs = NewStructNum;
385 CurRing = FirstRing;
386}

427bool AddToCurrentFireRing(FireRing* firering, long PointNum,
428 long SurfFuelType, long WoodyModel, double DuffLoad, double* mx,
429 double CrownLoadingBurned)
430{
431 if (!firering)
432 return false;
433
434 long i, NumFire;
435 double x, y, r, f, c;
436
437 NumFire = firering->OriginalFireNumber;
438 f = GetPerimeter2Value(PointNum, FLIVAL);
439 if (f < 0.0)
440 {
441 x = GetPerimeter2Value(PointNum, XCOORD);
442 y = GetPerimeter2Value(PointNum, YCOORD);
443 r = fabs(GetPerimeter2Value(PointNum, ROSVAL));
444 f = fabs(f);
445 c = GetPerimeter2Value(PointNum, RCXVAL);
446 }
447 else
448 {
449 x = GetPerimeter1Value(NumFire, PointNum, XCOORD);
450 y = GetPerimeter1Value(NumFire, PointNum, YCOORD);
451 r = fabs(GetPerimeter1Value(NumFire, PointNum, ROSVAL));
452 f = fabs(GetPerimeter1Value(NumFire, PointNum, FLIVAL));
453 c = GetPerimeter1Value(NumFire, PointNum, RCXVAL);
454 }
455 firering->perimpoints[PointNum].x2 = x;
456 firering->perimpoints[PointNum].y2 = y;
457 if (c < 1e-2)
458 c = 0.0;
459 firering->perimpoints[PointNum].hist.ReactionIntensity[1] = (float) c;
460 if (fabs(r) > 0.0 && c > 0.0)
461 firering->perimpoints[PointNum].hist.FlameResidenceTime[1] = (float)
462 ((f * 60.0) / (r * c)); // seconds
463 else
464 firering->perimpoints[PointNum].hist.FlameResidenceTime[1] = 0.0;
465
466 firering->perimpoints[PointNum].hist.WoodyFuelType = (unsigned char)
467 WoodyModel;
468 firering->perimpoints[PointNum].hist.SurfaceFuelModel = (unsigned char)
469 SurfFuelType;
470 if (DuffLoad < 0.1)
471 DuffLoad = 0.1;
472 firering->perimpoints[PointNum].hist.DuffLoad = (short) (DuffLoad * 10.0);
473 //mx=GetAllCurrentMoistures(&NumMx);
474
475 if (mx)
476 {
477 for (i = 0; i < MAXNO; i++)
478 firering->perimpoints[PointNum].hist.Moistures[i] = (unsigned char)
479 (mx[i] * 100.0);
480 }
481 //for(i=NumMx-1; i<20; i++)
482 // firering->perimpoints[PointNum].hist.Moistures[i]=0;
483 firering->perimpoints[PointNum].hist.LastWtRemoved = 0.0; // initialize last values of integration
484 firering->perimpoints[PointNum].hist.WeightPolyNum = 0;
485 firering->perimpoints[PointNum].hist.FlamePolyNum = 0;
486 firering->perimpoints[PointNum].hist.CrownLoadingBurned = (float)
487 CrownLoadingBurned;
488
489 return true;
490}

703bool RingBurn::CombustionHistoryExists(long CurIndex)
704{
705 bool Exists = false;
706 long i, j, k, m, n, nrings, nfires;
707 double heat, xheat, heatrat;
708 FireRing* xring;
709 BurnHistory* hist, * xhist;
710
711 hist = &ring->perimpoints[CurIndex].hist;
712 heat = hist->ReactionIntensity[1] * hist->FlameResidenceTime[1];
713 if (CurIndex > Begin)
714 {
715 for (i = Begin; i < CurIndex; i++)
716 {
717 if (ring->perimpoints[i].Area <= 0.0)
718 continue;
719 xhist = &ring->perimpoints[i].hist;
720 if (xhist->WeightPolyNum == 0)
721 return false;
722 if (hist->SurfaceFuelModel != xhist->SurfaceFuelModel)
723 continue;

4209 // over the area and radial distance of the fire quadrangle over the

4278 Area = 0.0;
4279 TempWeight += Area * ring->perimpoints[j].hist.TotalWeight;
4280
4281 OldFract = Fract1;
4282 }
4283 else
4284 {
4285 Fract = ((double) (i + 1) * TimeInc) / (double) ring->ElapsedTime;
4286 l2 = l1 + Fract * h * dldh;

4288 wtmid = (wt1 + wt2) / 2.0;
4289 TempWeight = (1.0 - wtmid) * Area * ring->perimpoints[j].hist.TotalWeight;
4290 AreaFract = 1.0;
