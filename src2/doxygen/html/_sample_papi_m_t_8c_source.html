102void Sample_ValueDiff (int i, long long V) {
103 struct Proc_INFO *pInfo= Sample_INFO + i;
104 long long dV;
105
106 dV = V - pInfo->tot[0];
107 pInfo->tot[0] = V; // store current V as old V
108 pInfo->count++;
109
110 if (pInfo->count == 1) {
111 if (pInfo->val[0] == 0) { // filter first case
112 pInfo->val[0] = 1;
113 pInfo->count = 0;
114 return;
115 }
116 // first difference
117 pInfo->tot[1] = dV; // maximum dV
118 pInfo->tot[2] = dV; // minimum dV
119 pInfo->tot[3] = dV; // most frequent dV
120 pInfo->val[0] = 0; // will be increased in brief
121 }
122
123 if (dV > pInfo->tot[1]) pInfo->tot[1] = dV; // maximum dV
124 if (dV < pInfo->tot[2]) pInfo->tot[2] = dV; // minimum dV
125 if (dV == pInfo->tot[3]) pInfo->val[0]++; // reinforce dV
126 else if (pInfo->val[0]) pInfo->val[0]--; // weaken dV
127 else {
128 pInfo->tot[3] = dV; // new most frequent dV
129 pInfo->val[0] = 1;
130 }
131
132 dV = llabs(dV);
133 pInfo->Sum += dV; pInfo->SumSQR += (double) dV * (double) dV;
