126void Embers::Loft(double CFlameLength, double CFBurned, double CrownHeight,
127 double /*LoadingBurned*/, double /*ROS*/, double /*SubTimeStep*/,
128 double curtime)
129{
130 // This version of Loft attempts to loft 16 embers between .005 and .08 ft diameter
131 double Diameter; // ember diameter in feet
132 double zo; // original height ember was lofted
133 double ZF; // height of lofted ember in feet
134
135 Plume(CFlameLength, CFBurned); // compute plume characteristics from torching trees, and temporarily crowning (TEMPORARY!)
136 CrownHeight *= 3.2808;
137 zo = CrownHeight;//GetCrownHeight()*3.28; // initial ember ht at tree top
138 for (long count = 1; count < 17; count++)
139 {
140 Diameter = .005 * count;
141 switch (SpotSource)
142 {
143 case 0:
144 ZF = torchheight(Diameter, zo); // allow for variable ember starting height
145 if (ZF > 0)
146 ZF += CrownHeight / 2.0;//GetCrownHeight()*1.64; // 3.28m/2 feet, base of flames at 1/2 tree height, so add to z for total particle height + elevation of ground for altitude
147 break;
148 case 1:
149 ZF = torchheight(Diameter, zo);
150 if (ZF > 0)
151 ZF += CrownHeight / 2.0;//GetCrownHeight()*1.64; // 3.28m/2 feet, base of flames at 1/2 tree height, so add to z for total particle height + elevation of ground for altitude
152 break;
153 // case 2:
154 // ZF=pileheight(Diameter);
155 // if(ZF>0)
156 // ZF+=3; // ember originate from top of 3m tall pile
157 // break;
158 // case 3:
159 // ZF= // spotting from line fire
160 // lineloft(.....);
161 // break;
162 }
163 if (ZF > 0) // if ember is lofted by flame
164 {
165 if (NumEmbers > 0)
166 {
167 //NextEmber = (emberdata *)
168 // CurEmber->next = (emberdata *) new emberdata;// GlobalAlloc(GMEM_FIXED | GMEM_ZEROINIT, 1*sizeof(emberdata));
169 emberdata* t_next;
170 t_next = new emberdata;
171 CurEmber->next = t_next;
172 NextEmber = t_next;

174 }
175 else
176 {
177 FirstEmber = (emberdata *) new emberdata;//GlobalAlloc(GMEM_FIXED | GMEM_ZEROINIT, 1*sizeof(emberdata)); // initialize first ember
178 CurEmber = FirstEmber;
179 }
180 CurEmber->x = Fcoord.x; // store ember coordinates
181 CurEmber->y = Fcoord.y; // UTM meters
182 CurEmber->PartDiam = Diameter; // ember diameters inches
183 CurEmber->ZHeight = ZF / 3.2808; // ember heights in m
184 CurEmber->StartElev = Fcoord.e; // original surface elevation of land from which ember was located
185 CurEmber->CurrentTime = curtime; // time of ember origination in simulation
186 CurEmber->ElapsedTime = 0.0; // time elapsed since ember attained max height
187 NumEmbers++; // calculate total number of embers;
188 }
189 else
190 count = 16; // force exit from for-loop
191 }
192}

195void Embers::Plume(double CFlameLength, double CFBurned)
196{
197 double part1, part2, part3, part4;
198
199 if (SpotSource <= 1) // active crown fire or torching trees
200 {
201 long treespecies = GetCanopySpecies();
202 long tnum = 1; //tnum= # trees simultaneously torching in group
203 double DBH = GetAverageDBH() / 2.54; // cm to inches
204
205 xdiffl = (Fcoord.x - Fcoord.xl) / 2.0 / MetricResolutionConvert(); // required to be in meters
206 ydiffl = (Fcoord.y - Fcoord.yl) / 2.0 / MetricResolutionConvert();
207 xdiffn = (Fcoord.x - Fcoord.xn) / 2.0 / MetricResolutionConvert();
208 ydiffn = (Fcoord.y - Fcoord.yn) / 2.0 / MetricResolutionConvert();
209 FrontDist = sqrt(pow2(xdiffl) + pow2(ydiffl)) + sqrt(pow2(xdiffn) +
210 pow2(ydiffn)); // linear frontal fire distance surrounding x,y coord
211 if (SpotSource == 0) // if active crown fire,
212 {
213 SteadyHeight = CFlameLength * 3.28; // then steady flameheight = crownfire flame length
214 tnum = (long) (FrontDist / 5.0); // assume 5m mean crown diameter for # trees on firefront
215 if (tnum < 1)
216 tnum = 1; // minimum number of trees is 1
217 part1 = A[treespecies][0]; // INCORRECT, USE DIFFERENT LOFTING MECH FOR ACTIVE CROWN FIRE !!
218 part2 = A[treespecies][1]; // "" ""
219 SteadyHeight = part1 * pow(DBH, part2) * pow(tnum, 0.4); // steady flame height
220 }
221 else // IF TORCHING TREES, THE CALCULATE FLAMEHT FROM ALBINI 1979 AND CHASE 1981
222 {
223 if (CFBurned > 0.5 && Fcoord.cover > 50) // cover>2
224 tnum++; // DEFAULT IS 1-TREE TORCHING INDEPENDENTLY, BUT MORE CROWN BURNED
225 if (CFBurned > 0.8) // INCREASES THE NUMBER OF TREES TORCHING TOGETHER with CFB and COVER
226 {
227 tnum++;
228 if (Fcoord.cover > 50) // cover>2
229 tnum = 6;
230 if (Fcoord.cover > 80) // cover>3
231 tnum = 10;
232 }
233 part1 = A[treespecies][0];
234 part2 = A[treespecies][1];
235 SteadyHeight = part1 * pow(DBH, part2) * pow(tnum, 0.4); // steady flame height
236 }
237 part3 = A[treespecies][2];
238 part4 = A[treespecies][3];
239 Duration = part3 * pow(DBH, -part4) * pow(tnum, -0.2); // steady flame Duration
240 }
241 else
242 {
243 if (SpotSource == 2) // burning piles or in this case homes or buildings
244 SteadyHeight = CFlameLength;
245
246 /* else // wind driven surface fires
247 {
248 }
249 */
250 }

254void Embers::Flight(double CurTime, double EndCurTimeStep)
255{
256 // Iterates ember flight down through logarithmic wind profiles
257 double seast = 0, snorth = 0, tseast = 0, tsnorth = 0, Diameter = 0;
258 double sstep = .25, SubSStep = 0, curtime, elapsedtime; // sstep is the spot time step, in minutes
259 //double part1=0, part2=0, part3=0, part4=0, IgChance=0, SmokeDispersionAngle=0; // H=tree ht in feet,
260 double MAXZ = 0, Z, mZt = 0, Xt = 0, mXt = 0, Xtot = 0; // mZt is the particle ht at time t, zo is the original height of the particle in tree canopy
261 double Z1AboveGround, Z2AboveGround, ZAvgAboveGround;
262 double zfuel1, zfuel2, Z1HtFromStart, Z2HtFromStart;
263 double ztest1, ztest2, zt1, zt2;
264 double voo, tao, DZRate, UH, rwinddir; // rwinddir=randomwinddirection
265 double dxdt, HtLimit, eZelev, Zelev = 0;
266 long posit, positlast = 0;
267 long count = 0, PassA, PassB;
268 long idum;
269 emberdata* LastCarryEmber;
270
271 //--------------------------------------------------------
272 //------Probability of ignition---------------------------
273
274 long IgProb = (long)PercentIgnition(GETVAL) * (long)100; // in FSXW.CPP
275 long ProbSpot;
276
277 srand(clock());
278 idum = -((rand() % 1000) + 1); // initialize random num gen
279 //--------------------------------------------------------
280
281 double eH;
282 double TempAccel = mech.A;
283
284 mech.cosslope = 99.0;
285 mech.RosT = 0.0;
286 EndCurTimeStep = EndCurTimeStep;//+GetActualTimeStep(); // update Current Time to end of current simulation time step
287 CurEmber = CarryEmber = FirstEmber; // pointer to first ember structure
288
289 if (CarrySpots > 0)
290 {
291 CurSpot = CarrySpot;
292 NumSpots = CarrySpots;
293 }
294 else
295 NumSpots = 0;
296
297 for (count = 0; count < NumEmbers; count++) // for all particles
298 {
299 NextEmber = (emberdata *) CurEmber->next;
300 seast = CurEmber->x;
301 snorth = CurEmber->y;
302 Diameter = CurEmber->PartDiam; // original particle diameter, stays constant
303 Z = CurEmber->ZHeight; // max height of ember m, stays constant
304 Zelev = CurEmber->StartElev; // elevation m of land producing ember, stays constant, and zfuel not constant
305 curtime = CurEmber->CurrentTime; // updated current time of ember
306 elapsedtime = CurEmber->ElapsedTime; // updated time elapsed since ember was at max height
307 CurEmber = NextEmber;
308 //if(Diameter>SuccessfulEmberDiameter)
309 // continue;
310 //--------------------------------------------------------
311 ProbSpot = (long)rand2(&idum) * (long)10000;//rand()%1000;
312 if (ProbSpot > IgProb)
313 continue;
314 //--------------------------------------------------------
315 if (curtime == EndCurTimeStep) // >=EndCurTimeStep-actual, wait 2 time steps to see if ember falls with burned zone
316 {
317 CarryEmber->x = seast;
318 CarryEmber->y = snorth; // resave ember coordinates
319 CarryEmber->PartDiam = Diameter; // original particle diameter, stays constant
320 CarryEmber->ZHeight = Z; // max height of ember m, stays constant
321 CarryEmber->StartElev = Zelev; // elevation m of land producing ember, stays constant
322 CarryEmber->CurrentTime = curtime;
323 CarryEmber->ElapsedTime = elapsedtime; // copy elapsed time from CurEmber
324 NextCarryEmber = (emberdata *) CarryEmber->next; // use next ember structure
325 LastCarryEmber = CarryEmber;
326 CarryEmber = NextCarryEmber;
327 CarryOverEmbers++; // increment number of carryover embers
328 continue; // go to next ember
329 }
330 else
331 {
332 posit = fe->GetLandscapeData(seast, snorth, mech.ld);
333 eH = mech.ld.height * 3.2808;//GetCrownHeight()*3.28; // metric to feet
334 if (eH <= 0.0)
335 eH = 1.0;
336 fe->GetFireEnvironment(env, CurTime, false);//, 0, 1, -1);
337 mech.LoadLocalFEData(fe);
338 if (mech.ld.fuel > 50) // if fuel is pile or building
339 zfuel1 = zfuel2 = mech.ld.elev + 3; // 3m high fuel pile
340 else
341 zfuel1 = zfuel2 = mech.ld.elev;
342 positlast = posit;
343 }
344 if (elapsedtime == 0)
345 Zelev = zfuel1; // sometimes Zfuel<>zfuel1 becasue of spotoffset along perimeter in LOFT
346 MAXZ = 39000.0 * Diameter; // maximum height (ft) particle can fall and still be burning
347 //voo=pow((1910.087*Diameter)/0.18,0.5); // terminal velocity of particle, ft/s; g=32 ft/sec2, partdens=19lb/ft3, plumedensity=.028 lb/ft3, drag coef=1.2
348 voo = sqrt((1910.087 * Diameter) / 0.18); // terminal velocity of particle, ft/s; g=32 ft/sec2, partdens=19lb/ft3, plumedensity=.028 lb/ft3, drag coef=1.2
349 tao = (4.8 * voo) / (0.0064 * PI * 32.0); // gravity= 32 ft/sec2
350 Xtot = 0; // reset total horizontal distance traveled
351 Z *= 3.28; // convert to feet above orignial land
352 eZelev = Zelev * 3.28; // convert Zelev to feet of original land surface
353
354 zt1 = 0.0;
355 rwinddir = mech.lmw.winddir;
356 do
357 {
358 ztest1 = voo * tao * ((elapsedtime * 60.0) /
359 tao -
360 .5 * pow2((elapsedtime * 60.0) /
361 tao));// compute test vertical drop of ember over sstep (0.25 min)
362 ztest2 = voo * tao * (((sstep + elapsedtime) * 60.0) /
363 tao -
364 .5 * pow2(((sstep + elapsedtime) * 60.0) /
365 tao));// compute test vertical drop of ember over sstep (0.25 min)
366 if (ztest1 < ztest2) // particle burned out before contacting ground
367 DZRate = (ztest2 - ztest1) / sstep; // average rate of vertical drop (ft per minute)
368 else
369 break;
370 for (PassA = 0; PassA < 3; PassA++)
371 {
372 positlast = posit; // file position of raster
373 HtLimit = 50.0; // 50 FT HEIGHT INCREMENT
374 for (PassB = 0; PassB < 2; PassB++)
375 {
376 SubSStep = HtLimit / DZRate; // Sub time step in minutes (ft/(ft/min))
377 if (curtime + SubSStep > EndCurTimeStep) // restrict substep not to exceed current time step
378 {
379 SubSStep = EndCurTimeStep - curtime;
380 if (SubSStep < 0.0)
381 {
382 SubSStep = 0.0;
383 curtime = EndCurTimeStep;
384 }
385 HtLimit = SubSStep * DZRate; // recalc ht limit in terms of new time step
386 }
387 zt2 = zt1 + HtLimit;
388 if (zt1 < 0)
389 Z1HtFromStart = eZelev - (Z - zt1); // EMBER DROPPED BELOW ORIGINAL LAND HEIGHT
390 else
391 Z1HtFromStart = eZelev + (Z - zt1);
392 if (zt2 < 0)
393 Z2HtFromStart = eZelev - (Z - zt2); // EMBER DROPPED BELOW ORIGINAL LAND HEIGHT
394 else
395 Z2HtFromStart = eZelev + (Z - zt2);
396 Z1AboveGround = Z1HtFromStart - zfuel1 * 3.28;
397 Z2AboveGround = Z2HtFromStart - zfuel2 * 3.28;
398 mZt = Z2AboveGround / 3.28;
399 if (mZt > 1.0) // 1.0 meter tolerance
400 {
401 UH = VertWindSpeed(Z1AboveGround, eH) / 2.0; // half of UeH in ft/s
402 UH += VertWindSpeed(Z2AboveGround, eH) / 2.0; // average UH in ft/s
403 if (Z1AboveGround > 0)
404 {
405 if (Z2AboveGround > 0)
406 ZAvgAboveGround = (Z1AboveGround +
407 Z2AboveGround) /
408 2.0;
409 else
410 ZAvgAboveGround = Z1AboveGround / 2.0;
411 if (ZAvgAboveGround > 0.1313 * eH)
412 {
413 dxdt = UH / 2.03 * log(ZAvgAboveGround /
414 (.1313 * eH)); // dxdt is ft/s
415 Xt = dxdt * 60.0 * SubSStep; //Albini 1979 spotting model, ft/min
416 mXt = Xt / 3.28; // incremental distance traveled m/min
417 }
418 else
419 mXt = 0.0;
420 if (PassB<
421 1 &&
422 mXt>
423 (GetCellResolutionX() /
424 MetricResolutionConvert() -
425 0.5))
426 HtLimit /= 2.0; // SubSStep=(95.12)/(dxdt*60.0);
427 else
428 PassB += 2;
429 }
430 }
431 else
432 {
433 if (mZt < -1.0)
434 {
435 HtLimit = Z1AboveGround +
436 (3.28 * (zfuel1 - zfuel2));
437 PassB--;
438 if (HtLimit < 0) // could happen if ember at zt1 is way below zfuel2 at zt2
439 {
440 mXt = 0.0;
441 if (!elapsedtime) // happens when ember is lofted to height below adjacent cell
442 PassA = 10;
443 else
444 PassA = 6;
445 break;
446 }
447 }
448 else
449 {
450 PassA = 6;
451 break;
452 }
453 }
454 } // LOOP WAS while(mXt>29.0);, BUT ONLY WANT 2 PASSES THROUGH THIS LOOP
455 if (PassA > 7)
456 break;
457 tseast = seast -
458 mXt * MetricResolutionConvert() * sin(rwinddir); // temporary position change, will be permanent if Z iterations ok
459 tsnorth = snorth -
460 mXt * MetricResolutionConvert() * cos(rwinddir);
461 if (tseast >= GetHiEast() ||
462 tseast <= GetLoEast() ||
463 tsnorth >= GetHiNorth() ||
464 tsnorth <= GetLoNorth())
465 {
466 PassA = 10; // EMBER IS OFF THE MAP SURFACE, OUT OF SIMULATION SPACE

468 }
469 posit = fe->GetLandscapeData(tseast, tsnorth, mech.ld);
470 //CellData(tseast, tsnorth, &posit); // retrieve data from nearest raster, including new tree height
471 eH = mech.ld.height * 3.2808;//GetCrownHeight()*3.28;
472 if (eH <= 0.0)
473 eH = 1.0;
474 if (posit != positlast)
475 {
476 posit = fe->GetLandscapeData(tseast, tsnorth, mech.ld);
477 eH = mech.ld.height * 3.28;//GetCrownHeight()*3.28;
478 if (eH <= 0.0)
479 eH = 1.0;
480 zfuel1 = zfuel2;
481 if (mech.ld.fuel > 50) // if fuel is a building or pile
482 zfuel2 = mech.ld.elev + 3; // 3m high fuel pile
483 else
484 zfuel2 = mech.ld.elev; // height of structure will be the fuel height at those locations
485 fe->GetFireEnvironment(env, CurTime, false);//, 0, 1, -1);
486 mech.LoadLocalFEData(fe);
487 // rwinddir=(part4*part1)*SmokeDispersionAngle+mech.lmw.winddir; // >99% of rwinddir will be <= SmoDispAngl
488 }
489 else
490 {
491 if (zt2 > MAXZ) // test if ember drop distance has exceeded burnout time
492 {
493 PassA = 10;
494 break;
495 }
496 else
497 PassA += 2; // terminate this loop early
498 }
499 }
500 seast = tseast;
501 snorth = tsnorth;
502 elapsedtime += SubSStep;
503 Xtot += Xt;
504 zt1 = zt2;
505 if (PassA == 10)
506 {
507 curtime = -1;
508 mZt = 2.0; // cause failure of ember
509 break;
510 }
511 else
512 {
513 if (PassA == 9)
514 mZt = 0.0;
515 if (curtime < EndCurTimeStep)
516 curtime += SubSStep;
517 else
518 break;
519 }
520
521 /************************************
522 if(count<=9000)
523 { filez=fopen(fln1, "a");
524 fprintf(filez,"%d %f %f %f %f %f %f \n",count, Diameter, elapsedtime, Xtot, Z1HtFromStart, Z2HtFromStart, 3.28*zfuel2);
525 if(mZt<1.0)
526 fprintf(filez, "\n");
527 fclose(filez);
528 }
529 ***********************************/
530 }
531 while (mZt > 1.0); // DISTANCE TOLERANCE IS 2m ABOVE TARGET IGNITION SURFACE
532
533 if (mZt >= -1.0 && mZt <= 1.0 && curtime < EndCurTimeStep)// && NumSpots<19000) // NumSpot restriction is TEMPORARY
534 {
535 if (mech.ld.fuel > 0) // if not a rock or lake etc.
536 {
537 if (NumSpots > 0)
538 {
539 //NextSpot = (spotdata *)
540 // CurSpot->next = (spotdata *) new spotdata;//GlobalAlloc(GMEM_FIXED | GMEM_ZEROINIT, 1*sizeof(spotdata));
541 spotdata* t_next;
542 t_next = new spotdata;
543 CurSpot->next = t_next;
544 NextSpot = t_next;
545 CurSpot = NextSpot;
546 }
547 else
548 {
549 FirstSpot = (spotdata *) new spotdata;//GlobalAlloc(GMEM_FIXED | GMEM_ZEROINIT, 1*sizeof(spotdata)); // initialize first ember
550 CurSpot = FirstSpot;
551 }
552 CurSpot->x = tseast; // store ember coordinates
553 CurSpot->y = tsnorth; // UTM meters
554 CurSpot->TimeRem = EndCurTimeStep - curtime;
555 NumSpots++;
556 }
557 }
558 else
559 {
560 if (curtime == EndCurTimeStep)
561 {
562 CarryEmber->x = seast;
563 CarryEmber->y = snorth; // resave ember coordinates
564 CarryEmber->PartDiam = Diameter; // original particle diameter, stays constant
565 CarryEmber->ZHeight = Z; // max height of ember m, stays constant
566 CarryEmber->StartElev = Zelev; // elevation m of land producing ember, stays constant

578 {

582 }
583 if (CarryOverEmbers > 0)

593{
594 // resets linked list if simulation interrupted when embers in air (carryover embers)
595 if (NumEmbers > 0)

672{

692 i++) // for each ember that landed
693 {
694 startx = CurSpot->x; // spots array for coords of ember rest position and potential spots
695 starty = CurSpot->y;
696 inside = 0;
697 GetNumSpots(&newspots, false);
698 for (j = 0; j < newspots; j++) // for each fire
699 {
700 if (GetInout(j) == 2) // search only inward burning fires
701 {
702 NumVertex = GetNumPoints(j); // get bounding rect
703 Xlo = GetPerimeter1Value(j, NumVertex, 0);

705 Ylo = GetPerimeter1Value(j, NumVertex, 2);
706 Yhi = GetPerimeter1Value(j, NumVertex, 3);
707 if (startx > Xlo && startx < Xhi)
708 {
709 if (starty > Ylo && starty < Yhi)
710 {
711 if (NumVertex > 6)
712 {
713 inside = APolygon::Overlap(j);
714 if (inside)
715 break; // ember landed inside inward fire
716 }
717 else
718 {
719 inside = 1;
720 break;
721 }
722 }
723 }
724 }
725 }
726 if (!inside)
727 {
728 for (j = 0; j < newspots; j++) // for each fire
729 {
730 if (GetInout(j) == 0 || GetNumPoints(j) == 0)
731 continue;
732 if (GetInout(j) != 2) // search only outward burning fires and barriers
733 {
734 NumVertex = GetNumPoints(j); // get bounding rect

748 }

751 inside = 1;
752 break;
753 }
754 }
755 }
756 }
757 }
758 }

774 {

783double Embers::torchheight(double partdiam, double zo)
784 // Iteratively calculates particle height z as a function of part diameter D, flame duration, and flame height
785 // for torching trees only (ALBINI 1979)
786{
787 double tf, tt, tT, aT, inc = 1.0, inc2 = 1.0, vowf, z=-1.0, zo1;
788 double Temp;
789
790 //vowf=40.0*pow(partdiam/SteadyHeight,.5);
791 vowf = 40.0 * sqrt(partdiam / SteadyHeight);
792 if (vowf < 1.0)
793 {
794 zo1 = sqrt(zo / SteadyHeight);
795 if (zo1 > vowf)
796 {
797 Temp = (1.0 - vowf) / (sqrt(zo / SteadyHeight) - vowf);
798 if (Temp < 0.0)
799 Temp *= -1.0;
800 tf = 1.0 - sqrt(zo / SteadyHeight) + vowf * log(Temp);
801 tt = .2 * vowf * (1.0 + vowf * log(1.0 + 1.0 / (1.0 - vowf)));
802 aT = Duration + 1.2 - tf - tt;
803 if (aT >= 0.0)
804 {
805 //z=SteadyHeight;
806 z = ++inc * SteadyHeight;
807 tT = partcalc(vowf, z);
808 while (fabs(tT - aT) > .01) // tolerance for iteration is set to .01
809 {
810 if (tT == aT)
811 break;
812 else
813 {
814 if (tT - aT < 0)
815 inc += inc2;
816 else
817 {
818 if (tT - aT > 0)
819 {
820 inc -= inc2;
821 inc2 /= 2.0;
822 inc += inc2;
823 }
824 }
825 }
826 z = inc * SteadyHeight;
827 tT = partcalc(vowf, z);
828 }
829 }
830 }
831 }
832 return z;

837double Embers::partcalc(double vowf, double z)
838 // Calculates tT for iterating max particle height z as a function of particle diameter and flameheight (SteadyHeight)
839 // for torching trees only, function used by torcheight
840{
841 double r, tT, tp, a = 5.963, b = 4.563;
842 double Temp, r2, pt8vowf;
843
844 r2 = (b + z / SteadyHeight) / a;
845 r = sqrt(r2);
846 //tT=a/3.0*(pow((b+z/SteadyHeight)/a,1.5)-1.0);
847 tT = a / 3.0 * (r * r2 - 1.0); // optimized
848 pt8vowf = 0.8 * vowf; // optimized
849 Temp = (1.0 - pt8vowf) / (1.0 - pt8vowf * r);
850 if (Temp < 0.0)
851 Temp *= -1.0;
852 //tp=a/pow(.8*vowf,3.0)*(log(Temp)-(.8*vowf)*(r-1.0)-.5*pow(.8*vowf,2.0)*(pow(r,2.0)-1.0));
853 tp = a /
854 (pow2(pt8vowf) * pt8vowf) * (log(Temp) -
855 pt8vowf * (r - 1.0) -
856 .5 * pow2(pt8vowf) * (r2 - 1.0));
857 tT = tp - tT;
858
859 return tT;
