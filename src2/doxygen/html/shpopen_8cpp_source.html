199/************************************************************************/
200/* SwapWord() */
201/* */
202/* Swap a 2, 4 or 8 byte word. */
203/************************************************************************/
204
205static void SwapWord(int length, void* wordP)
206{
207 int i;
208 uchar temp;
209
210 for (i = 0; i < length / 2; i++)
211 {
212 temp = ((uchar *) wordP)[i];
213 ((uchar *) wordP)[i] = ((uchar *) wordP)[length - i - 1];
214 ((uchar *) wordP)[length - i - 1] = temp;
215 }
216}

233/************************************************************************/
234/* SHPWriteHeader() */
235/* */
236/* Write out a header for the .shp and .shx files as well as the */
237/* contents of the index (.shx) file. */
238/************************************************************************/
239
240static void SHPWriteHeader(SHPHandle psSHP)
241{
242 uchar abyHeader[100];
243 int i;
244 int32 i32;
245 double dValue;
246 int32* panSHX;
247
248 /* -------------------------------------------------------------------- */
249 /* Prepare header block for .shp file. */
250 /* -------------------------------------------------------------------- */
251 for (i = 0; i < 100; i++)
252 abyHeader[i] = 0;
253
254 abyHeader[2] = 0x27; /* magic cookie */
255 abyHeader[3] = 0x0a;
256
257 i32 = psSHP->nFileSize / 2; /* file size */
258 ByteCopy(&i32, abyHeader + 24, 4);
259 if (!bBigEndian)
260 SwapWord(4, abyHeader + 24);
261
262 i32 = 1000; /* version */
263 ByteCopy(&i32, abyHeader + 28, 4);
264 if (bBigEndian)
265 SwapWord(4, abyHeader + 28);
266
267 i32 = psSHP->nShapeType; /* shape type */
268 ByteCopy(&i32, abyHeader + 32, 4);
269 if (bBigEndian)
270 SwapWord(4, abyHeader + 32);
271
272 dValue = psSHP->adBoundsMin[0]; /* set bounds */
273 ByteCopy(&dValue, abyHeader + 36, 8);
274 if (bBigEndian)
275 SwapWord(8, abyHeader + 36);
276
277 dValue = psSHP->adBoundsMin[1];
278 ByteCopy(&dValue, abyHeader + 44, 8);
279 if (bBigEndian)
280 SwapWord(8, abyHeader + 44);
281
282 dValue = psSHP->adBoundsMax[0];
283 ByteCopy(&dValue, abyHeader + 52, 8);
284 if (bBigEndian)
285 SwapWord(8, abyHeader + 52);
286
287 dValue = psSHP->adBoundsMax[1];
288 ByteCopy(&dValue, abyHeader + 60, 8);
289 if (bBigEndian)
290 SwapWord(8, abyHeader + 60);
291
292 dValue = psSHP->adBoundsMin[2]; /* z */
293 ByteCopy(&dValue, abyHeader + 68, 8);
294 if (bBigEndian)
295 SwapWord(8, abyHeader + 68);
296
297 dValue = psSHP->adBoundsMax[2];
298 ByteCopy(&dValue, abyHeader + 76, 8);
299 if (bBigEndian)
300 SwapWord(8, abyHeader + 76);
301
302 dValue = psSHP->adBoundsMin[3]; /* m */
303 ByteCopy(&dValue, abyHeader + 84, 8);
304 if (bBigEndian)
305 SwapWord(8, abyHeader + 84);
306
307 dValue = psSHP->adBoundsMax[3];
308 ByteCopy(&dValue, abyHeader + 92, 8);
309 if (bBigEndian)
310 SwapWord(8, abyHeader + 92);
311
312 /* -------------------------------------------------------------------- */
313 /* Write .shp file header. */
314 /* -------------------------------------------------------------------- */
315 fseek(psSHP->fpSHP, 0, 0);
316 fwrite(abyHeader, 100, 1, psSHP->fpSHP);
317
318 /* -------------------------------------------------------------------- */
319 /* Prepare, and write .shx file header. */
320 /* -------------------------------------------------------------------- */
321 i32 = (psSHP->nRecords * 2 * sizeof(int32) + 100) / 2; /* file size */
322 ByteCopy(&i32, abyHeader + 24, 4);
323 if (!bBigEndian)
324 SwapWord(4, abyHeader + 24);
325
326 fseek(psSHP->fpSHX, 0, 0);
327 fwrite(abyHeader, 100, 1, psSHP->fpSHX);
328
329 /* -------------------------------------------------------------------- */
330 /* Write out the .shx contents. */
331 /* -------------------------------------------------------------------- */
332 panSHX = (int32 *) malloc(sizeof(int32) * 2 * psSHP->nRecords);
333
334 for (i = 0; i < psSHP->nRecords; i++)
335 {
336 panSHX[i * 2] = psSHP->panRecOffset[i] / 2;
337 panSHX[i * 2 + 1] = psSHP->panRecSize[i] / 2;
338 if (!bBigEndian)
339 SwapWord(4, panSHX + i * 2);
340 if (!bBigEndian)
341 SwapWord(4, panSHX + i * 2 + 1);
342 }
343
344 fwrite(panSHX, sizeof(int32) * 2, psSHP->nRecords, psSHP->fpSHX);
345
346 free(panSHX);
347}

349/************************************************************************/
350/* SHPOpen() */
351/* */
352/* Open the .shp and .shx files based on the basename of the */
353/* files or either file name. */
354/************************************************************************/
355
356SHPHandle SHPAPI_CALL SHPOpen(const char* pszLayer, const char* pszAccess)
357{
358 char* pszFullname, * pszBasename;
359 SHPHandle psSHP;
360
361 uchar* pabyBuf;
362 int i;
363 double dValue;
364
365 /* -------------------------------------------------------------------- */
366 /* Ensure the access string is one of the legal ones. We */
367 /* ensure the result string indicates binary to avoid common */
368 /* problems on Windows. */
369 /* -------------------------------------------------------------------- */
370 if (strcmp(pszAccess, "rb+") ==
371 0 ||
372 strcmp(pszAccess, "r+b") ==
373 0 ||
374 strcmp(pszAccess,
375 "r+") ==
376 0)
377 pszAccess = "r+b";
378 else
379 pszAccess = "rb";
380
381 /* -------------------------------------------------------------------- */
382 /* Establish the byte order on this machine. */
383 /* -------------------------------------------------------------------- */
384 i = 1;
385 if (*((uchar *) &i) == 1)
386 bBigEndian = FALSE;
387 else
388 bBigEndian = TRUE;
389
390 /* -------------------------------------------------------------------- */
391 /* Initialize the info structure. */
392 /* -------------------------------------------------------------------- */
393 psSHP = (SHPHandle) malloc(sizeof(SHPInfo));
394
395 psSHP->bUpdated = FALSE;
396
397 /* -------------------------------------------------------------------- */
398 /* Compute the base (layer) name. If there is any extension */
399 /* on the passed in filename we will strip it off. */
400 /* -------------------------------------------------------------------- */
401 pszBasename = (char *) malloc(strlen(pszLayer) + 5);
402 strcpy(pszBasename, pszLayer);
403 for (i = strlen(pszBasename) - 1;
404 i > 0 &&
405 pszBasename[i] != '.' &&
406 pszBasename[i] != '/' &&
407 pszBasename[i] != '\\';
408 i--)
409 {
410 }
411
412 if (pszBasename[i] == '.')
413 pszBasename[i] = '\0';
414
415 /* -------------------------------------------------------------------- */
416 /* Open the .shp and .shx files. Note that files pulled from */
417 /* a PC to Unix with upper case filenames won't work! */
418 /* -------------------------------------------------------------------- */
419 pszFullname = (char *) malloc(strlen(pszBasename) + 5);
420 sprintf(pszFullname, "%s.shp", pszBasename);
421 psSHP->fpSHP = fopen(pszFullname, pszAccess);
422 if (psSHP->fpSHP == NULL)
423 {
424 sprintf(pszFullname, "%s.SHP", pszBasename);
425 psSHP->fpSHP = fopen(pszFullname, pszAccess);
426 }
427
428 if (psSHP->fpSHP == NULL)
429 {
430 free(psSHP);
431 free(pszBasename);
432 free(pszFullname);
433 return(NULL);
434 }
435
436 sprintf(pszFullname, "%s.shx", pszBasename);
437 psSHP->fpSHX = fopen(pszFullname, pszAccess);
438 if (psSHP->fpSHX == NULL)
439 {
440 sprintf(pszFullname, "%s.SHX", pszBasename);
441 psSHP->fpSHX = fopen(pszFullname, pszAccess);
442 }
443
444 if (psSHP->fpSHX == NULL)
445 {
446 fclose(psSHP->fpSHX);
447 free(psSHP);
448 free(pszBasename);
449 free(pszFullname);
450 return(NULL);
451 }
452
453 free(pszFullname);
454 free(pszBasename);
455
456 /* -------------------------------------------------------------------- */
457 /* Read the file size from the SHP file. */
458 /* -------------------------------------------------------------------- */
459 pabyBuf = (uchar *) malloc(100);
460 fread(pabyBuf, 100, 1, psSHP->fpSHP);
461
462 psSHP->nFileSize = (pabyBuf[24] * 256 * 256 * 256 +
463 pabyBuf[25] * 256 * 256 +
464 pabyBuf[26] * 256 +
465 pabyBuf[27]) * 2;
466
467 /* -------------------------------------------------------------------- */
468 /* Read SHX file Header info */
469 /* -------------------------------------------------------------------- */
470 fread(pabyBuf, 100, 1, psSHP->fpSHX);
471
472 if (pabyBuf[0] != 0 ||
473 pabyBuf[1] != 0 ||
474 pabyBuf[2] != 0x27 ||
475 (pabyBuf[3] != 0x0a && pabyBuf[3] != 0x0d))
476 {
477 fclose(psSHP->fpSHP);
478 fclose(psSHP->fpSHX);
479 free(psSHP);
480
481 return(NULL);
482 }
483
484 psSHP->nRecords = pabyBuf[27] +
485 pabyBuf[26] * 256 +
486 pabyBuf[25] * 256 * 256 +
487 pabyBuf[24] * 256 * 256 * 256;
488 psSHP->nRecords = (psSHP->nRecords * 2 - 100) / 8;
489
490 psSHP->nShapeType = pabyBuf[32];
491
492 if (psSHP->nRecords <0 || psSHP->nRecords> 256000000)
493 {
494 /* this header appears to be corrupt. Give up. */
495 fclose(psSHP->fpSHP);
496 fclose(psSHP->fpSHX);
497 free(psSHP);
498
499 return(NULL);
500 }
501
502 /* -------------------------------------------------------------------- */
503 /* Read the bounds. */
504 /* -------------------------------------------------------------------- */
505 if (bBigEndian)
506 SwapWord(8, pabyBuf + 36);
507 memcpy(&dValue, pabyBuf + 36, 8);
508 psSHP->adBoundsMin[0] = dValue;
509
510 if (bBigEndian)
511 SwapWord(8, pabyBuf + 44);
512 memcpy(&dValue, pabyBuf + 44, 8);
513 psSHP->adBoundsMin[1] = dValue;
514
515 if (bBigEndian)
516 SwapWord(8, pabyBuf + 52);
517 memcpy(&dValue, pabyBuf + 52, 8);
518 psSHP->adBoundsMax[0] = dValue;
519
520 if (bBigEndian)
521 SwapWord(8, pabyBuf + 60);
522 memcpy(&dValue, pabyBuf + 60, 8);
523 psSHP->adBoundsMax[1] = dValue;
524
525 if (bBigEndian)
526 SwapWord(8, pabyBuf + 68); /* z */
527 memcpy(&dValue, pabyBuf + 68, 8);
528 psSHP->adBoundsMin[2] = dValue;
529
530 if (bBigEndian)
531 SwapWord(8, pabyBuf + 76);
532 memcpy(&dValue, pabyBuf + 76, 8);
533 psSHP->adBoundsMax[2] = dValue;
534
535 if (bBigEndian)
536 SwapWord(8, pabyBuf + 84); /* z */
537 memcpy(&dValue, pabyBuf + 84, 8);
538 psSHP->adBoundsMin[3] = dValue;
539
540 if (bBigEndian)
541 SwapWord(8, pabyBuf + 92);
542 memcpy(&dValue, pabyBuf + 92, 8);
543 psSHP->adBoundsMax[3] = dValue;
544
545 free(pabyBuf);
546
547 /* -------------------------------------------------------------------- */
548 /* Read the .shx file to get the offsets to each record in */
549 /* the .shp file. */
550 /* -------------------------------------------------------------------- */
551 psSHP->nMaxRecords = psSHP->nRecords;
552
553 psSHP->panRecOffset = (int *)
554 malloc(sizeof(int) * MAX(1, psSHP->nMaxRecords));
555 psSHP->panRecSize = (int *)
556 malloc(sizeof(int) * MAX(1, psSHP->nMaxRecords));
557
558 pabyBuf = (uchar *) malloc(8 * MAX(1, psSHP->nRecords));
559 fread(pabyBuf, 8, psSHP->nRecords, psSHP->fpSHX);
560
561 for (i = 0; i < psSHP->nRecords; i++)
562 {
563 int32 nOffset, nLength;
564
565 memcpy(&nOffset, pabyBuf + i * 8, 4);
566 if (!bBigEndian)
567 SwapWord(4, &nOffset);
568
569 memcpy(&nLength, pabyBuf + i * 8 + 4, 4);
570 if (!bBigEndian)
571 SwapWord(4, &nLength);
572
573 psSHP->panRecOffset[i] = nOffset * 2;
574 psSHP->panRecSize[i] = nLength * 2;
575 }
576 free(pabyBuf);
577
578 return(psSHP);
579}

581/************************************************************************/
582/* SHPClose() */
583/* */
584/* Close the .shp and .shx files. */
585/************************************************************************/
586
587void SHPAPI_CALL SHPClose(SHPHandle psSHP)
588{
589 /* -------------------------------------------------------------------- */
590 /* Update the header if we have modified anything. */
591 /* -------------------------------------------------------------------- */
592 if (psSHP->bUpdated)
593 {
594 SHPWriteHeader(psSHP);
595 }
596
597 /* -------------------------------------------------------------------- */
598 /* Free all resources, and close files. */
599 /* -------------------------------------------------------------------- */
600 free(psSHP->panRecOffset);
601 free(psSHP->panRecSize);
602
603 fclose(psSHP->fpSHX);
604 fclose(psSHP->fpSHP);
605
606 free(psSHP);
607
608 if (pabyRec != NULL)
609 {
610 free(pabyRec);
611 pabyRec = NULL;
612 nBufSize = 0;
613 }
614}

657

662 if (*((uchar *) &i) == 1)
663 bBigEndian = FALSE;

695 fpSHX = fopen(pszFullname, "wb");

703 /* Prepare header block for .shp file. */
704 /* -------------------------------------------------------------------- */
705 for (i = 0; i < 100; i++)

709 abyHeader[3] = 0x0a;

711 i32 = 50; /* file size */
712 ByteCopy(&i32, abyHeader + 24, 4);
713 if (!bBigEndian)
714 SwapWord(4, abyHeader + 24);

720
721 i32 = nShapeType; /* shape type */
722 ByteCopy(&i32, abyHeader + 32, 4);
723 if (bBigEndian)
724 SwapWord(4, abyHeader + 32);
725
726 dValue = 0.0; /* set bounds */
727 ByteCopy(&dValue, abyHeader + 36, 8);
728 ByteCopy(&dValue, abyHeader + 44, 8);
729 ByteCopy(&dValue, abyHeader + 52, 8);
730 ByteCopy(&dValue, abyHeader + 60, 8);
731
732 /* -------------------------------------------------------------------- */
733 /* Write .shp file header. */
734 /* -------------------------------------------------------------------- */
735 fwrite(abyHeader, 100, 1, fpSHP);
736
737 /* -------------------------------------------------------------------- */
738 /* Prepare, and write .shx file header. */
739 /* -------------------------------------------------------------------- */
740 i32 = 50; /* file size */
741 ByteCopy(&i32, abyHeader + 24, 4);
742 if (!bBigEndian)
743 SwapWord(4, abyHeader + 24);
744
745 fwrite(abyHeader, 100, 1, fpSHX);
746
747 /* -------------------------------------------------------------------- */
748 /* Close the files, and then open them as regular existing files. */
749 /* -------------------------------------------------------------------- */
750 fclose(fpSHP);
751 fclose(fpSHX);
752
753 return(SHPOpen(pszLayer, "r+b"));
754}

815/************************************************************************/
816/* SHPCreateObject() */
817/* */
818/* Create a shape object. It should be freed with */
819/* SHPDestroyObject(). */
820/************************************************************************/
821
822SHPObject SHPAPI_CALL1(*)
823SHPCreateObject(int nSHPType, int nShapeId, int nParts, int* panPartStart,
824 int* panPartType, int nVertices, double* padfX, double* padfY,
825 double* padfZ, double* padfM)
826
827{
828SHPObject *psObject;
829int i, bHasM, bHasZ;
830
831 psObject = (SHPObject*) calloc(1, sizeof(SHPObject));
832psObject-> nSHPType = nSHPType;
833psObject-> nShapeId = nShapeId;
834
835/* -------------------------------------------------------------------- */
836/* Establish whether this shape type has M, and Z values. */
837/* -------------------------------------------------------------------- */
838if (nSHPType == SHPT_ARCM ||
839 nSHPType == SHPT_POINTM ||
840 nSHPType == SHPT_POLYGONM ||
841 nSHPType == SHPT_MULTIPOINTM)
842{
843 bHasM = TRUE;
844 bHasZ = FALSE;
845}
846else if (nSHPType == SHPT_ARCZ ||
847 nSHPType == SHPT_POINTZ ||
848 nSHPType == SHPT_POLYGONZ ||
849 nSHPType == SHPT_MULTIPOINTZ ||
850 nSHPType == SHPT_MULTIPATCH)
851{
852 bHasM = TRUE;
853 bHasZ = TRUE;
854}
855else
856{
857 bHasM = FALSE;
858 bHasZ = FALSE;
859}
860
861/* -------------------------------------------------------------------- */
862/* Capture parts. Note that part type is optional, and */
863/* defaults to ring. */
864/* -------------------------------------------------------------------- */
865if (nSHPType == SHPT_ARC ||
866 nSHPType == SHPT_POLYGON ||
867 nSHPType == SHPT_ARCM ||
868 nSHPType == SHPT_POLYGONM ||
869 nSHPType == SHPT_ARCZ ||
870 nSHPType == SHPT_POLYGONZ ||
871 nSHPType == SHPT_MULTIPATCH)
872{
873 psObject->nParts = MAX(1, nParts);
874
875 psObject->panPartStart = (int *) malloc(sizeof(int) * psObject->nParts);
876 psObject->panPartType = (int *) malloc(sizeof(int) * psObject->nParts);
877
878 psObject->panPartStart[0] = 0;
879 psObject->panPartType[0] = SHPP_RING;
880
881 for (i = 0; i < nParts; i++)
882 {
883 psObject->panPartStart[i] = panPartStart[i];
884 if (panPartType != NULL)
885 psObject->panPartType[i] = panPartType[i];
886 else
887 psObject->panPartType[i] = SHPP_RING;
888 }
889}
890
891/* -------------------------------------------------------------------- */
892/* Capture vertices. Note that Z and M are optional, but X and */
893/* Y are not. */
894/* -------------------------------------------------------------------- */
895if (nVertices > 0)
896{
897 psObject->padfX = (double *) calloc(sizeof(double), nVertices);
898 psObject->padfY = (double *) calloc(sizeof(double), nVertices);
899 psObject->padfZ = (double *) calloc(sizeof(double), nVertices);
900 psObject->padfM = (double *) calloc(sizeof(double), nVertices);
901
902 assert(padfX != NULL);
903 assert(padfY != NULL);
904
905 for (i = 0; i < nVertices; i++)
906 {
907 psObject->padfX[i] = padfX[i];
908 psObject->padfY[i] = padfY[i];
909 if (padfZ != NULL && bHasZ)
910 psObject->padfZ[i] = padfZ[i];
911 if (padfM != NULL && bHasM)
912 psObject->padfM[i] = padfM[i];
913 }
914}
915
916/* -------------------------------------------------------------------- */
917/* Compute the extents. */
918/* -------------------------------------------------------------------- */
919psObject-> nVertices = nVertices;
920SHPComputeExtents(psObject);
921
922return(psObject);
923}

941/************************************************************************/
942/* SHPWriteObject() */
943/* */
944/* Write out the vertices of a new structure. Note that it is */
945/* only possible to write vertices at the end of the file. */
946/************************************************************************/
947
948int SHPAPI_CALL SHPWriteObject(SHPHandle psSHP, int nShapeId,
949 SHPObject* psObject)
950{
951 int nRecordOffset, i, nRecordSize;
952 uchar* pabyRec;
953 int32 i32;
954
955 psSHP->bUpdated = TRUE;
956
957 /* -------------------------------------------------------------------- */
958 /* Ensure that shape object matches the type of the file it is */
959 /* being written to. */
960 /* -------------------------------------------------------------------- */
961 assert(psObject->nSHPType == psSHP->nShapeType ||
962 psObject->nSHPType == SHPT_NULL);
963
964 /* -------------------------------------------------------------------- */
965 /* Ensure that -1 is used for appends. Either blow an */
966 /* assertion, or if they are disabled, set the shapeid to -1 */
967 /* for appends. */
968 /* -------------------------------------------------------------------- */
969 assert(nShapeId == -1 || (nShapeId >= 0 && nShapeId < psSHP->nRecords));
970
971 if (nShapeId != -1 && nShapeId >= psSHP->nRecords)
972 nShapeId = -1;
973
974 /* -------------------------------------------------------------------- */
975 /* Add the new entity to the in memory index. */
976 /* -------------------------------------------------------------------- */
977 if (nShapeId == -1 && psSHP->nRecords + 1 > psSHP->nMaxRecords)
978 {
979 psSHP->nMaxRecords = (int) (psSHP->nMaxRecords * 1.3 + 100);
980
981 psSHP->panRecOffset = (int *) SfRealloc(psSHP->panRecOffset,
982 sizeof(int) * psSHP->nMaxRecords);
983 psSHP->panRecSize = (int *) SfRealloc(psSHP->panRecSize,
984 sizeof(int) * psSHP->nMaxRecords);
985 }
986
987 /* -------------------------------------------------------------------- */
988 /* Initialize record. */
989 /* -------------------------------------------------------------------- */
990 pabyRec = (uchar *) malloc(psObject->nVertices * 4 * sizeof(double) +
991 psObject->nParts * 8 +
992 128);
993
994 /* -------------------------------------------------------------------- */
995 /* Extract vertices for a Polygon or Arc. */
996 /* -------------------------------------------------------------------- */
997 if (psObject->nSHPType == SHPT_POLYGON ||
998 psObject->nSHPType == SHPT_POLYGONZ ||
999 psObject->nSHPType == SHPT_POLYGONM ||
1000 psObject->nSHPType == SHPT_ARC ||
1001 psObject->nSHPType == SHPT_ARCZ ||
1002 psObject->nSHPType == SHPT_ARCM ||
1003 psObject->nSHPType == SHPT_MULTIPATCH)
1004 {
1005 int32 nPoints, nParts;
1006 int i;
1007
1008 nPoints = psObject->nVertices;
1009 nParts = psObject->nParts;
1010
1011 _SHPSetBounds(pabyRec + 12, psObject);
1012
1013 if (bBigEndian)
1014 SwapWord(4, &nPoints);
1015 if (bBigEndian)
1016 SwapWord(4, &nParts);
1017
1018 ByteCopy(&nPoints, pabyRec + 40 + 8, 4);
1019 ByteCopy(&nParts, pabyRec + 36 + 8, 4);
1020
1021 nRecordSize = 52;
1022
1023 /*
1024 * Write part start positions.
1025 */
1026 ByteCopy(psObject->panPartStart, pabyRec + 44 + 8,
1027 4 * psObject->nParts);
1028 for (i = 0; i < psObject->nParts; i++)
1029 {
1030 if (bBigEndian)
1031 SwapWord(4, pabyRec + 44 + 8 + 4 * i);
1032 nRecordSize += 4;
1033 }
1034
1035 /*
1036 * Write multipatch part types if needed.
1037 */
1038 if (psObject->nSHPType == SHPT_MULTIPATCH)
1039 {
1040 memcpy(pabyRec + nRecordSize, psObject->panPartType,
1041 4 * psObject->nParts);
1042 for (i = 0; i < psObject->nParts; i++)
1043 {
1044 if (bBigEndian)
1045 SwapWord(4, pabyRec + nRecordSize);
1046 nRecordSize += 4;
1047 }
1048 }
1049
1050 /*
1051 * Write the (x,y) vertex values.
1052 */
1053 for (i = 0; i < psObject->nVertices; i++)
1054 {
1055 ByteCopy(psObject->padfX + i, pabyRec + nRecordSize, 8);
1056 ByteCopy(psObject->padfY + i, pabyRec + nRecordSize + 8, 8);
1057
1058 if (bBigEndian)
1059 SwapWord(8, pabyRec + nRecordSize);
1060
1061 if (bBigEndian)
1062 SwapWord(8, pabyRec + nRecordSize + 8);
1063
1064 nRecordSize += 2 * 8;
1065 }
1066
1067 /*
1068 * Write the Z coordinates (if any).
1069 */
1070 if (psObject->nSHPType == SHPT_POLYGONZ ||
1071 psObject->nSHPType == SHPT_ARCZ ||
1072 psObject->nSHPType == SHPT_MULTIPATCH)
1073 {
1074 ByteCopy(&(psObject->dfZMin), pabyRec + nRecordSize, 8);
1075 if (bBigEndian)
1076 SwapWord(8, pabyRec + nRecordSize);
1077 nRecordSize += 8;
1078
1079 ByteCopy(&(psObject->dfZMax), pabyRec + nRecordSize, 8);
1080 if (bBigEndian)
1081 SwapWord(8, pabyRec + nRecordSize);
1082 nRecordSize += 8;
1083
1084 for (i = 0; i < psObject->nVertices; i++)
1085 {
1086 ByteCopy(psObject->padfZ + i, pabyRec + nRecordSize, 8);
1087 if (bBigEndian)
1088 SwapWord(8, pabyRec + nRecordSize);
1089 nRecordSize += 8;
1090 }
1091 }
1092
1093 /*
1094 * Write the M values, if any.
1095 */
1096 if (psObject->nSHPType == SHPT_POLYGONM ||
1097 psObject->nSHPType == SHPT_ARCM
1098#ifndef DISABLE_MULTIPATCH_MEASURE
1099 ||
1100 psObject->nSHPType == SHPT_MULTIPATCH
1101#endif
1102 ||
1103 psObject->nSHPType == SHPT_POLYGONZ ||
1104 psObject->nSHPType == SHPT_ARCZ)
1105 {
1106 ByteCopy(&(psObject->dfMMin), pabyRec + nRecordSize, 8);
1107 if (bBigEndian)
1108 SwapWord(8, pabyRec + nRecordSize);
1109 nRecordSize += 8;
1110
1111 ByteCopy(&(psObject->dfMMax), pabyRec + nRecordSize, 8);
1112 if (bBigEndian)
1113 SwapWord(8, pabyRec + nRecordSize);
1114 nRecordSize += 8;
1115
1116 for (i = 0; i < psObject->nVertices; i++)
1117 {
1118 ByteCopy(psObject->padfM + i, pabyRec + nRecordSize, 8);
1119 if (bBigEndian)
1120 SwapWord(8, pabyRec + nRecordSize);
1121 nRecordSize += 8;
1122 }
1123 }
1124 }
1125
1126 /* -------------------------------------------------------------------- */
1127 /* Extract vertices for a MultiPoint. */
1128 /* -------------------------------------------------------------------- */
1129 else if (psObject->nSHPType == SHPT_MULTIPOINT ||
1130 psObject->nSHPType == SHPT_MULTIPOINTZ ||
1131 psObject->nSHPType == SHPT_MULTIPOINTM)
1132 {
1133 int32 nPoints;
1134 int i;
1135
1136 nPoints = psObject->nVertices;
1137
1138 _SHPSetBounds(pabyRec + 12, psObject);
1139
1140 if (bBigEndian)
1141 SwapWord(4, &nPoints);
1142 ByteCopy(&nPoints, pabyRec + 44, 4);
1143
1144 for (i = 0; i < psObject->nVertices; i++)
1145 {
1146 ByteCopy(psObject->padfX + i, pabyRec + 48 + i * 16, 8);
1147 ByteCopy(psObject->padfY + i, pabyRec + 48 + i * 16 + 8, 8);
1148
1149 if (bBigEndian)
1150 SwapWord(8, pabyRec + 48 + i * 16);
1151 if (bBigEndian)
1152 SwapWord(8, pabyRec + 48 + i * 16 + 8);
1153 }
1154
1155 nRecordSize = 48 + 16 * psObject->nVertices;
1156
1157 if (psObject->nSHPType == SHPT_MULTIPOINTZ)
1158 {
1159 ByteCopy(&(psObject->dfZMin), pabyRec + nRecordSize, 8);
1160 if (bBigEndian)
1161 SwapWord(8, pabyRec + nRecordSize);
1162 nRecordSize += 8;
1163
1164 ByteCopy(&(psObject->dfZMax), pabyRec + nRecordSize, 8);
1165 if (bBigEndian)
1166 SwapWord(8, pabyRec + nRecordSize);
1167 nRecordSize += 8;
1168
1169 for (i = 0; i < psObject->nVertices; i++)
1170 {
1171 ByteCopy(psObject->padfZ + i, pabyRec + nRecordSize, 8);
1172 if (bBigEndian)
1173 SwapWord(8, pabyRec + nRecordSize);
1174 nRecordSize += 8;
1175 }
1176 }
1177
1178 if (psObject->nSHPType == SHPT_MULTIPOINTZ ||
1179 psObject->nSHPType == SHPT_MULTIPOINTM)
1180 {
1181 ByteCopy(&(psObject->dfMMin), pabyRec + nRecordSize, 8);
1182 if (bBigEndian)
1183 SwapWord(8, pabyRec + nRecordSize);
1184 nRecordSize += 8;
1185
1186 ByteCopy(&(psObject->dfMMax), pabyRec + nRecordSize, 8);
1187 if (bBigEndian)
1188 SwapWord(8, pabyRec + nRecordSize);
1189 nRecordSize += 8;
1190
1191 for (i = 0; i < psObject->nVertices; i++)
1192 {
1193 ByteCopy(psObject->padfM + i, pabyRec + nRecordSize, 8);
1194 if (bBigEndian)
1195 SwapWord(8, pabyRec + nRecordSize);
1196 nRecordSize += 8;
1197 }
1198 }
1199 }
1200
1201 /* -------------------------------------------------------------------- */
1202 /* Write point. */
1203 /* -------------------------------------------------------------------- */
1204 else if (psObject->nSHPType == SHPT_POINT ||
1205 psObject->nSHPType == SHPT_POINTZ ||
1206 psObject->nSHPType == SHPT_POINTM)
1207 {
1208 ByteCopy(psObject->padfX, pabyRec + 12, 8);
1209 ByteCopy(psObject->padfY, pabyRec + 20, 8);
1210
1211 if (bBigEndian)
1212 SwapWord(8, pabyRec + 12);
1213 if (bBigEndian)
1214 SwapWord(8, pabyRec + 20);
1215
1216 nRecordSize = 28;
1217
1218 if (psObject->nSHPType == SHPT_POINTZ)
1219 {
1220 ByteCopy(psObject->padfZ, pabyRec + nRecordSize, 8);
1221 if (bBigEndian)
1222 SwapWord(8, pabyRec + nRecordSize);
1223 nRecordSize += 8;
1224 }
1225
1226 if (psObject->nSHPType == SHPT_POINTZ ||
1227 psObject->nSHPType == SHPT_POINTM)
1228 {
1229 ByteCopy(psObject->padfM, pabyRec + nRecordSize, 8);
1230 if (bBigEndian)
1231 SwapWord(8, pabyRec + nRecordSize);
1232 nRecordSize += 8;
1233 }
1234 }
1235
1236 /* -------------------------------------------------------------------- */
1237 /* Not much to do for null geometries. */
1238 /* -------------------------------------------------------------------- */
1239 else if (psObject->nSHPType == SHPT_NULL)
1240 {
1241 nRecordSize = 12;
1242 }
1243 else
1244 {
1245 /* unknown type */
1246 assert(FALSE);
1247 }
1248
1249 /* -------------------------------------------------------------------- */
1250 /* Establish where we are going to put this record. If we are */
1251 /* rewriting and existing record, and it will fit, then put it */
1252 /* back where the original came from. Otherwise write at the end. */
1253 /* -------------------------------------------------------------------- */
1254 if (nShapeId == -1 || psSHP->panRecSize[nShapeId] < nRecordSize - 8)
1255 {
1256 if (nShapeId == -1)
1257 nShapeId = psSHP->nRecords++;
1258
1259 psSHP->panRecOffset[nShapeId] = nRecordOffset = psSHP->nFileSize;
1260 psSHP->panRecSize[nShapeId] = nRecordSize - 8;
1261 psSHP->nFileSize += nRecordSize;
1262 }
1263 else
1264 {
1265 nRecordOffset = psSHP->panRecOffset[nShapeId];
1266 }
1267
1268 /* -------------------------------------------------------------------- */
1269 /* Set the shape type, record number, and record size. */
1270 /* -------------------------------------------------------------------- */
1271 i32 = nShapeId + 1; /* record # */
1272 if (!bBigEndian)
1273 SwapWord(4, &i32);
1274 ByteCopy(&i32, pabyRec, 4);
1275
1276 i32 = (nRecordSize - 8) / 2; /* record size */
1277 if (!bBigEndian)
1278 SwapWord(4, &i32);
1279 ByteCopy(&i32, pabyRec + 4, 4);
1280
1281 i32 = psObject->nSHPType; /* shape type */
1282 if (bBigEndian)
1283 SwapWord(4, &i32);
1284 ByteCopy(&i32, pabyRec + 8, 4);
1285
1286 /* -------------------------------------------------------------------- */
1287 /* Write out record. */
1288 /* -------------------------------------------------------------------- */
1289 if (fseek(psSHP->fpSHP, nRecordOffset, 0) != 0 || fwrite(pabyRec,
1290 nRecordSize, 1,
1291 psSHP->fpSHP) < 1)
1292 {
1293 printf("Error in fseek() or fwrite().\n\n"); free(pabyRec);
1294 return -1;
1295 }
1296
1297 free(pabyRec);
1298
1299 /* -------------------------------------------------------------------- */
1300 /* Expand file wide bounds based on this shape. */
1301 /* -------------------------------------------------------------------- */
1302 if (psSHP->adBoundsMin[0] == 0.0 &&
1303 psSHP->adBoundsMax[0] == 0.0 &&
1304 psSHP->adBoundsMin[1] == 0.0 &&
1305 psSHP->adBoundsMax[1] == 0.0 &&
1306 psObject->nSHPType != SHPT_NULL)
1307 {
1308 psSHP->adBoundsMin[0] = psSHP->adBoundsMax[0] = psObject->padfX[0];
1309 psSHP->adBoundsMin[1] = psSHP->adBoundsMax[1] = psObject->padfY[0];
1310 psSHP->adBoundsMin[2] = psSHP->adBoundsMax[2] = psObject->padfZ[0];
1311 psSHP->adBoundsMin[3] = psSHP->adBoundsMax[3] = psObject->padfM[0];
1312 }
1313
1314 for (i = 0; i < psObject->nVertices; i++)
1315 {
1316 psSHP->adBoundsMin[0] = MIN(psSHP->adBoundsMin[0], psObject->padfX[i]);
1317 psSHP->adBoundsMin[1] = MIN(psSHP->adBoundsMin[1], psObject->padfY[i]);
1318 psSHP->adBoundsMin[2] = MIN(psSHP->adBoundsMin[2], psObject->padfZ[i]);
1319 psSHP->adBoundsMin[3] = MIN(psSHP->adBoundsMin[3], psObject->padfM[i]);
1320 psSHP->adBoundsMax[0] = MAX(psSHP->adBoundsMax[0], psObject->padfX[i]);
1321 psSHP->adBoundsMax[1] = MAX(psSHP->adBoundsMax[1], psObject->padfY[i]);
1322 psSHP->adBoundsMax[2] = MAX(psSHP->adBoundsMax[2], psObject->padfZ[i]);
1323 psSHP->adBoundsMax[3] = MAX(psSHP->adBoundsMax[3], psObject->padfM[i]);
1324 }
1325
1326 return(nShapeId);
1327}

1329/************************************************************************/
1330/* SHPReadObject() */
1331/* */
1332/* Read the vertices, parts, and other non-attribute information */
1333/* for one shape. */
1334/************************************************************************/
1335
1336SHPObject SHPAPI_CALL1(*)
1337SHPReadObject(SHPHandle psSHP, int hEntity)
1338
1339{
1340SHPObject *psShape;
1341
1342/* -------------------------------------------------------------------- */
1343/* Validate the record/entity number. */
1344/* -------------------------------------------------------------------- */
1345if (hEntity < 0 || hEntity >= psSHP->nRecords)
1346 return(NULL);
1347
1348/* -------------------------------------------------------------------- */
1349/* Ensure our record buffer is large enough. */
1350/* -------------------------------------------------------------------- */
1351if (psSHP->panRecSize[hEntity] + 8 > nBufSize)
1352{
1353 nBufSize = psSHP->panRecSize[hEntity] + 8;
1354 pabyRec = (uchar *) SfRealloc(pabyRec, nBufSize);
1355}
1356
1357/* -------------------------------------------------------------------- */
1358/* Read the record. */
1359/* -------------------------------------------------------------------- */
1360fseek(psSHP->fpSHP, psSHP->panRecOffset[hEntity], 0);
1361fread(pabyRec, psSHP->panRecSize[hEntity] + 8, 1, psSHP->fpSHP);
1362
1363/* -------------------------------------------------------------------- */
1364/* Allocate and minimally initialize the object. */
1365/* -------------------------------------------------------------------- */
1366 psShape = (SHPObject*) calloc(1, sizeof(SHPObject));
1367psShape-> nShapeId = hEntity;
1368
1369memcpy(&psShape->nSHPType, pabyRec + 8, 4);
1370if (bBigEndian)
1371 SwapWord(4, &(psShape->nSHPType));
1372
1373/* ==================================================================== */
1374/* Extract vertices for a Polygon or Arc. */
1375/* ==================================================================== */
