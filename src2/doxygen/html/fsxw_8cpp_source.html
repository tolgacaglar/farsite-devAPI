534bool AllocFirePerims(long num)
535{
536 FreeAllFirePerims();

555bool ReAllocFirePerims()
556{
557 long i, OldNumAlloc;
558 long* newinout, * newnumpts;
559 double* temp1;
560 double** newperim1;
561
562 newperim1 = perimeter1;
563 newinout = inout;
564 newnumpts = numpts;
565
566 perimeter1 = 0;
567 inout = 0;
568 numpts = 0;
569 OldNumAlloc = NumPerimAlloc;
570
571 if (!AllocFirePerims(NumPerimAlloc + SIZE_FIREPERIM_BLOCK))
572 return false;
573
574 if (newinout)
575 {
576 memcpy(inout, newinout, OldNumAlloc * sizeof(long));
577 delete[] newinout;
578 }
579 if (newnumpts)
580 {
581 memcpy(numpts, newnumpts, OldNumAlloc * sizeof(long));
582 delete[] newnumpts;
583 }
584
585 if (newperim1)
586 {
587 for (i = 0; i < OldNumAlloc; i++)
588 {
589 temp1 = perimeter1[i];
590 perimeter1[i] = newperim1[i];
591 if (numpts[i] > 0)
592 delete[] temp1;
593 }
594 delete[] newperim1;
595 }
596
597
598 return true;
599}

691long SwapFirePerims(long NumFire1, long NumFire2)
692{
693 double* TempFire;
694 long TempInout, TempNum;
695
696 if (NumFire1 >= 0 && NumFire2 >= 0) // two fires in perim1
697 {
698 TempFire = perimeter1[NumFire2];
699 perimeter1[NumFire2] = perimeter1[NumFire1];
700 perimeter1[NumFire1] = TempFire;
701 TempInout = inout[NumFire2];
702 inout[NumFire2] = inout[NumFire1];
703 inout[NumFire1] = TempInout;
704 TempNum = numpts[NumFire2];
705 numpts[NumFire2] = numpts[NumFire1];
706 numpts[NumFire1] = TempNum;
707
708 return 1;
709 }
710 else if (NumFire1 < 0 && NumFire2 >= 0)
711 {
712 AllocPerimeter2(numpts[NumFire2]);
713 if (memmove(perimeter2, perimeter1[NumFire2],
714 numpts[NumFire2] * NUMDATA * sizeof(double)))
715 return 1;
716 else
717 return 0;
718 }
719 else if (NumFire1 >= 0 && NumFire2 < 0)
720 {
721 if (perimeter1[NumFire1])
722 {
723 if (memmove(perimeter1[NumFire1], perimeter2,
724 (NumFire2 * -1) * NUMDATA * sizeof(double)))
725 return 1;
726 }
727 return 0;
728 }
729
730 return 0;
731}
