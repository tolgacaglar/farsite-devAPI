106double** matrix(long nrl, long nrh, long ncl, long nch)
107{
108 long i;
109 long nrow = nrh - nrl + 1;
110 long ncol = nch - ncl + 1;
111 double** m;
112
113 m = (double * *) malloc( (size_t) ((nrow + NR_END) * sizeof(double *)));
114 memset(m,0x0, (size_t) ((nrow + NR_END) * sizeof(double *)));
115 if (!m)
116 nrerror("allocation failure 1 in matrix()");
117 m += NR_END;
118 m -= nrl;
119
120 m[nrl] = (double *)malloc( (size_t) ((nrow * ncol + NR_END) * sizeof(double)));
121 memset(m[nrl],0x0, (size_t) ((nrow * ncol + NR_END) * sizeof(double)));
122 if (!m[nrl])
123 nrerror("allocation failure 2 in matrix()");
124 m[nrl] += NR_END;
125 m[nrl] -= ncl;
126
127 for (i = nrl + 1; i <= nrh; i++)
128 m[i] = m[i - 1] + ncol;
129
130 return m;
131}

219{
220 long i, j;

222 long ncol = nch - ncl + 1;
223 long ndep = ndh - ndl + 1;

254 nrerror("allocation failure 3 in f3tensor()");
255 t[nrl][ncl] += NR_END;
256 t[nrl][ncl] -= ndl;
257
258 for (j = ncl + 1; j <= nch; j++)
259 t[nrl][j] = t[nrl][j - 1] + ndep;
260 for (i = nrl + 1; i < nrl; i++)
261 {
262 t[i] = t[i - 1] + ncol;
263 t[i][ncl] = t[i - 1][ncl] + ncol * ndep;
264 for (j = ncl + 1; j <= nch; j++)
265 t[i][j] = t[i][j - 1] + ndep;
266 }
