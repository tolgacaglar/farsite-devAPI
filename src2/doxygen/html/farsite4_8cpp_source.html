285void TFarsiteInterface::SetInputsFromFile(char* settingsfilename)
286{
287 FILE* insettings;
288 char Line[256];
289 char* name;
290 char* value;
291 insettings = fopen(settingsfilename, "r");
292 if(insettings){
293 Inputs.ResetData();
294 }
295 else{
296 printf("cannot open %s",settingsfilename);
297 return;
298 }
299 while (!feof(insettings))
300 {
301 memset(Line, 0x0, 256 * sizeof(char));
302 fgets(Line, 255, insettings);
303 if (strlen(Line) == 0 || !strncmp(Line, "\n", 1))
304 {
305 continue;
306 }
307 if (Line[0] == '#')
308 {
309 continue;
310 }
311 name = Line;
312 value = strchr(Line, '=');
313 if (value != NULL)
314 {
315 value[0]='\0';
316 value++;
317 }
318 else
319 {
320 continue;
321 }
322 trim(name);
323 trim(value);
324 if (strlen(value) == 0)
325 {
326 continue;
327 }
328 processNameValue(name,value);
329
330 }
331 ResetDuration();
332 FlatSimulateInitiateTerminate();
333 //SetModelParams();
334 //on these in the start data case and end date case
335 SetEndDate(GetJulianDays(GetEndMonth()) + GetEndDay());
336 SetStartDate(GetJulianDays(GetStartMonth()) + GetStartDay()+GetStartHour());
