224bool MechCalls::NormalizeDirectionWithLocalSlope()
225{
226 double Theta;
227 double Angle;
228 double hx, hy, ds;
229 double XSign = 1.0, YSign = 1.0;
230 double Mx = -1.0, My = -1.0;
231 double slopef = ((double) ld.slope / 180.0) * PI;
232 // double PI=3.14159265358979324;
233 double xdiffln, ydiffln, dist;
234
235 if (ld.slope == 0)
236 return false;
237
238 xdiffln = xptl - xptn;
239 ydiffln = yptl - yptn;
240 dist = sqrt(pow2(xdiffln) + pow2(ydiffln));
241
242 if (dist == 0.0)
243 return false;
244
245 // calculate angles on horizontal coordinate system
246 Angle = atan2(xdiffln, ydiffln);
247 if (Angle < 0.0)
248 Angle += 2.0 * PI;
249 Theta = ld.aspectf - Angle;
250
251 // calculate angles on local surface coordinate system
252 Theta = -atan2(cos(Theta) / cos(slopef), sin(Theta)) + PI / 2.0;
253 //Angle=ld.aspectf-Theta;
254 //if(Angle<0.0)
255 // Angle+=2.0*PI;
256
257 ds = dist * cos(Theta) * (1.0 - cos(slopef));
258
259 // if(quadrants 1 and 3
260 if ((ld.aspectf <= PI / 2.0) ||
261 (ld.aspectf > PI && ld.aspectf <= 3.0 * PI / 2.0))
262 {
263 if (cos(Angle) != 0.0 || Angle == PI)
264 {
265 if (tan(Angle) < 0.0)
266 {
267 if (tan(Angle) < tan(ld.aspectf - PI / 2.0))
268 My = 1.0;
269 else
270 Mx = 1.0;
271 }
272 }
273 }
274 else // if quadrants 2 and 3
275 {
276 if (cos(Angle) != 0.0)
277 {
278 if (tan(Angle) > 0.0 && Angle != PI)
279 {
280 if (tan(Angle) > tan(ld.aspectf - PI / 2.0))
281 My = 1.0;
282 else
283 Mx = 1.0;
284 }
285 }
286 }
287
288 // transform x and y components to local surface coordinates
289 hx = fabs(xdiffln) - Mx * fabs(ds * sin(ld.aspectf));
290 hy = fabs(ydiffln) - My * fabs(ds * cos(ld.aspectf));
291
292 //determine signs of horiz x and y components
293 if (xdiff < 0.0)
294 XSign = -1.0;
295 if (ydiff < 0.0)
296 YSign = -1.0;
297
298 // substitute surface for horizontal components
299 xdiff = hx * XSign;
300 ydiff = hy * YSign;
301
302 return true;
303}

317void MechCalls::AccelSurf1()
318{
319 // does first acceleration of fire, computes Ros at T+1 given time step
320
321 if (RosT < 0.0) // if negative ros (initial fraction of equilibrium ROS at input only)
322 RosT *= (-vecros);
323 if (xt == 0.0 && yt == 0.0)
324 vecros = RosT;
325 else
326 SpreadCorrect(); // vecros=equilibrium spread rate in vectored direction
327
328 if (RosT < vecros && AccelerationON()) // if RosT is < equilibrium, then accelerate
329 {
330 cosslope = 99.0; // RosT1 is directional ROS with accelleration
331 RosT1 = accel(vecros, RosT, A, &avgros, &cosslope);
332 double SpreadRatio = avgros / vecros; // ratio of accel to equilibrium
333 xt *= SpreadRatio; // in spread direction
334 yt *= SpreadRatio;
335 } // avgros is actual ROS averaged over time-step and spread distance
336 else // else if new ROSEq is < RosT then reset new Ros to current equilibrium
337 RosT1 = avgros = vecros; // ROS after current time step is at steady state
338}

411void MechCalls::SlopeCorrect(int SurfaceSpread)
412{
413 // calls slope correction for spread rate
414 double RosReduct = 1.0; // spread rate reduction only for surface fire
415
416 if (SurfaceSpread) // if surface fire spread, subst. midx,y for x,ypts
417 {
418 midx = xpt; // not for crownfire spread
419 midy = ypt;
420 RosReduct = GetRosRed(ld.fuel);
421 }
422
423 ActualSurfaceSpread = pow2(xt) + pow2(yt);
424 if (ActualSurfaceSpread > 0.0)// Expansion Rate
425 {
426 ActualSurfaceSpread = sqrt(ActualSurfaceSpread);
427 if (ld.slope > 0)
428 {
429 scorrect(ld.slope, ld.aspectf); // CORRECTS ROS FOR SLOPE to get horizontal ROS
430 HorizSpread = sqrt(pow2(xt) + pow2(yt)); // gets Slope and Aspectf from FireEnvironment data members
431 cosslope = HorizSpread / ActualSurfaceSpread; // cosine of slope in spread direction
432 }
433 else
434 cosslope = 1.0; // slope is 0 then cosine is 1.0
435 xpt = midx - (timerem * xt * RosReduct * MetricResolutionConvert()); // computes new horizontal xpt
436 ypt = midy - (timerem * yt * RosReduct * MetricResolutionConvert()); // and ypt coordinates
437 }

539 ypt = GetPerimeter2Value(0, YCOORD);

541 for (i = 1; i <= NumPtTotal; i++)

544 {
545 xptn = GetPerimeter2Value(i, XCOORD);
546 yptn = GetPerimeter2Value(i, YCOORD);
547 flin = GetPerimeter2Value(i, FLIVAL);

551 xptn = GetPerimeter2Value(0, XCOORD);
552 yptn = GetPerimeter2Value(0, YCOORD);
553 flin = GetPerimeter2Value(0, FLIVAL);
554 }
555 Dist = sqrt(pow2(xpt - xptn) + pow2(ypt - yptn));
556 FirePerimeterDist += Dist;
557 if (fli >= 0.0 || flin >= 0.0)

559 ActivePerimDist += Dist;

561 }

577 SetDynamicDistRes(NewMinDist); // set Dynamic min dist to new dist
578 }
579 }
580 else
581 NumPtTotal = -1; // for debugging
582}

586void Mechanix::grow(double ivecdir)
587{
588 // grows fire perimeter points using Richards' equation
589 double part1, part2, part3, part4, part5, part6;
590 double f2, h2;
591
592 if (xdiff == 0 && ydiff == 0)
593 {
594 xt = 0.0;
595 yt = 0.0;
596 }
597 else
598 {
599 f2 = pow2(flank);
600 h2 = pow2(head);
601 part1 = f2 * cos(ivecdir) * (xdiff * sin(ivecdir) +
602 ydiff * cos(ivecdir));
603 part2 = h2 * sin(ivecdir) * (xdiff * cos(ivecdir) -
604 ydiff * sin(ivecdir));
605 part3 = h2 * pow2((xdiff * cos(ivecdir) - ydiff * sin(ivecdir)));
606 part4 = f2 * pow2((xdiff * sin(ivecdir) + ydiff * cos(ivecdir)));
607 part5 = f2 * sin(ivecdir) * (xdiff * sin(ivecdir) +
608 ydiff * cos(ivecdir));
609 part6 = h2 * cos(ivecdir) * (xdiff * cos(ivecdir) -
610 ydiff * sin(ivecdir));
611 xt = ((part1 - part2) / sqrt((part3 + part4))) + back * sin(ivecdir);
612 yt = ((-part5 - part6) / sqrt((part3 + part4))) + back * cos(ivecdir);
613 }
614}

618void Mechanix::limgrow()
619{
620 // limits distance growth to distance check and decrements remaining time step
621 double fdist, dist, distx, disty, MINDIST, step;
622
623 MINDIST = GetDynamicDistRes() * MetricResolutionConvert();
624 step = timerem;
625 distx = midx - xpt;
626 disty = midy - ypt;
627 dist = sqrt(pow2(distx) + pow2(disty)); // computes horizontal distance traveled
628 if (dist > MINDIST)
629 {
630 fdist = MINDIST / dist;
631 xpt = midx - distx * fdist; // new xpt and new ypt computed
632 ypt = midy - disty * fdist; // from mindist distance along spread direction
633 timerem = step - (MINDIST / (dist / step));// remainder of time step, distance traveled/ros ONLY IF CONSTANT ROS
634 HorizSpread = GetDynamicDistRes(); // horizontal spread now equals the distance check
635 }
636 else
637 timerem = 0.0; // no time in timestep remaining
638 if (xpt <= GetLoEast()) // keeps fire on visible landscape
639 {
640 xpt = midx; // keep old point to prevent coincident points on edge of landscape
641 timerem = -1.0;
642 }
643 else if (xpt >= GetHiEast())
644 {
645 xpt = midx;
646 timerem = -1.0;
647 }
648 if (ypt <= GetLoNorth())
649 {
650 ypt = midy;
651 timerem = -1.0;
652 }
653 else if (ypt >= GetHiNorth())
654 {
655 ypt = midy;
656 timerem = -1.0;
657 }
658}

661double Mechanix::accel(double RosE, double RosT, double A, double* avgros,
662 double* cosslope)
663{
664 // ROS acceleration from point source,VanWagner's (McAlpine and Wakimoto 1991)
665 double RosT1, Tt, Dt, Dt1;
666
667 Tt = 0.0;
668 Dt = 0.0;
669 if (RosT != 0.0)
670 {
671 Tt = log(1.0 - RosT / RosE) / (-A); // time to reach current RosT given the equilibrium rate RosE
672 Dt = RosE * (Tt + exp(-A * Tt) / A - 1.0 / A); // distance traveled to achieve current ROS,
673 } // Alexander et al. 1992 eq. 71
674 if (*cosslope == 99.0) // forward distance traveled to eventual forward ROS timestep
675 {
676 Dt1 = RosE * (timerem + Tt + exp(-A * (timerem + Tt)) / A - 1.0 / A);
677 *avgros = (Dt1 - Dt) / (timerem); // average rate of spread over forward distance Dt1-Dt
678 RosT1 = RosE * (1.0 - exp(-A * (timerem + Tt))); // eventual forward ROS at end of timestep
679 *cosslope = 0.0;
680 }
681 else
682 {
683 double t, f1, f2, B = RosE / A;
684 double LimSpread = GetDynamicDistRes() / (*cosslope); // convert horizontal distance check to slope distance equivalent
685 //double LimForwardSpread=(head*timerem+back*timerem)*
686 // LimSpread/(ActualSurfaceSpread*timerem);
687 // LimForwardSpread is the max spread in forward direction proportional to
688 if (vecros > 0.0 && ExpansionRate > 0.0)
689 LimSpread *= vecros / ExpansionRate;
690
691 if (LimSpread < B) // the Limited spread in actual direction based on elliptical dimensions
692 t = sqrt((2.0 * LimSpread) / B); // if LimSpread is "small"
693 else
694 t = LimSpread / RosE; // if LimSpread is "large"
695 do // travel time to end of distance check
696 {
697 B = exp(-A * t); // iterate using Newton's Method
698 f1 = RosE * (t + B / A - 1.0 / A) - (LimSpread + Dt); // total forward distance traveled to reach RosE
699 f2 = RosE * (1.0 - B);
700 t -= f1 / f2;
701 }
702 while (fabs(f1) > 1e-4); // tolerance for convergence of iteration
703 *avgros = LimSpread / (t - Tt); // average surface ROS over distance in subtimestep in spread direction (not forward)
704 RosT1 = RosE * (1.0 - exp(-A * t)); // eventual directional ROS at end of subtimestep
705 timerem = timerem - (t - Tt); // decrement time remaining in timestep
706 if (timerem < 0.0)
707 timerem = 0.0;
708 *cosslope = 0.0; // change cosslope to 0
709 }
710
711 return fabs(RosT1);
712}

715void Mechanix::scorrect(long slope, double aspectf)
716{
717 double Theta;
718 double Angle;
719 double hxt, hyt, ds;
720 double XSign = 1.0, YSign = 1.0;
721 double Mx = -1.0, My = -1.0;
722 double slopef = ((double) slope / 180.0) * PI;
723 //double PI=3.14159265358979324;
724
725 // compute angles of spread direction in surface plane
726 Angle = atan2(-xt, -yt);
727 if (Angle < 0.0)
728 Angle += 2.0 * PI;
729 Theta = aspectf - Angle;
730 if (fabs(Theta) < 0.0000000001) // precision problem
731 Theta = 0;
732
733 // compute horizontal spread differential
734 ds = ActualSurfaceSpread * cos(Theta) * (1.0 - cos(slopef));
735
736 // if quadrants 1 and 3
737 if ((aspectf <= PI / 2.0) || (aspectf > PI && aspectf <= 3.0 * PI / 2.0))
738 {
739 if (cos(Angle) != 0.0 || Angle == PI)
740 {
741 if (tan(Angle) < 0)
742 {
743 if (tan(Angle) < tan(aspectf - PI / 2.0))
744 My = 1.0;
745 else
746 Mx = 1.0;
747 }
748 }
749 }
750 else // if quadrants 2 and 4
751 {
752 if (cos(Angle) != 0.0)
753 {
754 if (tan(Angle) > 0 && Angle != PI)
755 {
756 if (tan(Angle) > tan(aspectf - PI / 2.0))
757 My = 1.0;
758 else
759 Mx = 1.0;
760 }
761 }
762 }
763
764 // transfrom spread components in surface coords to horizontal coords
765 hxt = fabs(xt) + Mx * fabs(ds * sin(aspectf));
766 hyt = fabs(yt) + My * fabs(ds * cos(aspectf));
767
768 //determine signs of horiz x and y components
769 if (xt < 0.0)
770 XSign = -1.0;
771 if (yt < 0.0)
772 YSign = -1.0;
773
774 xt = hxt * XSign;
775 yt = hyt * YSign;
776}
